631 TYPVAL_ENCODE_SCOPE int _TYPVAL_ENCODE_ENCODE(
632  TYPVAL_ENCODE_FIRST_ARG_TYPE TYPVAL_ENCODE_FIRST_ARG_NAME,
633  typval_T *const top_tv, const char *const objname)
634 {
635  const int copyID = get_copyID();
636  MPConvStack mpstack;
637  _mp_init(mpstack);
638  if (_TYPVAL_ENCODE_CONVERT_ONE_VALUE(TYPVAL_ENCODE_FIRST_ARG_NAME, &mpstack,
639  NULL,
640  top_tv, copyID, objname)
641  == FAIL) {
642  goto encode_vim_to__error_ret;
643  }
646 typval_encode_stop_converting_one_item:
647  while (_mp_size(mpstack)) {
648  MPConvStackVal *cur_mpsv = &_mp_last(mpstack);
649  typval_T *tv = NULL;
650  switch (cur_mpsv->type) {
651  case kMPConvDict: {
652  if (!cur_mpsv->data.d.todo) {
653  (void)_mp_pop(mpstack);
654  cur_mpsv->data.d.dict->dv_copyID = cur_mpsv->saved_copyID;
655  TYPVAL_ENCODE_CONV_DICT_END(cur_mpsv->tv, *cur_mpsv->data.d.dictp);
656  continue;
657  } else if (cur_mpsv->data.d.todo
658  != cur_mpsv->data.d.dict->dv_hashtab.ht_used) {
659  TYPVAL_ENCODE_CONV_DICT_BETWEEN_ITEMS(cur_mpsv->tv,
660  *cur_mpsv->data.d.dictp);
661  }
662  while (HASHITEM_EMPTY(cur_mpsv->data.d.hi)) {
663  cur_mpsv->data.d.hi++;
664  }
665  dictitem_T *const di = HI2DI(cur_mpsv->data.d.hi);
666  cur_mpsv->data.d.todo--;
667  cur_mpsv->data.d.hi++;
668  TYPVAL_ENCODE_CONV_STR_STRING(NULL, &di->di_key[0],
669  strlen((char *)&di->di_key[0]));
670  TYPVAL_ENCODE_CONV_DICT_AFTER_KEY(cur_mpsv->tv,
671  *cur_mpsv->data.d.dictp);
672  tv = &di->di_tv;
673  break;
674  }
675  case kMPConvList: {
676  if (cur_mpsv->data.l.li == NULL) {
677  (void)_mp_pop(mpstack);
678  cur_mpsv->data.l.list->lv_copyID = cur_mpsv->saved_copyID;
679  TYPVAL_ENCODE_CONV_LIST_END(cur_mpsv->tv);
680  continue;
681  } else if (cur_mpsv->data.l.li != cur_mpsv->data.l.list->lv_first) {
682  TYPVAL_ENCODE_CONV_LIST_BETWEEN_ITEMS(cur_mpsv->tv);
683  }
684  tv = &cur_mpsv->data.l.li->li_tv;
685  cur_mpsv->data.l.li = cur_mpsv->data.l.li->li_next;
686  break;
687  }
688  case kMPConvPairs: {
689  if (cur_mpsv->data.l.li == NULL) {
690  (void)_mp_pop(mpstack);
691  cur_mpsv->data.l.list->lv_copyID = cur_mpsv->saved_copyID;
692  TYPVAL_ENCODE_CONV_DICT_END(cur_mpsv->tv, TYPVAL_ENCODE_NODICT_VAR);
693  continue;
694  } else if (cur_mpsv->data.l.li != cur_mpsv->data.l.list->lv_first) {
695  TYPVAL_ENCODE_CONV_DICT_BETWEEN_ITEMS(
696  cur_mpsv->tv, TYPVAL_ENCODE_NODICT_VAR);
697  }
698  const list_T *const kv_pair = cur_mpsv->data.l.li->li_tv.vval.v_list;
699  TYPVAL_ENCODE_SPECIAL_DICT_KEY_CHECK(
700  encode_vim_to__error_ret, kv_pair->lv_first->li_tv);
701  if (_TYPVAL_ENCODE_CONVERT_ONE_VALUE(TYPVAL_ENCODE_FIRST_ARG_NAME,
702  &mpstack, cur_mpsv,
703  &kv_pair->lv_first->li_tv,
704  copyID,
705  objname) == FAIL) {
706  goto encode_vim_to__error_ret;
707  }
708  TYPVAL_ENCODE_CONV_DICT_AFTER_KEY(cur_mpsv->tv,
709  TYPVAL_ENCODE_NODICT_VAR);
710  tv = &kv_pair->lv_last->li_tv;
711  cur_mpsv->data.l.li = cur_mpsv->data.l.li->li_next;
712  break;
713  }
714  case kMPConvPartial: {
715  partial_T *const pt = cur_mpsv->data.p.pt;
716  tv = cur_mpsv->tv;
717  switch (cur_mpsv->data.p.stage) {
718  case kMPConvPartialArgs: {
719  TYPVAL_ENCODE_CONV_FUNC_BEFORE_ARGS(tv,
720  pt == NULL ? 0 : pt->pt_argc);
721  cur_mpsv->data.p.stage = kMPConvPartialSelf;
722  if (pt != NULL && pt->pt_argc > 0) {
723  TYPVAL_ENCODE_CONV_LIST_START(NULL, pt->pt_argc);
724  _mp_push(mpstack, ((MPConvStackVal) {
725  .type = kMPConvPartialList,
726  .tv = NULL,
727  .saved_copyID = copyID - 1,
728  .data = {
729  .a = {
730  .arg = pt->pt_argv,
731  .argv = pt->pt_argv,
732  .todo = (size_t)pt->pt_argc,
733  },
734  },
735  }));
736  }
737  break;
738  }
739  case kMPConvPartialSelf: {
740  cur_mpsv->data.p.stage = kMPConvPartialEnd;
741  dict_T *const dict = pt == NULL ? NULL : pt->pt_dict;
742  if (dict != NULL) {
743  TYPVAL_ENCODE_CONV_FUNC_BEFORE_SELF(tv, dict->dv_hashtab.ht_used);
744  if (dict->dv_hashtab.ht_used == 0) {
745  TYPVAL_ENCODE_CONV_EMPTY_DICT(NULL, pt->pt_dict);
746  continue;
747  }
748  const int saved_copyID = dict->dv_copyID;
749  const int te_csr_ret = _TYPVAL_ENCODE_CHECK_SELF_REFERENCE(
750  TYPVAL_ENCODE_FIRST_ARG_NAME,
751  dict, &dict->dv_copyID, &mpstack, copyID, kMPConvDict,
752  objname);
753  if (te_csr_ret != NOTDONE) {
754  if (te_csr_ret == FAIL) {
755  goto encode_vim_to__error_ret;
756  } else {
757  continue;
758  }
759  }
760  TYPVAL_ENCODE_CONV_DICT_START(NULL, pt->pt_dict,
761  dict->dv_hashtab.ht_used);
762  assert(saved_copyID != copyID && saved_copyID != copyID - 1);
763  _mp_push(mpstack, ((MPConvStackVal) {
764  .type = kMPConvDict,
765  .tv = NULL,
766  .saved_copyID = saved_copyID,
767  .data = {
768  .d = {
769  .dict = dict,
770  .dictp = &pt->pt_dict,
771  .hi = dict->dv_hashtab.ht_array,
772  .todo = dict->dv_hashtab.ht_used,
773  },
774  },
775  }));
776  TYPVAL_ENCODE_CONV_REAL_DICT_AFTER_START(NULL, pt->pt_dict,
777  _mp_last(mpstack));
778  } else {
779  TYPVAL_ENCODE_CONV_FUNC_BEFORE_SELF(tv, -1);
780  }
781  break;
782  }
783  case kMPConvPartialEnd: {
784  TYPVAL_ENCODE_CONV_FUNC_END(tv);
785  (void)_mp_pop(mpstack);
786  break;
787  }
788  }
789  continue;
790  }
791  case kMPConvPartialList: {
