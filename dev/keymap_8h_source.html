17 /*
18  * For MSDOS some keys produce codes larger than 0xff. They are split into two
19  * chars, the first one is K_NUL (same value used in term_defs.h).
20  */
21 #define K_NUL (0xce) /* for MSDOS: special key follows */
22 
23 /*
24  * K_SPECIAL is the first byte of a special key code and is always followed by
25  * two bytes.
26  * The second byte can have any value. ASCII is used for normal termcap
27  * entries, 0x80 and higher for special keys, see below.
28  * The third byte is guaranteed to be between 0x02 and 0x7f.
29  */
30 
31 #define K_SPECIAL (0x80)
32 
33 /*
34  * Positive characters are "normal" characters.
35  * Negative characters are special key codes. Only characters below -0x200
36  * are used to so that the absolute value can't be mistaken for a single-byte
37  * character.
38  */
39 #define IS_SPECIAL(c) ((c) < 0)
40 
41 /*
42  * Characters 0x0100 - 0x01ff have a special meaning for abbreviations.
43  * Multi-byte characters also have ABBR_OFF added, thus are above 0x0200.
44  */
45 #define ABBR_OFF 0x100

117 /*
118  * translation of three byte code "K_SPECIAL a b" into int "K_xxx" and back
119  */
120 #define TERMCAP2KEY(a, b) (-((a) + ((int)(b) << 8)))
121 #define KEY2TERMCAP0(x) ((-(x)) & 0xff)
122 #define KEY2TERMCAP1(x) (((unsigned)(-(x)) >> 8) & 0xff)

124 /*
125  * get second or third byte when translating special key code into three bytes
126  */
127 #define K_SECOND(c) ((c) == K_SPECIAL ? KS_SPECIAL : (c) == \
128  NUL ? KS_ZERO : KEY2TERMCAP0(c))
129 
130 #define K_THIRD(c) (((c) == K_SPECIAL || (c) == \
131  NUL) ? KE_FILLER : KEY2TERMCAP1(c))

137  KS_ZERO ? K_ZERO : TERMCAP2KEY(a, b))
