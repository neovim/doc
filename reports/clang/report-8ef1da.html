219/*
220 * Reverse text into allocated memory.
221 * Returns the allocated string.
222 *
223 * TODO(philix): move reverse_text() to strings.c
224 */
225char_u *reverse_text(char_u *s) FUNC_ATTR_NONNULL_RET
226{
227 /*
228 * Reverse the pattern.
229 */
230 size_t len = STRLEN(s)strlen((char *)(s));
231 char_u *rev = xmalloc(len + 1);
232 size_t rev_i = len;
233 for (size_t s_i = 0; s_i < len; ++s_i) {
234 if (has_mbyte) {
235 int mb_len = (*mb_ptr2len)(s + s_i);
236 rev_i -= mb_len;
237 memmove(rev + rev_i, s + s_i, mb_len);
238 s_i += mb_len - 1;
239 } else
240 rev[--rev_i] = s[s_i];
241 }
242 rev[len] = NUL'\000';
243
244 return rev;
245}

327/*
328 * Return TRUE if patter "pat" has an uppercase character.
329 */
330int pat_has_uppercase(char_u *pat)
331{
332 char_u *p = pat;
333
334 while (*p != NUL'\000') {
335 int l;
336
337 if (has_mbyte && (l = (*mb_ptr2len)(p)) > 1) {
338 if (enc_utf8 && utf_isupper(utf_ptr2char(p)))
339 return TRUE1;
340 p += l;
341 } else if (*p == '\\') {
342 if (p[1] == '_' && p[2] != NUL'\000') /* skip "\_X" */
343 p += 3;
344 else if (p[1] == '%' && p[2] != NUL'\000') /* skip "\%X" */
345 p += 3;
346 else if (p[1] != NUL'\000') /* skip "\X" */
347 p += 2;
348 else
349 p += 1;
350 } else if (vim_isupper(*p))
351 return TRUE1;

894 long count,

901 struct soffset old_off;
902 int retval; /* Return value */
903 char_u *p;
904 long c;
905 char_u *dircp;

908
909 /*
910 * A line offset is not remembered, this is vi compatible.
911 */

913 spats[0].off.line = FALSE0;

918 * Save the values for when (options & SEARCH_KEEP) is used.
919 * (there is no "if ()" around this because gcc wants them initialized)
920 */

923 pos = curwin->w_cursor; /* start searching at the cursor position */

925 /*
926 * Find out the direction of the search.
927 */

932 set_vv_searchforward();

999 spats[0].off.end = FALSE0;
1000 spats[0].off.off = 0;

1005 * substitute command.

1010 (*p == 'e' || *p == 's' || *p == 'b')) {
1011 if (*p == 'e') /* end */
1012 spats[0].off.end = SEARCH_END0x40;
1013 ++p;
1014 }
1015 if (VIM_ISDIGIT(*p)((unsigned)(*p) - '0' < 10) || *p == '+' || *p == '-') { /* got an offset */
1016 /* 'nr' or '+nr' or '-nr' */
1017 if (VIM_ISDIGIT(*p)((unsigned)(*p) - '0' < 10) || VIM_ISDIGIT(*(p + 1))((unsigned)(*(p + 1)) - '0' < 10))
1018 spats[0].off.off = atol((char *)p);

1021 else /* single '+' */
1022 spats[0].off.off = 1;

1025 ++p;
1026 }

1028 /* compute length of search command for get_address() */
1029 searchcmdlen += (int)(p - pat);
1030

1035 && !cmd_silent && msg_silent == 0) {

1096 * position, so we don't get stuck at "?pat?e+2" or "/pat/s-2".
1097 * Skip this if pos.col is near MAXCOL (closed fold).
1098 * This is not done for a line offset, because then we would not be vi
1099 * compatible.
1100 */
1101 if (!spats[0].off.line && spats[0].off.off && pos.col < MAXCOL(0x7fffffffL) - 2) {

1199 curwin->w_set_curswant = TRUE1;
1200

1204 free(strcopy);
1205

1207}

1213 * white-space), starting from pos and going in direction dir. pos will
1214 * contain the position of the match found. Blank lines match only if

1234 pos->lnum = 1;
