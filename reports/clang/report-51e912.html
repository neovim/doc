13/*
14 * The buffer list is a double linked list of all buffers.
15 * Each buffer can be in one of these states:
16 * never loaded: BF_NEVERLOADED is set, only the file name is valid
17 * not loaded: b_ml.ml_mfp == NULL, no memfile allocated
18 * hidden: b_nwindows == 0, loaded but not displayed in a window
19 * normal: loaded and displayed in a window
20 *
21 * Instead of storing file names all over the place, each file name is
22 * stored in the buffer list. It can be referenced by a number.
23 *
24 * The current implementation remembers all file names ever used.

1694 * Return < 0 for error.
1695 */

1697buflist_findpat (

1700 int unlisted, /* find unlisted buffers */
1701 int diffmode, /* find diff-mode buffers only */
1702 int curtab_only /* find buffers in current tab only */
1703)

1723 * Try four ways of matching a listed buffer:
1724 * attempt == 0: without '^' or '$' (at any position)
1725 * attempt == 1: with '^' at start (only at position 0)
1726 * attempt == 2: with '$' at end (only match at end)
1727 * attempt == 3: with '^' at start and '$' at end (only full match)
1728 * Repeat this for finding an unlisted buffer if there was no matching
1729 * listed buffer.

1750 if (prog == NULL((void*)0)) {

tp_firstwin; wp != ((void*)0); wp = wp->w_next) {

1765 found_window = true1;
1766 break;
1767 }
1768 }
1769 if (!found_window) {
1770 continue;
1771 }
1772 }
1773 if (match >= 0) { /* already found a match */
1774 match = -2;
1775 break;
1776 }
1777 match = buf->b_fnum; /* remember first match */
1778 }
1779 }
1780
1781 vim_regfree(prog);
1782 if (match >= 0) /* found one match */
1783 break;
1784 }
1785
1786 /* Only search for unlisted buffers if there was no match with
1787 * a listed buffer. */
1788 if (!unlisted || !find_listed || match != -1)
1789 break;
1790 find_listed = FALSE0;

1793 free(pat);
1794 }

1796 if (match == -2)
1797 EMSG2(_("E93: More than one match for %s"), pattern)emsg2((char_u *)(gettext((char *)("E93: More than one match for %s"
))), (char_u *)(pattern));
1798 else if (match < 0)
1799 EMSG2(_("E94: No matching buffer for %s"), pattern)emsg2((char_u *)(gettext((char *)("E94: No matching buffer for %s"
))), (char_u *)(pattern));
1800 return match;
