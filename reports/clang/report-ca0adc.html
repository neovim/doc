1623 // after writing the file. That's because only the meta-data is
1624 // journalled. Syncing the file slows down the system, but assures it has
1625 // been written to disk and we don't lose it.
1626 // For a device do try the fsync() but don't complain if it does not work
1627 // (could be a pipe).
1628 // If the 'fsync' option is false, don't fsync(). Useful for laptops.
1629 int error;
1630 if (p_fs && (error = os_fsync(fd)) != 0 && !device
1631 // fsync not supported on this storage.
1632 && error != UV_ENOTSUP) {
1633 err = set_err_arg(e_fsync, error);
1634 end = 0;
1635 }
1636
1637#ifdef UNIX
1638 // When creating a new file, set its owner/group to that of the original
1639 // file. Get the new device and inode number.
1640 if (backup != NULL((void*)0) && !backup_copy) {
1641 // don't change the owner when it's already OK, some systems remove
1642 // permission or ACL stuff
1643 FileInfo file_info;
1644 if (!os_fileinfo(wfname, &file_info)
1645 || file_info.stat.st_uid != file_info_old.stat.st_uid
1646 || file_info.stat.st_gid != file_info_old.stat.st_gid) {
1647 os_fchown(fd, (uv_uid_t)file_info_old.stat.st_uid, (uv_gid_t)file_info_old.stat.st_gid);
1648 if (perm >= 0) { // Set permission again, may have changed.
1649 (void)os_setperm(wfname, (int)perm);
1650 }
1651 }
1652 buf_set_file_id(buf);
1653 } else if (!buf->file_id_valid) {
1654 // Set the file_id when creating a new file.
1655 buf_set_file_id(buf);
1656 }
1657#endif
1658
1659 if ((error = os_close(fd)) != 0) {
1660 err = set_err_arg(_("E512: Close failed: %s")gettext("E512: Close failed: %s"), error);
1661 end = 0;
1662 }
1663
1664#ifdef UNIX
1665 if (made_writable) {
1666 perm &= ~0200; // reset 'w' bit for security reasons
1667 }
1668#endif
1669 if (perm >= 0) { // Set perm. of new file same as old file.
1670 (void)os_setperm(wfname, (int)perm);
1671 }
1672 // Probably need to set the ACL before changing the user (can't set the
1673 // ACL on a file the user doesn't own).
1674 if (!backup_copy) {
1675 os_set_acl(wfname, acl);
1676 }
1677
1678 if (wfname != fname) {
1679 // The file was written to a temp file, now it needs to be converted
1680 // with 'charconvert' to (overwrite) the output file.
1681 if (end != 0) {
1682 if (eval_charconvert("utf-8", fenc, wfname, fname) == FAIL0) {
1683 write_info.bw_conv_error = true1;
1684 end = 0;
1685 }
1686 }
1687 os_remove(wfname);
1688 xfree(wfname);
1689 }
1690 }
1691
1692 if (end == 0) {
1693 // Error encountered.
1694 if (err.msg == NULL((void*)0)) {
1695 if (write_info.bw_conv_error) {
1696 if (write_info.bw_conv_error_lnum == 0) {
1697 err = set_err(_(e_write_error_conversion_failed_make_fenc_empty_to_override)gettext(e_write_error_conversion_failed_make_fenc_empty_to_override
));
1698 } else {
1699 err = set_err(xmalloc(300));
1700 err.alloc = true1;
1701 vim_snprintf(err.msg, 300, // NOLINT(runtime/printf)
1702 _(e_write_error_conversion_failed_in_line_nr_make_fenc_empty_to_override)gettext(e_write_error_conversion_failed_in_line_nr_make_fenc_empty_to_override
),
1703 write_info.bw_conv_error_lnum);
1704 }
1705 } else if (got_int) {
1706 err = set_err(_(e_interr)gettext(e_interr));
1707 } else {
1708 err = set_err(_(e_write_error_file_system_full)gettext(e_write_error_file_system_full));
1709 }
1710 }
1711
1712 // If we have a backup file, try to put it in place of the new file,
1713 // because the new file is probably corrupt. This avoids losing the
1714 // original file when trying to make a backup when writing the file a
1715 // second time.
1716 // When "backup_copy" is set we need to copy the backup over the new
1717 // file. Otherwise rename the backup file.
1718 // If this is OK, don't give the extra warning message.
1719 if (backup != NULL((void*)0)) {
1720 if (backup_copy) {
1721 // This may take a while, if we were interrupted let the user
1722 // know we got the message.
1723 if (got_int) {
1724 msg(_(e_interr)gettext(e_interr));
1725 ui_flush();
1726 }
1727
1728 // copy the file.
1729 if (os_copy(backup, fname, UV_FS_COPYFILE_FICLONE0x0002)
1730 == 0) {
1731 end = 1; // success
1732 }
1733 } else {
1734 if (vim_rename(backup, fname) == 0) {
1735 end = 1;
1736 }
1737 }
1738 }
1739 goto fail;
1740 }
1741
1742 lnum -= start; // compute number of written lines
1743 no_wait_return--; // may wait for return now
1744
1745#if !defined(UNIX)
1746 fname = sfname; // use shortname now, for the messages
1747#endif
1748 if (!filtering) {
1749 add_quoted_fname(IObuff, IOSIZE(1024 + 1), buf, fname);
1750 bool_Bool insert_space = false0;
1751 if (write_info.bw_conv_error) {
1752 xstrlcat(IObuff, _(" CONVERSION ERROR")gettext(" CONVERSION ERROR"), IOSIZE(1024 + 1));
1753 insert_space = true1;
1754 if (write_info.bw_conv_error_lnum != 0) {
1755 vim_snprintf_add(IObuff, IOSIZE(1024 + 1), _(" in line %" PRId64 ";")gettext(" in line %" "l" "d" ";"),
1756 (int64_t)write_info.bw_conv_error_lnum);
1757 }
1758 } else if (notconverted) {
1759 xstrlcat(IObuff, _("[NOT converted]")gettext("[NOT converted]"), IOSIZE(1024 + 1));
1760 insert_space = true1;
1761 } else if (converted) {
1762 xstrlcat(IObuff, _("[converted]")gettext("[converted]"), IOSIZE(1024 + 1));
1763 insert_space = true1;
1764 }
1765 if (device) {
1766 xstrlcat(IObuff, _("[Device]")gettext("[Device]"), IOSIZE(1024 + 1));
1767 insert_space = true1;
1768 } else if (newfile) {
1769 xstrlcat(IObuff, new_file_message(), IOSIZE(1024 + 1));
1770 insert_space = true1;
1771 }
1772 if (no_eol) {
1773 msg_add_eol();
1774 insert_space = true1;
1775 }
1776 // may add [unix/dos/mac]
1777 if (msg_add_fileformat(fileformat)) {
1778 insert_space = true1;
1779 }
1780 msg_add_lines(insert_space, (long)lnum, nchars); // add line/char count
1781 if (!shortmess(SHM_WRITE)) {
1782 if (append) {
1783 xstrlcat(IObuff, shortmess(SHM_WRI) ? _(" [a]")gettext(" [a]") : _(" appended")gettext(" appended"), IOSIZE(1024 + 1));
1784 } else {
1785 xstrlcat(IObuff, shortmess(SHM_WRI) ? _(" [w]")gettext(" [w]") : _(" written")gettext(" written"), IOSIZE(1024 + 1));
1786 }
1787 }
1788
1789 set_keep_msg(msg_trunc_attr(IObuff, false0, 0), 0);
1790 }
1791
1792 // When written everything correctly: reset 'modified'. Unless not
1793 // writing to the original file and '+' is not in 'cpoptions'.
1794 if (reset_changed && whole && !append
1795 && !write_info.bw_conv_error
1796 && (overwriting || vim_strchr(p_cpo, CPO_PLUS'+') != NULL((void*)0))) {
1797 unchanged(buf, true1, false0);
1798 const varnumber_T changedtick = buf_get_changedtick(buf);
1799 if (buf->b_last_changedtick + 1 == changedtick) {
1800 // b:changedtick may be incremented in unchanged() but that
1801 // should not trigger a TextChanged event.
1802 buf->b_last_changedtick = changedtick;
1803 }
1804 u_unchanged(buf);
1805 u_update_save_nr(buf);
1806 }
1807
1808 // If written to the current file, update the timestamp of the swap file
1809 // and reset the BF_WRITE_MASK flags. Also sets buf->b_mtime.
1810 if (overwriting) {
1811 ml_timestamp(buf);
1812 if (append) {
1813 buf->b_flags &= ~BF_NEW0x10;
1814 } else {
1815 buf->b_flags &= ~BF_WRITE_MASK(0x08 + 0x10 + 0x40);
1816 }
1817 }
1818
1819 // If we kept a backup until now, and we are in patch mode, then we make
1820 // the backup file our 'original' file.
1821 if (*p_pm && dobackup) {
1822 char *const org = modname(fname, p_pm, false0);
1823
1824 if (backup != NULL((void*)0)) {
1825 // If the original file does not exist yet
1826 // the current backup file becomes the original file
1827 if (org == NULL((void*)0)) {
1828 emsg(_("E205: Patchmode: can't save original file")gettext("E205: Patchmode: can't save original file"));
1829 } else if (!os_path_exists(org)) {
1830 vim_rename(backup, org);
1831 XFREE_CLEAR(backup)do { void **ptr_ = (void **)&(backup); xfree(*ptr_); *ptr_
= ((void*)0); (void)(*ptr_); } while (0); // don't delete the file
1832#ifdef UNIX
1833 os_file_settime(org,
1834 (double)file_info_old.stat.st_atim.tv_sec,
1835 (double)file_info_old.stat.st_mtim.tv_sec);
1836#endif
1837 }
1838 } else {
1839 // If there is no backup file, remember that a (new) file was
1840 // created.
1841 int empty_fd;
1842
1843 if (org == NULL((void*)0)
1844 || (empty_fd = os_open(org,
1845 O_CREAT0100 | O_EXCL0200 | O_NOFOLLOW0400000,
1846 perm < 0 ? 0666 : (perm & 0777))) < 0) {
1847 emsg(_(e_patchmode_cant_touch_empty_original_file)gettext(e_patchmode_cant_touch_empty_original_file));
1848 } else {
1849 close(empty_fd);
1850 }
1851 }
1852 if (org != NULL((void*)0)) {
1853 os_setperm(org, os_getperm(fname) & 0777);
1854 xfree(org);
1855 }
1856 }
1857
1858 // Remove the backup unless 'backup' option is set
1859 if (!p_bk && backup != NULL((void*)0)
1860 && !write_info.bw_conv_error
1861 && os_remove(backup) != 0) {
1862 emsg(_("E207: Can't delete backup file")gettext("E207: Can't delete backup file"));
1863 }
1864
1865 goto nofail;
1866
1867 // Finish up. We get here either after failure or success.
1868fail:
1869 no_wait_return--; // may wait for return now
1870nofail:
1871
1872 // Done saving, we accept changed buffer warnings again
1873 buf->b_saving = false0;
1874
1875 xfree(backup);
1876 if (buffer != smallbuf) {
1877 xfree(buffer);
1878 }
1879 xfree(fenc_tofree);
1880 xfree(write_info.bw_conv_buf);
1881 if (write_info.bw_iconv_fd != (iconv_t)-1) {
1882 iconv_close(write_info.bw_iconv_fd);
1883 write_info.bw_iconv_fd = (iconv_t)-1;
1884 }
1885 os_free_acl(acl);
1886
1887 if (err.msg != NULL((void*)0)) {
1888 // - 100 to save some space for further error message
1889#ifndef UNIX
1890 add_quoted_fname(IObuff, IOSIZE(1024 + 1) - 100, buf, sfname);
1891#else
1892 add_quoted_fname(IObuff, IOSIZE(1024 + 1) - 100, buf, fname);
1893#endif
1894 emit_err(&err);
1895
1896 retval = FAIL0;
1897 if (end == 0) {
1898 const int attr = HL_ATTR(HLF_E)hl_attr_active[(int)(HLF_E)]; // Set highlight for error messages.
1899 msg_puts_attr(_("\nWARNING: Original file may be lost or damaged\n")gettext("\nWARNING: Original file may be lost or damaged\n"),
1900 attr | MSG_HIST0x1000);
1901 msg_puts_attr(_("don't quit the editor until the file is successfully written!")gettext("don't quit the editor until the file is successfully written!"
),
1902 attr | MSG_HIST0x1000);
1903
1904 // Update the timestamp to avoid an "overwrite changed file"
1905 // prompt when writing again.
1906 if (os_fileinfo(fname, &file_info_old)) {
1907 buf_store_file_info(buf, &file_info_old);
1908 buf->b_mtime_read = buf->b_mtime;
1909 buf->b_mtime_read_ns = buf->b_mtime_ns;
1910 }
1911 }
1912 }
1913 msg_scroll = msg_save;
1914

1916 // file.
1917 if (retval == OK1 && write_undo_file) {
1918 uint8_t hash[UNDO_HASH_SIZE32];
1919
1920 sha256_finish(&sha_ctx, hash);
1921 u_write_undo(NULL((void*)0), false0, buf, hash);
1922 }
1923
1924 if (!should_abort(retval)) {
1925 buf_write_do_post_autocmds(buf, fname, eap, append, filtering, reset_changed, whole);
1926 if (aborting()) { // autocmds may abort script processing
1927 retval = false0;
1928 }
1929 }
1930
1931 got_int |= prev_got_int;
1932
1933 return retval;
