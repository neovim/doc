248/// The xstpncpy() function shall copy not more than n bytes (bytes that follow
249/// a NUL character are not copied) from the array pointed to by src to the
250/// array pointed to by dst.
251///
252/// If a NUL character is written to the destination, the xstpncpy() function
253/// shall return the address of the first such NUL character. Otherwise, it
254/// shall return &dst[maxlen].
255///
256/// WARNING: If copying takes place between objects that overlap, the behavior is
257/// undefined.
258///
259/// WARNING: xstpncpy will ALWAYS write maxlen bytes. If src is shorter than
260/// maxlen, zeroes will be written to the remaining bytes.
261///
262/// TODO(aktau): I don't see a good reason to have this last behaviour, and
263/// it is potentially wasteful. Could we perhaps deviate from the standard
264/// and not zero the rest of the buffer?
265///
266/// @param dst
267/// @param src
268/// @param maxlen
269char *xstpncpy(char *restrict dst, const char *restrict src, size_t maxlen)
270 FUNC_ATTR_NONNULL_RET FUNC_ATTR_WARN_UNUSED_RESULT FUNC_ATTR_NONNULL_ALL
271{
272 const char *p = memchr(src, '\0', maxlen);
273 if (p) {
274 size_t srclen = (size_t)(p - src);
275 memcpy(dst, src, srclen);
276 memset(dst + srclen, 0, maxlen - srclen);
277 return dst + srclen;
278 } else {

284/// xstrlcpy - Copy a %NUL terminated string into a sized buffer
285///
286/// Compatible with *BSD strlcpy: the result is always a valid
287/// NUL-terminated string that fits in the buffer (unless,
288/// of course, the buffer size is zero). It does not pad
289/// out the result like strncpy() does.
290///
291/// @param dst Where to copy the string to
292/// @param src Where to copy the string from
293/// @param size Size of destination buffer
294/// @return Length of the source string (i.e.: strlen(src))
295size_t xstrlcpy(char *restrict dst, const char *restrict src, size_t size)
296{
297 size_t ret = strlen(src);
298
299 if (size) {
300 size_t len = (ret >= size) ? size - 1 : ret;
301 memcpy(dst, src, len);
302 dst[len] = '\0';
303 }
304
305 return ret;
306}
