34/*
35 * Exception handling terms:
36 *
37 * :try ":try" command \
38 * ... try block |
39 * :catch RE ":catch" command |
40 * ... catch clause |- try conditional
41 * :finally ":finally" command |
42 * ... finally clause |
43 * :endtry ":endtry" command /
44 *
45 * The try conditional may have any number of catch clauses and at most one
46 * finally clause. A ":throw" command can be inside the try block, a catch
47 * clause, the finally clause, or in a function called or script sourced from
48 * there or even outside the try conditional. Try conditionals may be nested.
49 */
50
51/*
52 * Configuration whether an exception is thrown on error or interrupt. When
53 * the preprocessor macros below evaluate to FALSE, an error (did_emsg) or
54 * interrupt (got_int) under an active try conditional terminates the script
55 * after the non-active finally clauses of all active try conditionals have been
56 * executed. Otherwise, errors and/or interrupts are converted into catchable
57 * exceptions (did_throw additionally set), which terminate the script only if
58 * not caught. For user exceptions, only did_throw is set. (Note: got_int can
59 * be set asynchronously afterwards by a SIGINT, so did_throw && got_int is not
60 * a reliant test that the exception currently being thrown is an interrupt
61 * exception. Similarly, did_emsg can be set afterwards on an error in an
62 * (unskipped) conditional command inside an inactive conditional, so did_throw
63 * && did_emsg is not a reliant test that the exception currently being thrown
64 * is an error exception.) - The macros can be defined as expressions checking
65 * for a variable that is allowed to be changed during execution of a script.
66 */

68# define THROW_ON_ERROR1 TRUE1
69# define THROW_ON_ERROR_TRUE

1553 cstack->cs_lflags |= CSL_HAD_FINA8;

1735 *
1736 * Call enter_cleanup() with a pointer to a cleanup_T and pass the same

1743 * actually an extra try block around the part that failed and an error or
1744 * interrupt has not (yet) been converted to an exception. This function

1769 * there is an extra instance for every call of do_cmdline(), anyway.

1790/*
1791 * See comment above enter_cleanup() for how this function is used.
1792 *
1793 * This function is a bit like ex_endtry() except that there was not actually
1794 * an extra try block around the part that failed and an error or interrupt
1795 * had not (yet) been converted to an exception when the cleanup autocommand
1796 * sequence was invoked.
1797 *
1798 * This function has to be called with the address of the cleanup_T structure
1799 * filled by enter_cleanup() as an argument; it restores the error/interrupt/
1800 * exception state saved by that function - except there was an aborting
1801 * error, an interrupt or an uncaught exception during execution of the
1802 * cleanup autocommands. In the latter case, the saved error/interrupt/
1803 * exception state is discarded.
1804 */
1805void leave_cleanup(cleanup_T *csp)
1806{
1807 int pending = csp->pending;
1808
1809 if (pending == CSTP_NONE0) /* nothing to do */
1810 return;
1811
1812 /* If there was an aborting error, an interrupt, or an uncaught exception
1813 * after the corresponding call to enter_cleanup(), discard what has been
1814 * made pending by it. Report this to the user if required by the
1815 * 'verbose' option or when debugging. */
1816 if (aborting() || need_rethrow) {
