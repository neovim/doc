194uv_handle_type uv_guess_handle(uv_file file) {
195 struct sockaddr sa;
196 struct stat s;
197 socklen_t len;
198 int type;
199
200 if (file < 0)
201 return UV_UNKNOWN_HANDLE;
202
203 if (isatty(file))
204 return UV_TTY;
205
206 if (fstat(file, &s))
207 return UV_UNKNOWN_HANDLE;
208
209 if (S_ISREG(s.st_mode)((((s.st_mode)) & 0170000) == (0100000)))
210 return UV_FILE;
211
212 if (S_ISCHR(s.st_mode)((((s.st_mode)) & 0170000) == (0020000)))
213 return UV_FILE; /* XXX UV_NAMED_PIPE? */
214
215 if (S_ISFIFO(s.st_mode)((((s.st_mode)) & 0170000) == (0010000)))
216 return UV_NAMED_PIPE;
217
218 if (!S_ISSOCK(s.st_mode)((((s.st_mode)) & 0170000) == (0140000)))
219 return UV_UNKNOWN_HANDLE;
220
221 len = sizeof(type);
222 if (getsockopt(file, SOL_SOCKET1, SO_TYPE3, &type, &len))
223 return UV_UNKNOWN_HANDLE;
224
225 len = sizeof(sa);
226 if (getsockname(file, &sa, &len))
227 return UV_UNKNOWN_HANDLE;
228
229 if (type == SOCK_DGRAMSOCK_DGRAM)
230 if (sa.sa_family == AF_INET2 || sa.sa_family == AF_INET610)
231 return UV_UDP;
232
233 if (type == SOCK_STREAMSOCK_STREAM) {
234 if (sa.sa_family == AF_INET2 || sa.sa_family == AF_INET610)
235 return UV_TCP;
236 if (sa.sa_family == AF_UNIX1)
237 return UV_NAMED_PIPE;
238 }
239
240 return UV_UNKNOWN_HANDLE;
241}
