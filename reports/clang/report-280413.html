1// VT220/xterm-like terminal emulator implementation for Neovim. Powered by
2// libvterm(http://www.leonerd.org.uk/code/libvterm/).
3//
4// libvterm is a pure C99 terminal emulation library with abstract input and
5// display. This means that the library needs to read data from the master fd
6// and feed VTerm instances, which will invoke user callbacks with screen
7// update instructions that must be mirrored to the real display.
8//
9// Keys are pressed in VTerm instances by calling
10// vterm_keyboard_key/vterm_keyboard_unichar, which generates byte streams that
11// must be fed back to the master fd.
12//
13// This implementation uses Neovim buffers as the display mechanism for both
14// the visible screen and the scrollback buffer. When focused, the window
15// "pins" to the bottom of the buffer and mirrors libvterm screen state.
16//
17// When a line becomes invisible due to a decrease in screen height or because
18// a line was pushed up during normal terminal output, we store the line
19// information in the scrollback buffer, which is mirrored in the Neovim buffer
20// by appending lines just above the visible part of the buffer.
21//
22// When the screen height increases, libvterm will ask for a row in the
23// scrollback buffer, which is mirrored in the Neovim buffer displaying lines
24// that were previously invisible.
25//
26// The vterm->Neovim synchronization is performed in intervals of 10
27// milliseconds. This is done to minimize screen updates when receiving when
28// receiving large bursts of data.
29//
30// This module is decoupled from the processes that normally feed it data, so
31// it's possible to use it as a general purpose console buffer(possibly as a
32// log/display mechanism for Neovim in the future)
33//
34// Inspired by vimshell(http://www.wana.at/vimshell/) and
35// Conque(https://code.google.com/p/conque/). Libvterm usage instructions (plus
36// some extra code) were taken from
37// pangoterm(http://www.leonerd.org.uk/code/pangoterm/)

615// the scrollback push/pop handlers were copied almost verbatim from pangoterm
616static int term_sb_push(int cols, const VTermScreenCell *cells, void *data)
617{
618 Terminal *term = data;
619 // copy vterm cells into sb_buffer
620 size_t c = (size_t)cols;
621 ScrollbackLine *sbrow = NULL((void*)0);
622 if (term->sb_current == term->sb_size) {
623 if (term->sb_buffer[term->sb_current - 1]->cols == c) {
624 // Recycle old row if it's the right size
625 sbrow = term->sb_buffer[term->sb_current - 1];
626 } else {
627 free(term->sb_buffer[term->sb_current - 1]);
628 }
629
630 memmove(term->sb_buffer + 1, term->sb_buffer,
631 sizeof(term->sb_buffer[0]) * (term->sb_current - 1));
632
633 } else if (term->sb_current > 0) {
634 memmove(term->sb_buffer + 1, term->sb_buffer,
635 sizeof(term->sb_buffer[0]) * term->sb_current);
636 }
637
638 if (!sbrow) {
639 sbrow = xmalloc(sizeof(ScrollbackLine) + c * sizeof(sbrow->cells[0]));
640 sbrow->cols = c;
641 }
642
643 term->sb_buffer[0] = sbrow;
644 if (term->sb_current < term->sb_size) {
645 term->sb_current++;
646 }
647
648 if (term->sb_pending < (int)term->sb_size) {
649 term->sb_pending++;
650 }
651
652 memcpy(sbrow->cells, cells, sizeof(cells[0]) * c);
653 pmap_put(ptr_t)map_ptr_t_ptr_t_put(invalidated_terminals, term, NULL((void*)0));
654
655 return 1;
656}
657
658static int term_sb_pop(int cols, VTermScreenCell *cells, void *data)
659{
660 Terminal *term = data;
661
662 if (!term->sb_current) {
663 return 0;
664 }
665
666 if (term->sb_pending) {
667 term->sb_pending--;
668 }
669
670 // restore vterm state
671 size_t c = (size_t)cols;
672 ScrollbackLine *sbrow = term->sb_buffer[0];
673 term->sb_current--;
674 memmove(term->sb_buffer, term->sb_buffer + 1,
675 sizeof(term->sb_buffer[0]) * (term->sb_current));
676
677 size_t cols_to_copy = c;
678 if (cols_to_copy > sbrow->cols) {
679 cols_to_copy = sbrow->cols;
680 }
681
682 // copy to vterm state
683 memcpy(cells, sbrow->cells, sizeof(cells[0]) * cols_to_copy);
684 for (size_t col = cols_to_copy; col < c; col++) {
685 cells[col].chars[0] = 0;
686 cells[col].width = 1;
687 }
688 free(sbrow);
689 pmap_put(ptr_t)map_ptr_t_ptr_t_put(invalidated_terminals, term, NULL((void*)0));
690
691 return 1;
692}

753static void mouse_action(Terminal *term, int button, int row, int col,
754 bool_Bool drag, VTermModifier mod)
755{
756 if (term->pressed_button && (term->pressed_button != button || !drag)) {
757 // release the previous button
758 vterm_mouse_button(term->vt, term->pressed_button, 0, mod);
759 term->pressed_button = 0;
760 }
761
762 // move the mouse
763 vterm_mouse_move(term->vt, row, col, mod);
764
765 if (!term->pressed_button) {
766 // press the button if not already pressed
767 vterm_mouse_button(term->vt, button, 1, mod);
768 term->pressed_button = button;
769 }

772// process a mouse event while the terminal is focused. return true if the
773// terminal should lose focus
774static bool_Bool send_mouse_event(Terminal *term, int c)
775{
776 int row = mouse_row, col = mouse_col;
777 win_T *mouse_win = mouse_find_win(&row, &col);
778
779 if (term->forward_mouse && mouse_win->w_buffer == term->buf) {
780 // event in the terminal window and mouse events was enabled by the
781 // program. translate and forward the event
782 int button;
783 bool_Bool drag = false0;
784
785 switch (c) {
786 case K_LEFTDRAG(-((253) + ((int)(KE_LEFTDRAG) << 8))): drag = true1;
787 case K_LEFTMOUSE(-((253) + ((int)(KE_LEFTMOUSE) << 8))): button = 1; break;
788 case K_MIDDLEDRAG(-((253) + ((int)(KE_MIDDLEDRAG) << 8))): drag = true1;
789 case K_MIDDLEMOUSE(-((253) + ((int)(KE_MIDDLEMOUSE) << 8))): button = 2; break;
790 case K_RIGHTDRAG(-((253) + ((int)(KE_RIGHTDRAG) << 8))): drag = true1;
791 case K_RIGHTMOUSE(-((253) + ((int)(KE_RIGHTMOUSE) << 8))): button = 3; break;
792 case K_MOUSEDOWN(-((253) + ((int)(KE_MOUSEDOWN) << 8))): button = 4; break;
793 case K_MOUSEUP(-((253) + ((int)(KE_MOUSEUP) << 8))): button = 5; break;
794 default: return false0;
795 }
796
797 mouse_action(term, button, row, col, drag, 0);
798 size_t len = vterm_output_read(term->vt, term->textbuf,
799 sizeof(term->textbuf));
800 terminal_send(term, term->textbuf, (size_t)len);
801 return false0;
802 }
803
804 if (c == K_MOUSEDOWN(-((253) + ((int)(KE_MOUSEDOWN) << 8))) || c == K_MOUSEUP(-((253) + ((int)(KE_MOUSEUP) << 8)))) {
805 win_T *save_curwin = curwin;
806 // switch window/buffer to perform the scroll
807 curwin = mouse_win;
808 curbuf = curwin->w_buffer;
809 int direction = c == K_MOUSEDOWN(-((253) + ((int)(KE_MOUSEDOWN) << 8))) ? MSCR_DOWN0 : MSCR_UP1;
810 if (mod_mask & (MOD_MASK_SHIFT0x02 | MOD_MASK_CTRL0x04)) {
811 scroll_redraw(direction, curwin->w_botline - curwin->w_topline);
812 } else {
813 scroll_redraw(direction, 3L);
814 }
815
816 curwin->w_redr_status = true1;
817 curwin = save_curwin;
818 curbuf = curwin->w_buffer;
819 redraw_win_later(mouse_win, NOT_VALID40);
820 invalidate_terminal(term, -1, -1);
821 // Only need to exit focus if the scrolled window is the terminal window
822 return mouse_win == curwin;
823 }
824
825 ins_char_typebuf(c);
826 return true1;
827}

833void fetch_row(Terminal *term, int row, int end_col)
834{
835 int col = 0;
836 size_t line_len = 0;
837 char *ptr = term->textbuf;
838
839 while (col < end_col) {
840 VTermScreenCell cell;
841 fetch_cell(term, row, col, &cell);
842 int cell_len = 0;
843 if (cell.chars[0]) {
844 for (int i = 0; cell.chars[i]; i++) {
845 cell_len += utf_char2bytes((int)cell.chars[i],
846 (uint8_t *)ptr + cell_len);
847 }
848 } else {
849 *ptr = ' ';
850 cell_len = 1;
851 }
852 char c = *ptr;
853 ptr += cell_len;
854 if (c != ' ') {
855 // only increase the line length if the last character is not whitespace
856 line_len = (size_t)(ptr - term->textbuf);
857 }
858 col += cell.width;
859 }
860
861 // trim trailing whitespace
862 term->textbuf[line_len] = 0;
863}
864
865static void fetch_cell(Terminal *term, int row, int col,
866 VTermScreenCell *cell)
867{
868 if (row < 0) {
869 ScrollbackLine *sbrow = term->sb_buffer[-row - 1];
870 if ((size_t)col < sbrow->cols) {
871 *cell = sbrow->cells[col];
872 } else {
873 // fill the pointer with an empty cell
874 *cell = (VTermScreenCell) {
875 .chars = { 0 },
876 .width = 1,
877 .bg = default_vt_bg_rgb
878 };
879 }
880 } else {
881 vterm_screen_get_cell(term->vts, (VTermPos){.row = row, .col = col},
882 cell);
883 }
884}

955// Refresh the scrollback of a invalidated terminal
956static void refresh_scrollback(Terminal *term)
957{
958 int width, height;
959 vterm_get_size(term->vt, &height, &width);
960
961 while (term->sb_pending > 0) {
962 // This means that either the window height has decreased or the screen
963 // became full and libvterm had to push all rows up. Convert the first
964 // pending scrollback row into a string and append it just above the visible
965 // section of the buffer
966 if (((int)term->buf->b_ml.ml_line_count - height) >= (int)term->sb_size) {
967 // scrollback full, delete lines at the top
968 ml_delete(1, false0);
969 deleted_lines(1, 1);
970 }
971 fetch_row(term, -term->sb_pending, width);
972 int buf_index = (int)term->buf->b_ml.ml_line_count - height;
973 ml_append(buf_index, (uint8_t *)term->textbuf, 0, false0);
974 appended_lines(buf_index, 1);
975 term->sb_pending--;
976 }
977
978 // Remove extra lines at the bottom
979 int max_line_count = (int)term->sb_current + height;
980 while (term->buf->b_ml.ml_line_count > max_line_count) {
981 ml_delete(term->buf->b_ml.ml_line_count, false0);
982 deleted_lines(term->buf->b_ml.ml_line_count, 1);
