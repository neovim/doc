579/*
580 * Search between "start" and "end" and try to recognize a
581 * character class in expanded form. For example [0-9].
582 * On success, return the id the character class to be emitted.
583 * On failure, return 0 (=FAIL)
584 * Start points to the first char of the range, while end should point
585 * to the closing brace.
586 * Keep in mind that 'ignorecase' applies at execution time, thus [a-z] may
587 * need to be interpreted as [a-zA-Z].
588 */
589static int nfa_recognize_char_class(char_u *start, char_u *end, int extra_newl)
590{
591# define CLASS_not0x80 0x80
592# define CLASS_af0x40 0x40
593# define CLASS_AF0x20 0x20
594# define CLASS_az0x10 0x10
595# define CLASS_AZ0x08 0x08
596# define CLASS_o70x04 0x04
597# define CLASS_o90x02 0x02
598# define CLASS_underscore0x01 0x01
599
600 int newl = FALSE0;
601 char_u *p;
602 int config = 0;
603
604 if (extra_newl == TRUE1)
605 newl = TRUE1;
606
607 if (*end != ']')
608 return FAIL0;
609 p = start;
610 if (*p == '^') {
611 config |= CLASS_not0x80;
612 p++;
613 }
614
615 while (p < end) {
616 if (p + 2 < end && *(p + 1) == '-') {
617 switch (*p) {
618 case '0':
619 if (*(p + 2) == '9') {
620 config |= CLASS_o90x02;
621 break;
622 } else if (*(p + 2) == '7') {
623 config |= CLASS_o70x04;
624 break;
625 }
626 case 'a':
627 if (*(p + 2) == 'z') {
628 config |= CLASS_az0x10;
629 break;
630 } else if (*(p + 2) == 'f') {
631 config |= CLASS_af0x40;
632 break;
633 }
634 case 'A':
635 if (*(p + 2) == 'Z') {
636 config |= CLASS_AZ0x08;
637 break;
638 } else if (*(p + 2) == 'F') {
639 config |= CLASS_AF0x20;
640 break;
641 }
642 /* FALLTHROUGH */
643 default:
644 return FAIL0;
645 }
646 p += 3;
647 } else if (p + 1 < end && *p == '\\' && *(p + 1) == 'n') {
648 newl = TRUE1;
649 p += 2;
650 } else if (*p == '_') {
651 config |= CLASS_underscore0x01;
652 p++;
653 } else if (*p == '\n') {
654 newl = TRUE1;
655 p++;
656 } else
657 return FAIL0;
658 } /* while (p < end) */
659
660 if (p != end)
661 return FAIL0;
662
663 if (newl == TRUE1)
664 extra_newl = NFA_ADD_NL31;
665
666 switch (config) {
667 case CLASS_o90x02:
668 return extra_newl + NFA_DIGIT;
669 case CLASS_not0x80 | CLASS_o90x02:
670 return extra_newl + NFA_NDIGIT;
671 case CLASS_af0x40 | CLASS_AF0x20 | CLASS_o90x02:
672 return extra_newl + NFA_HEX;
673 case CLASS_not0x80 | CLASS_af0x40 | CLASS_AF0x20 | CLASS_o90x02:
674 return extra_newl + NFA_NHEX;
675 case CLASS_o70x04:
676 return extra_newl + NFA_OCTAL;
677 case CLASS_not0x80 | CLASS_o70x04:
678 return extra_newl + NFA_NOCTAL;

3341/*
3342 * After building the NFA program, inspect it to add optimization hints.
3343 */
3344static void nfa_postprocess(nfa_regprog_T *prog)
3345{
3346 int i;
3347 int c;
3348
3349 for (i = 0; i < prog->nstate; ++i) {
3350 c = prog->state[i].c;
3351 if (c == NFA_START_INVISIBLE
3352 || c == NFA_START_INVISIBLE_NEG
3353 || c == NFA_START_INVISIBLE_BEFORE
3354 || c == NFA_START_INVISIBLE_BEFORE_NEG) {
3355 int directly;
3356
3357 /* Do it directly when what follows is possibly the end of the
3358 * match. */
3359 if (match_follows(prog->state[i].out1->out, 0))
3360 directly = TRUE1;
3361 else {
3362 int ch_invisible = failure_chance(prog->state[i].out, 0);
3363 int ch_follows = failure_chance(prog->state[i].out1->out, 0);
3364
3365 /* Postpone when the invisible match is expensive or has a
3366 * lower chance of failing. */
3367 if (c == NFA_START_INVISIBLE_BEFORE
3368 || c == NFA_START_INVISIBLE_BEFORE_NEG) {
3369 /* "before" matches are very expensive when
3370 * unbounded, always prefer what follows then,
3371 * unless what follows will always match.
3372 * Otherwise strongly prefer what follows. */
3373 if (prog->state[i].val <= 0 && ch_follows > 0)
3374 directly = FALSE0;
3375 else
3376 directly = ch_follows * 10 < ch_invisible;
3377 } else {
3378 /* normal invisible, first do the one with the
3379 * highest failure chance */
3380 directly = ch_follows < ch_invisible;
3381 }
3382 }
3383 if (directly)
3384 /* switch to the _FIRST state */
3385 ++prog->state[i].c;
3386 }
3387 }

3537 linenr_T s2;

3539 char_u *sp2;

3542 if (REG_MULTI(reg_match == ((void*)0))) {
3543 for (i = 0; i < todo; ++i) {
3544 if (i < sub1->in_use)
3545 s1 = sub1->list.multi[i].start.lnum;
3546 else
3547 s1 = -1;
3548 if (i < sub2->in_use)
3549 s2 = sub2->list.multi[i].start.lnum;

3555 != sub2->list.multi[i].start.col)

3560 s1 = sub1->list.multi[i].end.lnum;

3566 } else {
3567 s2 = -1;
3568 }

3576 }
3577 }

3580 if (i < sub1->in_use)
3581 sp1 = sub1->list.line[i].start;

4140/*
4141 * Like addstate(), but the new state(s) are put at position "*ip".
4142 * Used for zero-width matches, next state to use is the added one.
4143 * This makes sure the order of states to be tried does not change, which
4144 * matters for alternatives.
4145 */
4146static void
4147addstate_here (
4148 nfa_list_T *l, /* runtime state list */
4149 nfa_state_T *state, /* state to update */
4150 regsubs_T *subs, /* pointers to subexpressions */
4151 nfa_pim_T *pim, /* postponed look-behind match */
4152 int *ip
4153)
4154{
4155 int tlen = l->n;
4156 int count;
4157 int listidx = *ip;
4158
4159 /* first add the state(s) at the end, so that we know how many there are */
4160 addstate(l, state, subs, pim, 0);
4161
4162 /* when "*ip" was at the end of the list, nothing to do */
4163 if (listidx + 1 == tlen)
4164 return;
4165
4166 /* re-order to put the new state at the current position */
4167 count = l->n - tlen;
4168 if (count == 0)
4169 return; /* no state got added */
4170 if (count == 1) {
4171 /* overwrite the current state */
4172 l->t[listidx] = l->t[l->n - 1];
4173 } else if (count > 1) {
4174 if (l->n + count - 1 >= l->len) {
4175 /* not enough space to move the new states, reallocate the list
4176 * and move the states to the right position */
4177
4178 l->len = l->len * 3 / 2 + 50;
4179 nfa_thread_T *newl = xmalloc(l->len * sizeof(nfa_thread_T));
4180 memmove(&(newl[0]),
4181 &(l->t[0]),
4182 sizeof(nfa_thread_T) * listidx);
4183 memmove(&(newl[listidx]),
4184 &(l->t[l->n - count]),
4185 sizeof(nfa_thread_T) * count);
4186 memmove(&(newl[listidx + count]),
4187 &(l->t[listidx + 1]),
4188 sizeof(nfa_thread_T) * (l->n - count - listidx - 1));
4189 free(l->t);
4190 l->t = newl;
4191 } else {
4192 /* make space for new states, then move them from the
4193 * end to the current position */
4194 memmove(&(l->t[listidx + count]),
4195 &(l->t[listidx + 1]),
4196 sizeof(nfa_thread_T) * (l->n - listidx - 1));
4197 memmove(&(l->t[listidx]),
4198 &(l->t[l->n - 1]),
4199 sizeof(nfa_thread_T) * count);
4200 }
4201 }
4202 --l->n;
4203 *ip = listidx - 1;

5905 * Also don't start a match past the first line. */
5906 if (nfa_match == FALSE0
5907 && ((toplevel
5908 && reglnum == 0
5909 && clen != 0
5910 && (ireg_maxcol == 0
5911 || (colnr_T)(reginput - regline) < ireg_maxcol))
5912 || (nfa_endp != NULL((void*)0)
5913 && (REG_MULTI(reg_match == ((void*)0))
5914 ? (reglnum < nfa_endp->se_u.pos.lnum
5915 || (reglnum == nfa_endp->se_u.pos.lnum
5916 && (int)(reginput - regline)
5917 < nfa_endp->se_u.pos.col))
5918 : reginput < nfa_endp->se_u.ptr)))) {
5919#ifdef REGEXP_DEBUG
5920 fprintf(log_fd, "(---) STARTSTATE\n");
5921#endif
5922 /* Inline optimized code for addstate() if we know the state is
5923 * the first MOPEN. */
5924 if (toplevel) {
5925 int add = TRUE1;
5926 int c;
5927
5928 if (prog->regstart != NUL'\000' && clen != 0) {
5929 if (nextlist->n == 0) {
5930 colnr_T col = (colnr_T)(reginput - regline) + clen;
5931
5932 /* Nextlist is empty, we can skip ahead to the
5933 * character that must appear at the start. */
5934 if (skip_to_start(prog->regstart, &col) == FAIL0)
5935 break;
5936#ifdef REGEXP_DEBUG
5937 fprintf(log_fd, " Skipping ahead %d bytes to regstart\n",
5938 col - ((colnr_T)(reginput - regline) + clen));
5939#endif
5940 reginput = regline + col - clen;
5941 } else {
5942 /* Checking if the required start character matches is
5943 * cheaper than adding a state that won't match. */
5944 c = PTR2CHAR(reginput + clen)(has_mbyte ? mb_ptr2char(reginput + clen) : (int)*(reginput +
clen));
5945 if (c != prog->regstart && (!ireg_ic || vim_tolower(c)
5946 != vim_tolower(prog->regstart))) {
5947#ifdef REGEXP_DEBUG
5948 fprintf(log_fd,
5949 " Skipping start state, regstart does not match\n");
5950#endif
5951 add = FALSE0;
5952 }
5953 }
5954 }
5955
5956 if (add) {
5957 if (REG_MULTI(reg_match == ((void*)0)))
5958 m->norm.list.multi[0].start.col =
5959 (colnr_T)(reginput - regline) + clen;
5960 else
5961 m->norm.list.line[0].start = reginput + clen;
5962 addstate(nextlist, start->out, m, NULL((void*)0), clen);
5963 }
5964 } else
5965 addstate(nextlist, start, m, NULL((void*)0), clen);
5966 }
5967
5968#ifdef REGEXP_DEBUG
5969 fprintf(log_fd, ">>> Thislist had %d states available: ", thislist->n);
5970 {
5971 int i;
5972
5973 for (i = 0; i < thislist->n; i++)
5974 fprintf(log_fd, "%d ", abs(thislist->t[i].state->id));
5975 }
5976 fprintf(log_fd, "\n");
5977#endif
5978
5979nextchar:
5980 /* Advance to the next character, or advance to the next line, or
5981 * finish. */
5982 if (clen != 0)
5983 reginput += clen;
5984 else if (go_to_nextline || (nfa_endp != NULL((void*)0) && REG_MULTI(reg_match == ((void*)0))
5985 && reglnum < nfa_endp->se_u.pos.lnum))
5986 reg_nextline();
5987 else
5988 break;
5989 }
5990
5991#ifdef REGEXP_DEBUG
5992 if (log_fd != stderrstderr)
5993 fclose(log_fd);
5994 log_fd = NULL((void*)0);
5995#endif
5996
5997 /* Free memory */
5998 free(list[0].t);
5999 free(list[1].t);
6000 free(listids);
6001#undef ADD_STATE_IF_MATCH
6002#ifdef NFA_REGEXP_DEBUG_LOG
6003 fclose(debug);
6004#endif
6005
6006 return nfa_match;
6007}

6114 * lines ("line" is NULL, use reg_getline()).

6119nfa_regexec_both (
6120 char_u *line,
6121 colnr_T startcol /* column to start looking for match */
6122)
6123{
