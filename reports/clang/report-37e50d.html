544/* foldCreate() {{{2 */
545/*
546 * Create a fold from line "start" to line "end" (inclusive) in the current
547 * window.
548 */
549void foldCreate(linenr_T start, linenr_T end)
550{
551 fold_T *fp;
552 garray_T *gap;
553 garray_T fold_ga;
554 int i, j;
555 int cont;
556 int use_level = FALSE0;
557 int closed = FALSE0;
558 int level = 0;
559 linenr_T start_rel = start;
560 linenr_T end_rel = end;
561
562 if (start > end) {
563 /* reverse the range */
564 end = start_rel;
565 start = end_rel;
566 start_rel = start;
567 end_rel = end;
568 }
569
570 /* When 'foldmethod' is "marker" add markers, which creates the folds. */
571 if (foldmethodIsMarker(curwin)) {
572 foldCreateMarkers(start, end);
573 return;
574 }
575
576 checkupdate(curwin);
577
578 /* Find the place to insert the new fold. */
579 gap = &curwin->w_folds;
580 for (;; ) {
581 if (!foldFind(gap, start_rel, &fp))
582 break;
583 if (fp->fd_top + fp->fd_len > end_rel) {
584 /* New fold is completely inside this fold: Go one level deeper. */
585 gap = &fp->fd_nested;
586 start_rel -= fp->fd_top;
587 end_rel -= fp->fd_top;
588 if (use_level || fp->fd_flags == FD_LEVEL2) {
589 use_level = TRUE1;
590 if (level >= curwin->w_p_fdlw_onebuf_opt.wo_fdl)
591 closed = TRUE1;
592 } else if (fp->fd_flags == FD_CLOSED1)
593 closed = TRUE1;
594 ++level;
595 } else {
596 /* This fold and new fold overlap: Insert here and move some folds
597 * inside the new fold. */
598 break;
599 }
600 }
601
602 i = (int)(fp - (fold_T *)gap->ga_data);
603 ga_grow(gap, 1);
604 {
605 fp = (fold_T *)gap->ga_data + i;
606 ga_init(&fold_ga, (int)sizeof(fold_T), 10);
607
608 /* Count number of folds that will be contained in the new fold. */
609 for (cont = 0; i + cont < gap->ga_len; ++cont)
610 if (fp[cont].fd_top > end_rel)
611 break;
612 if (cont > 0) {
613 ga_grow(&fold_ga, cont);
614 /* If the first fold starts before the new fold, let the new fold
615 * start there. Otherwise the existing fold would change. */
616 if (start_rel > fp->fd_top)
617 start_rel = fp->fd_top;
618
619 /* When last contained fold isn't completely contained, adjust end
620 * of new fold. */
621 if (end_rel < fp[cont - 1].fd_top + fp[cont - 1].fd_len - 1)
622 end_rel = fp[cont - 1].fd_top + fp[cont - 1].fd_len - 1;
623 /* Move contained folds to inside new fold. */
624 memmove(fold_ga.ga_data, fp, sizeof(fold_T) * cont);
625 fold_ga.ga_len += cont;
626 i += cont;
627
628 /* Adjust line numbers in contained folds to be relative to the
629 * new fold. */
630 for (j = 0; j < cont; ++j)
631 ((fold_T *)fold_ga.ga_data)[j].fd_top -= start_rel;
632 }
633 /* Move remaining entries to after the new fold. */
634 if (i < gap->ga_len)
635 memmove(fp + 1, (fold_T *)gap->ga_data + i,
636 sizeof(fold_T) * (gap->ga_len - i));
637 gap->ga_len = gap->ga_len + 1 - cont;
638
639 /* insert new fold */
640 fp->fd_nested = fold_ga;
641 fp->fd_top = start_rel;
642 fp->fd_len = end_rel - start_rel + 1;
643
644 /* We want the new fold to be closed. If it would remain open because
645 * of using 'foldlevel', need to adjust fd_flags of containing folds.
646 */
647 if (use_level && !closed && level < curwin->w_p_fdlw_onebuf_opt.wo_fdl)
648 closeFold(start, 1L);
649 if (!use_level)
650 curwin->w_fold_manual = TRUE1;
651 fp->fd_flags = FD_CLOSED1;
652 fp->fd_small = MAYBE2;
653
654 /* redraw */
655 changed_window_setting();
656 }

806/* foldMoveTo() {{{2 */
807/*
808 * If "updown" is FALSE: Move to the start or end of the fold.
809 * If "updown" is TRUE: move to fold at the same level.
810 * If not moved return FAIL.
811 */
812int
813foldMoveTo (
814 int updown,
815 int dir, /* FORWARD or BACKWARD */
816 long count
817)
818{
819 long n;
820 int retval = FAIL0;
821 linenr_T lnum_off;
822 linenr_T lnum_found;
823 linenr_T lnum;
824 int use_level;
825 int maybe_small;
826 garray_T *gap;
827 fold_T *fp;
828 int level;
829 int last;
830
831 checkupdate(curwin);
832
833 /* Repeat "count" times. */
834 for (n = 0; n < count; ++n) {
835 /* Find nested folds. Stop when a fold is closed. The deepest fold
836 * that moves the cursor is used. */
837 lnum_off = 0;
838 gap = &curwin->w_folds;
839 use_level = FALSE0;
840 maybe_small = FALSE0;
841 lnum_found = curwin->w_cursor.lnum;
842 level = 0;
843 last = FALSE0;
844 for (;; ) {
845 if (!foldFind(gap, curwin->w_cursor.lnum - lnum_off, &fp)) {
846 if (!updown)
847 break;
848
849 /* When moving up, consider a fold above the cursor; when
850 * moving down consider a fold below the cursor. */
851 if (dir == FORWARD1) {
852 if (fp - (fold_T *)gap->ga_data >= gap->ga_len)
853 break;
854 --fp;
855 } else {
856 if (fp == (fold_T *)gap->ga_data)
857 break;
858 }
859 /* don't look for contained folds, they will always move
860 * the cursor too far. */
861 last = TRUE1;
862 }
863
864 if (!last) {
865 /* Check if this fold is closed. */
866 if (check_closed(curwin, fp, &use_level, level,
867 &maybe_small, lnum_off))
868 last = TRUE1;
869
870 /* "[z" and "]z" stop at closed fold */
871 if (last && !updown)
872 break;
873 }
874
875 if (updown) {
876 if (dir == FORWARD1) {
877 /* to start of next fold if there is one */
878 if (fp + 1 - (fold_T *)gap->ga_data < gap->ga_len) {
879 lnum = fp[1].fd_top + lnum_off;
880 if (lnum > curwin->w_cursor.lnum)
881 lnum_found = lnum;
882 }
883 } else {
884 /* to end of previous fold if there is one */
885 if (fp > (fold_T *)gap->ga_data) {
886 lnum = fp[-1].fd_top + lnum_off + fp[-1].fd_len - 1;
887 if (lnum < curwin->w_cursor.lnum)
888 lnum_found = lnum;
889 }
890 }
891 } else {
892 /* Open fold found, set cursor to its start/end and then check
893 * nested folds. */
894 if (dir == FORWARD1) {
895 lnum = fp->fd_top + lnum_off + fp->fd_len - 1;
896 if (lnum > curwin->w_cursor.lnum)
897 lnum_found = lnum;
898 } else {
899 lnum = fp->fd_top + lnum_off;
900 if (lnum < curwin->w_cursor.lnum)
901 lnum_found = lnum;
902 }
903 }
904
905 if (last)
906 break;
907
908 /* Check nested folds (if any). */
909 gap = &fp->fd_nested;
910 lnum_off += fp->fd_top;
911 ++level;
912 }
913 if (lnum_found != curwin->w_cursor.lnum) {
914 if (retval == FAIL0)
915 setpcmark();
916 curwin->w_cursor.lnum = lnum_found;
917 curwin->w_cursor.col = 0;
918 retval = OK1;
919 } else
920 break;
921 }
922
923 return retval;
924}

1029/* foldFind() {{{2 */
1030/*
1031 * Search for line "lnum" in folds of growarray "gap".
1032 * Set *fpp to the fold struct for the fold that contains "lnum" or
1033 * the first fold below it (careful: it can be beyond the end of the array!).
1034 * Returns FALSE when there is no fold that contains "lnum".
1035 */
1036static int foldFind(garray_T *gap, linenr_T lnum, fold_T **fpp)
1037{
1038 linenr_T low, high;
1039 fold_T *fp;
1040
1041 /*
1042 * Perform a binary search.
1043 * "low" is lowest index of possible match.
1044 * "high" is highest index of possible match.
1045 */
1046 fp = (fold_T *)gap->ga_data;
1047 low = 0;
1048 high = gap->ga_len - 1;
1049 while (low <= high) {
1050 int i = (low + high) / 2;
1051 if (fp[i].fd_top > lnum)
1052 /* fold below lnum, adjust high */
1053 high = i - 1;
1054 else if (fp[i].fd_top + fp[i].fd_len <= lnum)
1055 /* fold above lnum, adjust low */
1056 low = i + 1;
1057 else {
1058 /* lnum is inside this fold */
1059 *fpp = fp + i;
1060 return TRUE1;
1061 }
1062 }
1063 *fpp = fp + low;
1064 return FALSE0;
1065}

1175 int *donep

1179 fold_T *fp2;

1281/* deleteFoldEntry() {{{2 */
1282/*
1283 * Delete fold "idx" from growarray "gap".
1284 * When "recursive" is TRUE also delete all the folds contained in it.
1285 * When "recursive" is FALSE contained folds are moved one level up.
1286 */
1287static void deleteFoldEntry(garray_T *gap, int idx, int recursive)
1288{
1289 fold_T *fp;
1290 int i;

1293
1294 fp = (fold_T *)gap->ga_data + idx;

1301 } else {

1303 * deleted. */
1304 moved = fp->fd_nested.ga_len;

1317 nfp[i].fd_small = MAYBE2;
1318 }

1322 memmove(fp + moved, fp + 1,
1323 sizeof(fold_T) * (gap->ga_len - (idx + 1)));
1324 /* move the contained folds one level up */
1325 memmove(fp, nfp, (size_t)(sizeof(fold_T) * moved));

1328 }

1952 /* Backup to a line for which the fold level is defined. Since it's
1953 * always defined for line one, we will stop there. */
1954 fline.lvl = -1;
1955 for (; !got_int; --fline.lnum) {
1956 /* Reset lvl_next each time, because it will be set to a value for
1957 * the next line, but we search backwards here. */
1958 fline.lvl_next = -1;
1959 getlevel(&fline);
1960 if (fline.lvl >= 0)
1961 break;
1962 }
1963 }
1964
1965 /*
1966 * If folding is defined by the syntax, it is possible that a change in
1967 * one line will cause all sub-folds of the current fold to change (e.g.,
1968 * closing a C-style comment can cause folds in the subsequent lines to
1969 * appear). To take that into account we should adjust the value of "bot"
1970 * to point to the end of the current fold:
1971 */
1972 if (foldlevelSyntax == getlevel) {
1973 garray_T *gap = &wp->w_folds;
1974 fold_T *fpn = NULL((void*)0);
1975 int current_fdl = 0;
1976 linenr_T fold_start_lnum = 0;
1977 linenr_T lnum_rel = fline.lnum;
1978
1979 while (current_fdl < fline.lvl) {
1980 if (!foldFind(gap, lnum_rel, &fpn))
1981 break;
1982 ++current_fdl;
1983
1984 fold_start_lnum += fpn->fd_top;
1985 gap = &fpn->fd_nested;
1986 lnum_rel -= fpn->fd_top;
1987 }
1988 if (fpn != NULL((void*)0) && current_fdl == fline.lvl) {
1989 linenr_T fold_end_lnum = fold_start_lnum + fpn->fd_len;
1990
1991 if (fold_end_lnum > bot)
1992 bot = fold_end_lnum;
1993 }
1994 }
1995
1996 start = fline.lnum;
1997 end = bot;
1998 /* Do at least one line. */
1999 if (start > end && end < wp->w_buffer->b_ml.ml_line_count)
2000 end = start;
2001 while (!got_int) {
2002 /* Always stop at the end of the file ("end" can be past the end of
2003 * the file). */
2004 if (fline.lnum > wp->w_buffer->b_ml.ml_line_count)
2005 break;
2006 if (fline.lnum > end) {
2007 /* For "marker", "expr" and "syntax" methods: If a change caused
2008 * a fold to be removed, we need to continue at least until where
2009 * it ended. */
2010 if (getlevel != foldlevelMarker
2011 && getlevel != foldlevelSyntax
2012 && getlevel != foldlevelExpr)
2013 break;
2014 if ((start <= end
2015 && foldFind(&wp->w_folds, end, &fp)
2016 && fp->fd_top + fp->fd_len - 1 > end)
2017 || (fline.lvl == 0
2018 && foldFind(&wp->w_folds, fline.lnum, &fp)
2019 && fp->fd_top < fline.lnum))
2020 end = fp->fd_top + fp->fd_len - 1;
2021 else if (getlevel == foldlevelSyntax
2022 && foldLevelWin(wp, fline.lnum) != fline.lvl)
2023 /* For "syntax" method: Compare the foldlevel that the syntax
2024 * tells us to the foldlevel from the existing folds. If they
2025 * don't match continue updating folds. */
2026 end = fline.lnum;
2027 else
2028 break;
2029 }
2030
2031 /* A level 1 fold starts at a line with foldlevel > 0. */
2032 if (fline.lvl > 0) {
2033 invalid_top = fline.lnum;
2034 invalid_bot = end;
2035 end = foldUpdateIEMSRecurse(&wp->w_folds,
2036 1, start, &fline, getlevel, end, FD_LEVEL2);
2037 start = fline.lnum;
2038 } else {
2039 if (fline.lnum == wp->w_buffer->b_ml.ml_line_count)
2040 break;
2041 ++fline.lnum;
2042 fline.lvl = fline.lvl_next;
2043 getlevel(&fline);
2044 }
2045 }
2046
2047 /* There can't be any folds from start until end now. */
2048 foldRemove(&wp->w_folds, start, end);
2049
2050 /* If some fold changed, need to redraw and position cursor. */
2051 if (fold_changed && wp->w_p_fenw_onebuf_opt.wo_fen)
2052 changed_window_setting_win(wp);
2053
2054 /* If we updated folds past "bot", need to redraw more lines. Don't do
2055 * this in other situations, the changed lines will be redrawn anyway and
2056 * this method can cause the whole window to be updated. */
2057 if (end != bot) {
2058 if (wp->w_redraw_top == 0 || wp->w_redraw_top > top)
2059 wp->w_redraw_top = top;
2060 if (wp->w_redraw_bot < end)
2061 wp->w_redraw_bot = end;
2062 }
2063
2064 invalid_top = (linenr_T)0;
2065}

2067/* foldUpdateIEMSRecurse() {{{2 */
2068/*
2069 * Update a fold that starts at "flp->lnum". At this line there is always a
2070 * valid foldlevel, and its level >= "level".
2071 * "flp" is valid for "flp->lnum" when called and it's valid when returning.
2072 * "flp->lnum" is set to the lnum just below the fold, if it ends before
2073 * "bot", it's "bot" plus one if the fold continues and it's bigger when using
2074 * the marker method and a text change made following folds to change.
2075 * When returning, "flp->lnum_save" is the line number that was used to get
2076 * the level when the level at "flp->lnum" is invalid.
2077 * Remove any folds from "startlnum" up to here at this level.
2078 * Recursively update nested folds.
2079 * Below line "bot" there are no changes in the text.
2080 * "flp->lnum", "flp->lnum_save" and "bot" are relative to the start of the
2081 * outer fold.
2082 * "flp->off" is the offset to the real line number in the buffer.
2083 *
2084 * All this would be a lot simpler if all folds in the range would be deleted
2085 * and then created again. But we would lose all information about the
2086 * folds, even when making changes that don't affect the folding (e.g. "vj~").
2087 *
2088 * Returns bot, which may have been increased for lines that also need to be
2089 * updated as a result of a detected change in the fold.
2090 */
2091static linenr_T foldUpdateIEMSRecurse(garray_T *gap, int level,
2092 linenr_T startlnum, fline_T *flp,
2093 LevelGetter getlevel,
2094 linenr_T bot,
2095 int topflags /* flags used by containing fold */
2096 )
2097{
2098 linenr_T ll;
2099 fold_T *fp = NULL((void*)0);
2100 fold_T *fp2;
2101 int lvl = level;
2102 linenr_T startlnum2 = startlnum;
2103 linenr_T firstlnum = flp->lnum; /* first lnum we got */
2104 int i;
2105 int finish = FALSE0;
2106 linenr_T linecount = flp->wp->w_buffer->b_ml.ml_line_count - flp->off;
2107 int concat;
2108
2109 /*
2110 * If using the marker method, the start line is not the start of a fold
2111 * at the level we're dealing with and the level is non-zero, we must use
2112 * the previous fold. But ignore a fold that starts at or below
2113 * startlnum, it must be deleted.
2114 */
2115 if (getlevel == foldlevelMarker && flp->start <= flp->lvl - level
2116 && flp->lvl > 0) {
2117 foldFind(gap, startlnum - 1, &fp);
2118 if (fp >= ((fold_T *)gap->ga_data) + gap->ga_len
2119 || fp->fd_top >= startlnum)
2120 fp = NULL((void*)0);
2121 }
2122
2123 /*
2124 * Loop over all lines in this fold, or until "bot" is hit.
2125 * Handle nested folds inside of this fold.
2126 * "flp->lnum" is the current line. When finding the end of the fold, it
2127 * is just below the end of the fold.
2128 * "*flp" contains the level of the line "flp->lnum" or a following one if
2129 * there are lines with an invalid fold level. "flp->lnum_save" is the
2130 * line number that was used to get the fold level (below "flp->lnum" when
2131 * it has an invalid fold level). When called the fold level is always
2132 * valid, thus "flp->lnum_save" is equal to "flp->lnum".
2133 */
2134 flp->lnum_save = flp->lnum;
2135 while (!got_int) {
2136 /* Updating folds can be slow, check for CTRL-C. */
2137 line_breakcheck();
2138
2139 /* Set "lvl" to the level of line "flp->lnum". When flp->start is set
2140 * and after the first line of the fold, set the level to zero to
2141 * force the fold to end. Do the same when had_end is set: Previous
2142 * line was marked as end of a fold. */
2143 lvl = flp->lvl;
2144 if (lvl > MAX_LEVEL20)
2145 lvl = MAX_LEVEL20;
2146 if (flp->lnum > firstlnum
2147 && (level > lvl - flp->start || level >= flp->had_end))
2148 lvl = 0;
2149
2150 if (flp->lnum > bot && !finish && fp != NULL((void*)0)) {
2151 /* For "marker" and "syntax" methods:
2152 * - If a change caused a nested fold to be removed, we need to
2153 * delete it and continue at least until where it ended.
2154 * - If a change caused a nested fold to be created, or this fold
2155 * to continue below its original end, need to finish this fold.
2156 */
2157 if (getlevel != foldlevelMarker
2158 && getlevel != foldlevelExpr
2159 && getlevel != foldlevelSyntax)
2160 break;
2161 i = 0;
2162 fp2 = fp;
2163 if (lvl >= level) {
2164 /* Compute how deep the folds currently are, if it's deeper
2165 * than "lvl" then some must be deleted, need to update
2166 * at least one nested fold. */
2167 ll = flp->lnum - fp->fd_top;
2168 while (foldFind(&fp2->fd_nested, ll, &fp2)) {
2169 ++i;
2170 ll -= fp2->fd_top;
2171 }
2172 }
2173 if (lvl < level + i) {
2174 foldFind(&fp->fd_nested, flp->lnum - fp->fd_top, &fp2);
2175 if (fp2 != NULL((void*)0))
2176 bot = fp2->fd_top + fp2->fd_len - 1 + fp->fd_top;
2177 } else if (fp->fd_top + fp->fd_len <= flp->lnum && lvl >= level)
2178 finish = TRUE1;
2179 else
2180 break;
2181 }
2182
2183 /* At the start of the first nested fold and at the end of the current
2184 * fold: check if existing folds at this level, before the current
2185 * one, need to be deleted or truncated. */
2186 if (fp == NULL((void*)0)
2187 && (lvl != level
2188 || flp->lnum_save >= bot
2189 || flp->start != 0
2190 || flp->had_end <= MAX_LEVEL20
2191 || flp->lnum == linecount)) {
2192 /*
2193 * Remove or update folds that have lines between startlnum and
2194 * firstlnum.
2195 */
2196 while (!got_int) {
2197 /* set concat to 1 if it's allowed to concatenated this fold
2198 * with a previous one that touches it. */
2199 if (flp->start != 0 || flp->had_end <= MAX_LEVEL20)
2200 concat = 0;
2201 else
2202 concat = 1;
2203
2204 /* Find an existing fold to re-use. Preferably one that
2205 * includes startlnum, otherwise one that ends just before
2206 * startlnum or starts after it. */
2207 if (foldFind(gap, startlnum, &fp)
2208 || (fp < ((fold_T *)gap->ga_data) + gap->ga_len
2209 && fp->fd_top <= firstlnum)
2210 || foldFind(gap, firstlnum - concat, &fp)
2211 || (fp < ((fold_T *)gap->ga_data) + gap->ga_len
2212 && ((lvl < level && fp->fd_top < flp->lnum)
2213 || (lvl >= level
2214 && fp->fd_top <= flp->lnum_save)))) {
2215 if (fp->fd_top + fp->fd_len + concat > firstlnum) {
2216 /* Use existing fold for the new fold. If it starts
2217 * before where we started looking, extend it. If it
2218 * starts at another line, update nested folds to keep
2219 * their position, compensating for the new fd_top. */
2220 if (fp->fd_top >= startlnum && fp->fd_top != firstlnum) {
2221 if (fp->fd_top > firstlnum)
2222 /* like lines are inserted */
2223 foldMarkAdjustRecurse(&fp->fd_nested,
2224 (linenr_T)0, (linenr_T)MAXLNUM(0x7fffffffL),
2225 (long)(fp->fd_top - firstlnum), 0L);
2226 else
2227 /* like lines are deleted */
2228 foldMarkAdjustRecurse(&fp->fd_nested,
2229 (linenr_T)0,
2230 (long)(firstlnum - fp->fd_top - 1),
2231 (linenr_T)MAXLNUM(0x7fffffffL),
2232 (long)(fp->fd_top - firstlnum));
2233 fp->fd_len += fp->fd_top - firstlnum;
2234 fp->fd_top = firstlnum;
2235 fold_changed = TRUE1;
2236 } else if (flp->start != 0 && lvl == level
2237 && fp->fd_top != firstlnum) {
2238 /* Existing fold that includes startlnum must stop
2239 * if we find the start of a new fold at the same
2240 * level. Split it. Delete contained folds at
2241 * this point to split them too. */
2242 foldRemove(&fp->fd_nested, flp->lnum - fp->fd_top,
2243 flp->lnum - fp->fd_top);
2244 i = (int)(fp - (fold_T *)gap->ga_data);
2245 foldSplit(gap, i, flp->lnum, flp->lnum - 1);
2246 fp = (fold_T *)gap->ga_data + i + 1;
2247 /* If using the "marker" or "syntax" method, we
2248 * need to continue until the end of the fold is
2249 * found. */
2250 if (getlevel == foldlevelMarker
2251 || getlevel == foldlevelExpr
2252 || getlevel == foldlevelSyntax)
2253 finish = TRUE1;
2254 }
2255 break;
2256 }
2257 if (fp->fd_top >= startlnum) {
2258 /* A fold that starts at or after startlnum and stops
2259 * before the new fold must be deleted. Continue
2260 * looking for the next one. */
2261 deleteFoldEntry(gap,
2262 (int)(fp - (fold_T *)gap->ga_data), TRUE1);
2263 } else {
2264 /* A fold has some lines above startlnum, truncate it
2265 * to stop just above startlnum. */
2266 fp->fd_len = startlnum - fp->fd_top;
2267 foldMarkAdjustRecurse(&fp->fd_nested,
2268 (linenr_T)fp->fd_len, (linenr_T)MAXLNUM(0x7fffffffL),
2269 (linenr_T)MAXLNUM(0x7fffffffL), 0L);
2270 fold_changed = TRUE1;
2271 }
2272 } else {
2273 /* Insert new fold. Careful: ga_data may be NULL and it
2274 * may change! */
2275 i = (int)(fp - (fold_T *)gap->ga_data);
2276 foldInsert(gap, i);
2277 fp = (fold_T *)gap->ga_data + i;
2278 /* The new fold continues until bot, unless we find the
2279 * end earlier. */
2280 fp->fd_top = firstlnum;
2281 fp->fd_len = bot - firstlnum + 1;
2282 /* When the containing fold is open, the new fold is open.
2283 * The new fold is closed if the fold above it is closed.
2284 * The first fold depends on the containing fold. */
2285 if (topflags == FD_OPEN0) {
2286 flp->wp->w_fold_manual = TRUE1;
2287 fp->fd_flags = FD_OPEN0;
2288 } else if (i <= 0) {
2289 fp->fd_flags = topflags;
2290 if (topflags != FD_LEVEL2)
2291 flp->wp->w_fold_manual = TRUE1;
2292 } else
2293 fp->fd_flags = (fp - 1)->fd_flags;
2294 fp->fd_small = MAYBE2;
2295 /* If using the "marker", "expr" or "syntax" method, we
2296 * need to continue until the end of the fold is found. */
2297 if (getlevel == foldlevelMarker
2298 || getlevel == foldlevelExpr
2299 || getlevel == foldlevelSyntax)
2300 finish = TRUE1;
2301 fold_changed = TRUE1;
2302 break;
2303 }
2304 }
2305 }
2306
2307 if (lvl < level || flp->lnum > linecount) {
2308 /*
2309 * Found a line with a lower foldlevel, this fold ends just above
2310 * "flp->lnum".
2311 */
2312 break;
2313 }
2314
2315 /*
2316 * The fold includes the line "flp->lnum" and "flp->lnum_save".
2317 * Check "fp" for safety.
2318 */
2319 if (lvl > level && fp != NULL((void*)0)) {
2320 /*
2321 * There is a nested fold, handle it recursively.
2322 */
2323 /* At least do one line (can happen when finish is TRUE). */
2324 if (bot < flp->lnum)
2325 bot = flp->lnum;
2326
2327 /* Line numbers in the nested fold are relative to the start of
2328 * this fold. */
2329 flp->lnum = flp->lnum_save - fp->fd_top;
2330 flp->off += fp->fd_top;
2331 i = (int)(fp - (fold_T *)gap->ga_data);
2332 bot = foldUpdateIEMSRecurse(&fp->fd_nested, level + 1,
2333 startlnum2 - fp->fd_top, flp, getlevel,
2334 bot - fp->fd_top, fp->fd_flags);
2335 fp = (fold_T *)gap->ga_data + i;
2336 flp->lnum += fp->fd_top;
2337 flp->lnum_save += fp->fd_top;
2338 flp->off -= fp->fd_top;
2339 bot += fp->fd_top;
2340 startlnum2 = flp->lnum;
2341
2342 /* This fold may end at the same line, don't incr. flp->lnum. */
2343 } else {
2344 /*
2345 * Get the level of the next line, then continue the loop to check
2346 * if it ends there.
2347 * Skip over undefined lines, to find the foldlevel after it.
2348 * For the last line in the file the foldlevel is always valid.
2349 */
2350 flp->lnum = flp->lnum_save;
2351 ll = flp->lnum + 1;
2352 while (!got_int) {
2353 /* Make the previous level available to foldlevel(). */
2354 prev_lnum = flp->lnum;
2355 prev_lnum_lvl = flp->lvl;
2356
2357 if (++flp->lnum > linecount)
2358 break;
2359 flp->lvl = flp->lvl_next;
2360 getlevel(flp);
2361 if (flp->lvl >= 0 || flp->had_end <= MAX_LEVEL20)
2362 break;
2363 }
2364 prev_lnum = 0;
2365 if (flp->lnum > linecount)
2366 break;
2367
2368 /* leave flp->lnum_save to lnum of the line that was used to get
2369 * the level, flp->lnum to the lnum of the next line. */
2370 flp->lnum_save = flp->lnum;
2371 flp->lnum = ll;
2372 }
2373 }
2374
2375 if (fp == NULL((void*)0)) /* only happens when got_int is set */
2376 return bot;
2377
2378 /*
2379 * Get here when:
2380 * lvl < level: the folds ends just above "flp->lnum"
2381 * lvl >= level: fold continues below "bot"
2382 */
2383
2384 /* Current fold at least extends until lnum. */
2385 if (fp->fd_len < flp->lnum - fp->fd_top) {
2386 fp->fd_len = flp->lnum - fp->fd_top;
2387 fp->fd_small = MAYBE2;
2388 fold_changed = TRUE1;
2389 }
2390
2391 /* Delete contained folds from the end of the last one found until where
2392 * we stopped looking. */
2393 foldRemove(&fp->fd_nested, startlnum2 - fp->fd_top,
2394 flp->lnum - 1 - fp->fd_top);
2395
2396 if (lvl < level) {
2397 /* End of fold found, update the length when it got shorter. */
2398 if (fp->fd_len != flp->lnum - fp->fd_top) {
2399 if (fp->fd_top + fp->fd_len > bot + 1) {
2400 /* fold continued below bot */
2401 if (getlevel == foldlevelMarker
2402 || getlevel == foldlevelExpr
2403 || getlevel == foldlevelSyntax) {
2404 /* marker method: truncate the fold and make sure the
2405 * previously included lines are processed again */
2406 bot = fp->fd_top + fp->fd_len - 1;
2407 fp->fd_len = flp->lnum - fp->fd_top;
2408 } else {
2409 /* indent or expr method: split fold to create a new one
2410 * below bot */
2411 i = (int)(fp - (fold_T *)gap->ga_data);
2412 foldSplit(gap, i, flp->lnum, bot);
2413 fp = (fold_T *)gap->ga_data + i;
2414 }
2415 } else
2416 fp->fd_len = flp->lnum - fp->fd_top;
2417 fold_changed = TRUE1;
2418 }
2419 }
2420
2421 /* delete following folds that end before the current line */
2422 for (;; ) {
2423 fp2 = fp + 1;
2424 if (fp2 >= (fold_T *)gap->ga_data + gap->ga_len
2425 || fp2->fd_top > flp->lnum)
2426 break;
2427 if (fp2->fd_top + fp2->fd_len > flp->lnum) {
2428 if (fp2->fd_top < flp->lnum) {
2429 /* Make fold that includes lnum start at lnum. */
2430 foldMarkAdjustRecurse(&fp2->fd_nested,
2431 (linenr_T)0, (long)(flp->lnum - fp2->fd_top - 1),
2432 (linenr_T)MAXLNUM(0x7fffffffL), (long)(fp2->fd_top - flp->lnum));
2433 fp2->fd_len -= flp->lnum - fp2->fd_top;
2434 fp2->fd_top = flp->lnum;
2435 fold_changed = TRUE1;
2436 }
2437
2438 if (lvl >= level) {
2439 /* merge new fold with existing fold that follows */
2440 foldMerge(fp, gap, fp2);
2441 }
2442 break;
2443 }
2444 fold_changed = TRUE1;
2445 deleteFoldEntry(gap, (int)(fp2 - (fold_T *)gap->ga_data), TRUE1);
2446 }
2447
2448 /* Need to redraw the lines we inspected, which might be further down than
2449 * was asked for. */
2450 if (bot < flp->lnum - 1)
2451 bot = flp->lnum - 1;
2452
2453 return bot;
2454}

2456/* foldInsert() {{{2 */
2457/*
2458 * Insert a new fold in "gap" at position "i".
2459 */
2460static void foldInsert(garray_T *gap, int i)
2461{
2462 fold_T *fp;
2463
2464 ga_grow(gap, 1);
2465
2466 fp = (fold_T *)gap->ga_data + i;
2467 if (i < gap->ga_len)
2468 memmove(fp + 1, fp, sizeof(fold_T) * (gap->ga_len - i));
2469 ++gap->ga_len;
2470 ga_init(&fp->fd_nested, (int)sizeof(fold_T), 10);
2471}

2473/* foldSplit() {{{2 */
2474/*
2475 * Split the "i"th fold in "gap", which starts before "top" and ends below
2476 * "bot" in two pieces, one ending above "top" and the other starting below
2477 * "bot".
2478 * The caller must first have taken care of any nested folds from "top" to
2479 * "bot"!
2480 */
2481static void foldSplit(garray_T *gap, int i, linenr_T top, linenr_T bot)
2482{
2483 fold_T *fp;
2484 fold_T *fp2;
2485 garray_T *gap1;
2486 garray_T *gap2;
2487 int idx;
2488 int len;
2489
2490 /* The fold continues below bot, need to split it. */
2491 foldInsert(gap, i + 1);
2492
2493 fp = (fold_T *)gap->ga_data + i;
2494 fp[1].fd_top = bot + 1;
2495 fp[1].fd_len = fp->fd_len - (fp[1].fd_top - fp->fd_top);
2496 fp[1].fd_flags = fp->fd_flags;
2497 fp[1].fd_small = MAYBE2;
2498 fp->fd_small = MAYBE2;
2499
2500 /* Move nested folds below bot to new fold. There can't be
2501 * any between top and bot, they have been removed by the caller. */
2502 gap1 = &fp->fd_nested;
2503 gap2 = &fp[1].fd_nested;
2504 (void)(foldFind(gap1, bot + 1 - fp->fd_top, &fp2));
2505 len = (int)((fold_T *)gap1->ga_data + gap1->ga_len - fp2);
2506 if (len > 0) {
2507 ga_grow(gap2, len);
2508 for (idx = 0; idx < len; ++idx) {
2509 ((fold_T *)gap2->ga_data)[idx] = fp2[idx];
2510 ((fold_T *)gap2->ga_data)[idx].fd_top
2511 -= fp[1].fd_top - fp->fd_top;
2512 }
2513 gap2->ga_len = len;
2514 gap1->ga_len -= len;
2515 }
2516 fp->fd_len = top - fp->fd_top;
2517 fold_changed = TRUE1;
2518}

2520/* foldRemove() {{{2 */
2521/*
2522 * Remove folds within the range "top" to and including "bot".
2523 * Check for these situations:
2524 * 1 2 3
2525 * 1 2 3
2526 * top 2 3 4 5
2527 * 2 3 4 5
2528 * bot 2 3 4 5
2529 * 3 5 6
2530 * 3 5 6
2531 *
2532 * 1: not changed
2533 * 2: truncate to stop above "top"
2534 * 3: split in two parts, one stops above "top", other starts below "bot".
2535 * 4: deleted
2536 * 5: made to start below "bot".
2537 * 6: not changed
2538 */
2539static void foldRemove(garray_T *gap, linenr_T top, linenr_T bot)
2540{
2541 fold_T *fp = NULL((void*)0);
2542
2543 if (bot < top)
2544 return; /* nothing to do */
2545
2546 for (;; ) {
2547 /* Find fold that includes top or a following one. */
2548 if (foldFind(gap, top, &fp) && fp->fd_top < top) {
2549 /* 2: or 3: need to delete nested folds */
2550 foldRemove(&fp->fd_nested, top - fp->fd_top, bot - fp->fd_top);
2551 if (fp->fd_top + fp->fd_len > bot + 1) {
2552 /* 3: need to split it. */
2553 foldSplit(gap, (int)(fp - (fold_T *)gap->ga_data), top, bot);
2554 } else {
2555 /* 2: truncate fold at "top". */
2556 fp->fd_len = top - fp->fd_top;
2557 }
2558 fold_changed = TRUE1;
2559 continue;
2560 }
2561 if (fp >= (fold_T *)(gap->ga_data) + gap->ga_len
2562 || fp->fd_top > bot) {
2563 /* 6: Found a fold below bot, can stop looking. */
2564 break;
2565 }
2566 if (fp->fd_top >= top) {
2567 /* Found an entry below top. */
2568 fold_changed = TRUE1;
2569 if (fp->fd_top + fp->fd_len - 1 > bot) {
2570 /* 5: Make fold that includes bot start below bot. */
2571 foldMarkAdjustRecurse(&fp->fd_nested,
2572 (linenr_T)0, (long)(bot - fp->fd_top),
2573 (linenr_T)MAXLNUM(0x7fffffffL), (long)(fp->fd_top - bot - 1));
2574 fp->fd_len -= bot - fp->fd_top + 1;
2575 fp->fd_top = bot + 1;
2576 break;
2577 }
2578
2579 /* 4: Delete completely contained fold. */
2580 deleteFoldEntry(gap, (int)(fp - (fold_T *)gap->ga_data), TRUE1);
2581 }
2582 }
2583}

2585/* foldMerge() {{{2 */
2586/*
2587 * Merge two adjacent folds (and the nested ones in them).
2588 * This only works correctly when the folds are really adjacent! Thus "fp1"
2589 * must end just above "fp2".
2590 * The resulting fold is "fp1", nested folds are moved from "fp2" to "fp1".
2591 * Fold entry "fp2" in "gap" is deleted.
2592 */
2593static void foldMerge(fold_T *fp1, garray_T *gap, fold_T *fp2)
2594{
2595 fold_T *fp3;
2596 fold_T *fp4;
2597 int idx;
2598 garray_T *gap1 = &fp1->fd_nested;
2599 garray_T *gap2 = &fp2->fd_nested;
2600
2601 /* If the last nested fold in fp1 touches the first nested fold in fp2,
2602 * merge them recursively. */
2603 if (foldFind(gap1, fp1->fd_len - 1L, &fp3) && foldFind(gap2, 0L, &fp4))
2604 foldMerge(fp3, gap2, fp4);
2605
2606 /* Move nested folds in fp2 to the end of fp1. */
2607 if (!GA_EMPTY(gap2)((gap2)->ga_len <= 0)) {
2608 ga_grow(gap1, gap2->ga_len);
2609 for (idx = 0; idx < gap2->ga_len; ++idx) {
2610 ((fold_T *)gap1->ga_data)[gap1->ga_len]
2611 = ((fold_T *)gap2->ga_data)[idx];
2612 ((fold_T *)gap1->ga_data)[gap1->ga_len].fd_top += fp1->fd_len;
2613 ++gap1->ga_len;
2614 }
2615 gap2->ga_len = 0;
2616 }
2617
2618 fp1->fd_len += fp2->fd_len;
2619 deleteFoldEntry(gap, (int)(fp2 - (fold_T *)gap->ga_data), TRUE1);
2620 fold_changed = TRUE1;
2621}
