654// This code was included to provide a portable vsnprintf() and snprintf().
655// Some systems may provide their own, but we always use this one for
656// consistency.
657//
658// This code is based on snprintf.c - a portable implementation of snprintf
659// by Mark Martinec <mark.martinec@ijs.si>, Version 2.2, 2000-10-06.
660// Included with permission. It was heavily modified to fit in Vim.
661// The original code, including useful comments, can be found here:
662//
663// http://www.ijs.si/software/snprintf/
664//
665// This snprintf() only supports the following conversion specifiers:
666// s, c, b, B, d, u, o, x, X, p (and synonyms: i, D, U, O - see below)
667// with flags: '-', '+', ' ', '0' and '#'.
668// An asterisk is supported for field width as well as precision.
669//
670// Limited support for floating point was added: 'f', 'e', 'E', 'g', 'G'.
671//
672// Length modifiers 'h' (short int), 'l' (long int) and "ll" (long long int) are
673// supported.
674//
675// The locale is not used, the string is used as a byte string. This is only
676// relevant for double-byte encodings where the second byte may be '%'.
677//
678// It is permitted for "str_m" to be zero, and it is permitted to specify NULL
679// pointer for resulting string argument if "str_m" is zero (as per ISO C99).
680//
681// The return value is the number of characters which would be generated
682// for the given input, excluding the trailing NUL. If this value
683// is greater or equal to "str_m", not all characters from the result
684// have been stored in str, output bytes beyond the ("str_m"-1) -th character
685// are discarded. If "str_m" is greater than zero it is guaranteed
686// the resulting string will be NUL-terminated.
687
688// vim_vsnprintf_typval() can be invoked with either "va_list" or a list of
689// "typval_T". When the latter is not used it must be NULL.
690
691/// Append a formatted value to the string
692///
693/// @see vim_vsnprintf_typval().
694int vim_snprintf_add(char *str, size_t str_m, const char *fmt, ...)
695 FUNC_ATTR_PRINTF(3, 4)
696{
697 const size_t len = strlen(str);
698 size_t space;
699
700 if (str_m <= len) {
701 space = 0;
702 } else {
703 space = str_m - len;
704 }
705 va_list ap;
706 va_start(ap, fmt)__builtin_va_start(ap, fmt);
707 const int str_l = vim_vsnprintf(str + len, space, fmt, ap);
708 va_end(ap)__builtin_va_end(ap);
709 return str_l;
710}

768/// Types that can be used in a format string
769static int format_typeof(const char *type)
770 FUNC_ATTR_NONNULL_ALL
771{
772 // allowed values: \0, h, l, L
773 char length_modifier = '\0';
774
775 // current conversion specifier character
776 char fmt_spec = '\0';
777
778 // parse 'h', 'l', 'll' and 'z' length modifiers
779 if (*type == 'h' || *type == 'l' || *type == 'z') {
780 length_modifier = *type;
781 type++;
782 if (length_modifier == 'l' && *type == 'l') {
783 // double l = long long
784 length_modifier = 'L';
785 type++;
786 }
787 }
788 fmt_spec = *type;
789
790 // common synonyms:
791 switch (fmt_spec) {
792 case 'i':
793 fmt_spec = 'd'; break;
794 case '*':
795 fmt_spec = 'd'; length_modifier = 'h'; break;
796 case 'D':
797 fmt_spec = 'd'; length_modifier = 'l'; break;
798 case 'U':
799 fmt_spec = 'u'; length_modifier = 'l'; break;
800 case 'O':
801 fmt_spec = 'o'; length_modifier = 'l'; break;
802 default:
803 break;
804 }
805
806 // get parameter value, do initial processing
807 switch (fmt_spec) {
808 // '%' and 'c' behave similar to 's' regarding flags and field
809 // widths
810 case '%':
811 return TYPE_PERCENT;
812
813 case 'c':
814 return TYPE_CHAR;
815
816 case 's':
817 case 'S':
818 return TYPE_STRING;
819
820 case 'd':
821 case 'u':
822 case 'b':
823 case 'B':
824 case 'o':
825 case 'x':
826 case 'X':
827 case 'p':
828 // NOTE: the u, b, o, x, X and p conversion specifiers
829 // imply the value is unsigned; d implies a signed
830 // value
831
832 // 0 if numeric argument is zero (or if pointer is
833 // NULL for 'p'), +1 if greater than zero (or nonzero
834 // for unsigned arguments), -1 if negative (unsigned
835 // argument is never negative)
836
837 if (fmt_spec == 'p') {
838 return TYPE_POINTER;
839 } else if (fmt_spec == 'b' || fmt_spec == 'B') {
840 return TYPE_UNSIGNEDLONGLONGINT;
841 } else if (fmt_spec == 'd') {
842 // signed
843 switch (length_modifier) {
844 case '\0':
845 case 'h':
846 // char and short arguments are passed as int.
847 return TYPE_INT;
848 case 'l':
849 return TYPE_LONGINT;
850 case 'L':
851 return TYPE_LONGLONGINT;
852 case 'z':
853 return TYPE_SIGNEDSIZET;
854 }
855 } else {
856 // unsigned
857 switch (length_modifier) {
858 case '\0':
859 case 'h':
860 return TYPE_UNSIGNEDINT;
861 case 'l':
862 return TYPE_UNSIGNEDLONGINT;
863 case 'L':
864 return TYPE_UNSIGNEDLONGLONGINT;
865 case 'z':
866 return TYPE_SIZET;
867 }
868 }
869 break;
870
871 case 'f':
872 case 'F':
873 case 'e':
874 case 'E':
875 case 'g':
876 case 'G':
877 return TYPE_FLOAT;
878 }
879
880 return TYPE_UNKNOWN;
881}
