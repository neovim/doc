250 * "s" and "buf" may be equal.

252void trunc_string(char_u *s, char_u *buf, int room, int buflen)
253{
254 int half;
255 int len;
256 int e;
257 int i;
258 int n;
259
260 room -= 3;
261 half = room / 2;
262 len = 0;
263
264 /* First part: Start of the string. */
265 for (e = 0; len < half && e < buflen; ++e) {
266 if (s[e] == NUL'\000') {
267 /* text fits without truncating! */
268 buf[e] = NUL'\000';
269 return;
270 }
271 n = ptr2cells(s + e);
272 if (len + n >= half)
273 break;
274 len += n;
275 buf[e] = s[e];
276 if (has_mbyte)
277 for (n = (*mb_ptr2len)(s + e); --n > 0; ) {
278 if (++e == buflen)
279 break;
280 buf[e] = s[e];
281 }
282 }
283
284 /* Last part: End of the string. */
285 i = e;
286 if (enc_dbcs != 0) {
287 /* For DBCS going backwards in a string is slow, but
288 * computing the cell width isn't too slow: go forward
289 * until the rest fits. */
290 n = vim_strsize(s + i);
291 while (len + n > room) {
292 n -= ptr2cells(s + i);
293 i += (*mb_ptr2len)(s + i);
294 }
295 } else if (enc_utf8) {
296 /* For UTF-8 we can go backwards easily. */
297 half = i = (int)STRLEN(s)strlen((char *)(s));
298 for (;; ) {

302 n = ptr2cells(s + half);

307 }
308 } else {

741 * if 'redraw' is TRUE, clear and redraw the screen
742 * if 'redraw' is FALSE, just redraw the screen
743 * if 'redraw' is -1, don't redraw at all

745void wait_return(int redraw)

758 * need_wait_return to do it later. */
759 if (msg_silent != 0)

763 * When inside vgetc(), we can't wait for a typed character at all.
764 * With the global command (and some others) we only need one return at
765 * the end. Adjust cmdline_row to avoid the next message overwriting the
766 * last one.
767 */
768 if (vgetc_busy > 0)
769 return;
770 need_wait_return = TRUE1;

796#endif

798 * to start an Ex command, but the file-changed dialog gets in the
799 * way. */
800 if (need_check_timestamps)
801 check_timestamps(FALSE0);

803 hit_return_msg();

807 * CTRL-C, but we need to loop then. */
808 had_got_int = got_int;

811 * typeahead buffer. */
812 ++no_mapping;
813 ++allow_keys;

816 * character will be recorded later, since it will be added to the

942 }
