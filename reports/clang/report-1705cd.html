670/*
671 * :move command - move lines line1-line2 to line dest
672 *
673 * return FAIL for failure, OK otherwise
674 */
675int do_move(linenr_T line1, linenr_T line2, linenr_T dest)
676{
677 char_u *str;
678 linenr_T l;
679 linenr_T extra; /* Num lines added before line1 */
680 linenr_T num_lines; /* Num lines moved */
681 linenr_T last_line; /* Last line in file after adding new text */
682
683 if (dest >= line1 && dest < line2) {
684 EMSG(_("E134: Move lines into themselves"))emsg((char_u *)(gettext((char *)("E134: Move lines into themselves"
))));
685 return FAIL0;
686 }
687
688 num_lines = line2 - line1 + 1;
689
690 /*
691 * First we copy the old text to its new location -- webb
692 * Also copy the flag that ":global" command uses.
693 */
694 if (u_save(dest, dest + 1) == FAIL0)
695 return FAIL0;
696 for (extra = 0, l = line1; l <= line2; l++) {
697 str = vim_strsave(ml_get(l + extra));
698 ml_append(dest + l - line1, str, (colnr_T)0, FALSE0);
699 free(str);
700 if (dest < line1)
701 extra++;
702 }
703
704 /*
705 * Now we must be careful adjusting our marks so that we don't overlap our
706 * mark_adjust() calls.
707 *
708 * We adjust the marks within the old text so that they refer to the
709 * last lines of the file (temporarily), because we know no other marks
710 * will be set there since these line numbers did not exist until we added
711 * our new lines.
712 *
713 * Then we adjust the marks on lines between the old and new text positions
714 * (either forwards or backwards).
715 *
716 * And Finally we adjust the marks we put at the end of the file back to
717 * their final destination at the new text position -- webb
718 */
719 last_line = curbuf->b_ml.ml_line_count;
720 mark_adjust(line1, line2, last_line - line2, 0L);
721 changed_lines(last_line - num_lines + 1, 0, last_line + 1, num_lines);
722 if (dest >= line2) {
723 mark_adjust(line2 + 1, dest, -num_lines, 0L);
724 curbuf->b_op_start.lnum = dest - num_lines + 1;
725 curbuf->b_op_end.lnum = dest;
726 } else {
727 mark_adjust(dest + 1, line1 - 1, num_lines, 0L);
728 curbuf->b_op_start.lnum = dest + 1;
729 curbuf->b_op_end.lnum = dest + num_lines;
730 }
731 curbuf->b_op_start.col = curbuf->b_op_end.col = 0;
732 mark_adjust(last_line - num_lines + 1, last_line,
733 -(last_line - dest - extra), 0L);
734 changed_lines(last_line - num_lines + 1, 0, last_line + 1, -extra);
735
736 /*
737 * Now we delete the original text -- webb
738 */
739 if (u_save(line1 + extra - 1, line2 + extra + 1) == FAIL0)
740 return FAIL0;
741
742 for (l = line1; l <= line2; l++)
743 ml_delete(line1 + extra, TRUE1);
744
745 if (!global_busy && num_lines > p_report) {
746 if (num_lines == 1)
747 MSG(_("1 line moved"))msg((char_u *)(gettext((char *)("1 line moved"))));
748 else
749 smsg((char_u *)_("%" PRId64 " lines moved")gettext((char *)("%" "l" "d" " lines moved")), (int64_t)num_lines);
750 }
751
752 /*
753 * Leave the cursor on the last of the moved lines.
754 */
755 if (dest >= line1)
756 curwin->w_cursor.lnum = dest;
757 else
758 curwin->w_cursor.lnum = dest + (line2 - line1) + 1;
759
760 if (line1 < dest) {
761 dest += num_lines + 1;
762 last_line = curbuf->b_ml.ml_line_count;
763 if (dest > last_line + 1)
764 dest = last_line + 1;
765 changed_lines(line1, 0, dest, 0L);
766 } else
767 changed_lines(dest + 1, 0, line1 + num_lines, 0L);
768
769 return OK1;
770}

959/*

1750/*
1751 * read_viminfo_up_to_marks() -- Only called from do_viminfo(). Reads in the
1752 * first part of the viminfo file which contains everything but the marks that
1753 * are local to a file. Returns TRUE when end-of-file is reached. -- webb
1754 */
1755static int read_viminfo_up_to_marks(vir_T *virp, int forceit, int writing)
1756{
1757 int eof;
1758
1759 prepare_viminfo_history(forceit ? 9999 : 0, writing);
1760 eof = viminfo_readline(virp);
1761 while (!eof && virp->vir_line[0] != '>') {
1762 switch (virp->vir_line[0]) {
1763 /* Characters reserved for future expansion, ignored now */
1764 case '+': /* "+40 /path/dir file", for running vim without args */
1765 case '|': /* to be defined */
1766 case '^': /* to be defined */
1767 case '<': /* long line - ignored */
1768 /* A comment or empty line. */
1769 case NUL'\000':
1770 case '\r':
1771 case '\n':
1772 case '#':
1773 eof = viminfo_readline(virp);
1774 break;
1775 case '*': /* "*encoding=value" */
1776 eof = viminfo_encoding(virp);
1777 break;
1778 case '!': /* global variable */
1779 eof = read_viminfo_varlist(virp, writing);
1780 break;
1781 case '%': /* entry for buffer list */
1782 eof = read_viminfo_bufferlist(virp, writing);
1783 break;
1784 case '"':
1785 eof = read_viminfo_register(virp, forceit);
1786 break;
1787 case '/': /* Search string */
1788 case '&': /* Substitute search string */
1789 case '~': /* Last search string, followed by '/' or '&' */
1790 eof = read_viminfo_search_pattern(virp, forceit);
1791 break;
1792 case '$':
1793 eof = read_viminfo_sub_string(virp, forceit);
1794 break;
1795 case ':':
1796 case '?':
1797 case '=':
1798 case '@':
1799 eof = read_viminfo_history(virp, writing);
1800 break;
1801 case '-':
1802 case '\'':
1803 eof = read_viminfo_filemark(virp, forceit);
1804 break;
1805 default:
1806 if (viminfo_error("E575: ", _("Illegal starting char")gettext((char *)("Illegal starting char")),
1807 virp->vir_line))
1808 eof = TRUE1;
1809 else
1810 eof = viminfo_readline(virp);
1811 break;
1812 }
1813 }

6247 // :sign {subcmd} {subcmd_args}

6251 p = skiptowhite(begin_subcmd_args);
