488static void pivot_right(MarkTree *b, mtnode_t *p, int i)
489{
490 mtnode_t *x = p->ptr[i], *y = p->ptr[i + 1];
491 memmove(&y->key[1], y->key, (size_t)y->n * sizeof(mtkey_t));

514static void pivot_left(MarkTree *b, mtnode_t *p, int i)
