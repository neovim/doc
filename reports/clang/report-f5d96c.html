1101static char *get_view_file(char c)
1102{
1103 if (curbuf->b_ffname == NULL((void*)0)) {

1112 // "=" -> "=="
1113 // ":" path separator -> "=-"
1114 size_t len = 0;
1115 for (char *p = sname; *p; p++) {
1116 if (*p == '=' || vim_ispathsep(*p)) {
1117 len++;
1118 }

1121 STRCPY(retval, p_vdir)strcpy((char *)(retval), (char *)(p_vdir));
1122 add_pathsep(retval);
1123 char *s = retval + strlen(retval);
1124 for (char *p = sname; *p; p++) {
1125 if (*p == '=') {
1126 *s++ = '=';
1127 *s++ = '=';
1128 } else if (vim_ispathsep(*p)) {
1129 *s++ = '=';
1130#if defined(BACKSLASH_IN_FILENAME)
