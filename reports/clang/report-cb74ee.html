1058/// do_filter: filter lines through a command given by the user
1059///
1060/// We mostly use temp files and the call_shell() routine here. This would
1061/// normally be done using pipes on a Unix system, but this is more portable
1062/// to non-Unix systems. The call_shell() routine needs to be able
1063/// to deal with redirection somehow, and should handle things like looking
1064/// at the PATH env. variable, and adding reasonable extensions to the
1065/// command name given by the user. All reasonable versions of call_shell()
1066/// do this.
1067/// Alternatively, if on Unix and redirecting input or output, but not both,
1068/// and the 'shelltemp' option isn't set, use pipes.
1069/// We use input redirection if do_in is true.
1070/// We use output redirection if do_out is true.
1071///
1072/// @param eap for forced 'ff' and 'fenc'
1073static void do_filter(linenr_T line1, linenr_T line2, exarg_T *eap, char *cmd, bool_Bool do_in,
1074 bool_Bool do_out)
1075{

1079 int shell_flags = 0;
1080 const pos_T orig_start = curbuf->b_op_start;
1081 const pos_T orig_end = curbuf->b_op_end;
1082 const int stmp = p_stmp;
1083
1084 if (*cmd == NUL'\000') { // no filter command
1085 return;
1086 }
1087
1088 const int save_cmod_flags = cmdmod.cmod_flags;
1089 // Temporarily disable lockmarks since that's needed to propagate changed
1090 // regions of the buffer for foldUpdate(), linecount, etc.
1091 cmdmod.cmod_flags &= ~CMOD_LOCKMARKS;
1092
1093 pos_T cursor_save = curwin->w_cursor;
1094 linenr_T linecount = line2 - line1 + 1;
1095 curwin->w_cursor.lnum = line1;
1096 curwin->w_cursor.col = 0;
1097 changed_line_abv_curs();
1098 invalidate_botline(curwin);
1099
1100 // When using temp files:
1101 // 1. * Form temp file names
1102 // 2. * Write the lines to a temp file
1103 // 3. Run the filter command on the temp file
1104 // 4. * Read the output of the command into the buffer
1105 // 5. * Delete the original lines to be filtered
1106 // 6. * Remove the temp files
1107 //
1108 // When writing the input with a pipe or when catching the output with a
1109 // pipe only need to do 3.
1110
1111 if (do_out) {
1112 shell_flags |= kShellOptDoOut;
1113 }
1114
1115 if (!do_in && do_out && !stmp) {
1116 // Use a pipe to fetch stdout of the command, do not use a temp file.
1117 shell_flags |= kShellOptRead;
1118 curwin->w_cursor.lnum = line2;
1119 } else if (do_in && !do_out && !stmp) {
1120 // Use a pipe to write stdin of the command, do not use a temp file.
1121 shell_flags |= kShellOptWrite;
1122 curbuf->b_op_start.lnum = line1;
1123 curbuf->b_op_end.lnum = line2;
1124 } else if (do_in && do_out && !stmp) {
1125 // Use a pipe to write stdin and fetch stdout of the command, do not
1126 // use a temp file.
1127 shell_flags |= kShellOptRead | kShellOptWrite;
1128 curbuf->b_op_start.lnum = line1;
1129 curbuf->b_op_end.lnum = line2;
1130 curwin->w_cursor.lnum = line2;
1131 } else if ((do_in && (itmp = vim_tempname()) == NULL((void*)0))
1132 || (do_out && (otmp = vim_tempname()) == NULL((void*)0))) {
1133 emsg(_(e_notmp)gettext(e_notmp));
1134 goto filterend;
1135 }
1136
1137 // The writing and reading of temp files will not be shown.
1138 // Vi also doesn't do this and the messages are not very informative.
1139 no_wait_return++; // don't call wait_return() while busy
1140 if (itmp != NULL((void*)0) && buf_write(curbuf, itmp, NULL((void*)0), line1, line2, eap,
1141 false0, false0, false0, true1) == FAIL0) {

1351/// Create a shell command from a command string, input redirection file and
1352/// output redirection file.
1353///
1354/// @param cmd Command to execute.
1355/// @param itmp NULL or the input file.
1356/// @param otmp NULL or the output file.
1357/// @returns an allocated string with the shell command.
1358char *make_filter_cmd(char *cmd, char *itmp, char *otmp)
1359{
1360 bool_Bool is_fish_shell =
1361#if defined(UNIX)
1362 strncmp(invocation_path_tail(p_sh, NULL((void*)0)), "fish", 4) == 0;
1363#else
1364 false0;
1365#endif
1366 bool_Bool is_pwsh = strncmp(invocation_path_tail(p_sh, NULL((void*)0)), "pwsh", 4) == 0
1367 || strncmp(invocation_path_tail(p_sh, NULL((void*)0)), "powershell",
1368 10) == 0;
1369
1370 size_t len = strlen(cmd) + 1; // At least enough space for cmd + NULL.
1371
1372 len += is_fish_shell ? sizeof("begin; " "; end") - 1
1373 : !is_pwsh ? sizeof("(" ")") - 1
1374 : 0;
1375
1376 if (itmp != NULL((void*)0)) {
1377 len += is_pwsh ? strlen(itmp) + sizeof("& { Get-Content " " | & " " }") - 1 + 6 // +6: #20530
1378 : strlen(itmp) + sizeof(" { " " < " " } ") - 1;
1379 }
1380 if (otmp != NULL((void*)0)) {
1381 len += strlen(otmp) + strlen(p_srr) + 2; // two extra spaces (" "),
1382 }
1383
1384 char *const buf = xmalloc(len);
1385
1386 if (is_pwsh) {
1387 if (itmp != NULL((void*)0)) {
1388 xstrlcpy(buf, "& { Get-Content ", len - 1); // FIXME: should we add "-Encoding utf8"?
1389 xstrlcat(buf, itmp, len - 1);
1390 xstrlcat(buf, " | & ", len - 1); // FIXME: add `&` ourself or leave to user?
1391 xstrlcat(buf, cmd, len - 1);
1392 xstrlcat(buf, " }", len - 1);
1393 } else {
1394 xstrlcpy(buf, cmd, len - 1);
1395 }
1396 } else {
1397#if defined(UNIX)
1398 // Put delimiters around the command (for concatenated commands) when
1399 // redirecting input and/or output.
1400 if (itmp != NULL((void*)0) || otmp != NULL((void*)0)) {
1401 char *fmt = is_fish_shell ? "begin; %s; end"
1402 : "(%s)";
1403 vim_snprintf(buf, len, fmt, cmd);
1404 } else {
1405 xstrlcpy(buf, cmd, len);
1406 }
1407
1408 if (itmp != NULL((void*)0)) {
1409 xstrlcat(buf, " < ", len - 1);
1410 xstrlcat(buf, itmp, len - 1);
1411 }
1412#else
1413 // For shells that don't understand braces around commands, at least allow
1414 // the use of commands in a pipe.
1415 xstrlcpy(buf, cmd, len);
1416 if (itmp != NULL((void*)0)) {
1417 // If there is a pipe, we have to put the '<' in front of it.
1418 // Don't do this when 'shellquote' is not empty, otherwise the
1419 // redirection would be inside the quotes.
1420 if (*p_shq == NUL'\000') {
1421 char *const p = find_pipe(buf);
1422 if (p != NULL((void*)0)) {
1423 *p = NUL'\000';
1424 }
1425 }
1426 xstrlcat(buf, " < ", len);
1427 xstrlcat(buf, itmp, len);
1428 if (*p_shq == NUL'\000') {
1429 const char *const p = find_pipe(cmd);
1430 if (p != NULL((void*)0)) {
1431 xstrlcat(buf, " ", len - 1); // Insert a space before the '|' for DOS
1432 xstrlcat(buf, p, len - 1);
1433 }
1434 }
1435 }
1436#endif
1437 }
1438 if (otmp != NULL((void*)0)) {
1439 append_redir(buf, len, p_srr, otmp);
1440 }
1441 return buf;
1442}
