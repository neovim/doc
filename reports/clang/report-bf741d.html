670/*
671 * :move command - move lines line1-line2 to line dest
672 *
673 * return FAIL for failure, OK otherwise
674 */
675int do_move(linenr_T line1, linenr_T line2, linenr_T dest)
676{
677 char_u *str;
678 linenr_T l;
679 linenr_T extra; /* Num lines added before line1 */
680 linenr_T num_lines; /* Num lines moved */
681 linenr_T last_line; /* Last line in file after adding new text */
682
683 if (dest >= line1 && dest < line2) {
684 EMSG(_("E134: Move lines into themselves"))emsg((char_u *)(gettext((char *)("E134: Move lines into themselves"
))));
685 return FAIL0;
686 }
687
688 num_lines = line2 - line1 + 1;
689
690 /*
691 * First we copy the old text to its new location -- webb
692 * Also copy the flag that ":global" command uses.
693 */
694 if (u_save(dest, dest + 1) == FAIL0)
695 return FAIL0;
696 for (extra = 0, l = line1; l <= line2; l++) {
697 str = vim_strsave(ml_get(l + extra));
698 ml_append(dest + l - line1, str, (colnr_T)0, FALSE0);
699 free(str);
700 if (dest < line1)
701 extra++;
702 }
703
704 /*
705 * Now we must be careful adjusting our marks so that we don't overlap our
706 * mark_adjust() calls.
707 *
708 * We adjust the marks within the old text so that they refer to the
709 * last lines of the file (temporarily), because we know no other marks
710 * will be set there since these line numbers did not exist until we added
711 * our new lines.
712 *
713 * Then we adjust the marks on lines between the old and new text positions
714 * (either forwards or backwards).
715 *
716 * And Finally we adjust the marks we put at the end of the file back to
717 * their final destination at the new text position -- webb
718 */
719 last_line = curbuf->b_ml.ml_line_count;
720 mark_adjust(line1, line2, last_line - line2, 0L);
721 changed_lines(last_line - num_lines + 1, 0, last_line + 1, num_lines);
722 if (dest >= line2) {
723 mark_adjust(line2 + 1, dest, -num_lines, 0L);
724 curbuf->b_op_start.lnum = dest - num_lines + 1;
725 curbuf->b_op_end.lnum = dest;
726 } else {
727 mark_adjust(dest + 1, line1 - 1, num_lines, 0L);
728 curbuf->b_op_start.lnum = dest + 1;
729 curbuf->b_op_end.lnum = dest + num_lines;
730 }
731 curbuf->b_op_start.col = curbuf->b_op_end.col = 0;
732 mark_adjust(last_line - num_lines + 1, last_line,
733 -(last_line - dest - extra), 0L);
734 changed_lines(last_line - num_lines + 1, 0, last_line + 1, -extra);
735
736 /*
737 * Now we delete the original text -- webb
738 */
739 if (u_save(line1 + extra - 1, line2 + extra + 1) == FAIL0)
740 return FAIL0;
741
742 for (l = line1; l <= line2; l++)
743 ml_delete(line1 + extra, TRUE1);
744
745 if (!global_busy && num_lines > p_report) {
746 if (num_lines == 1)
747 MSG(_("1 line moved"))msg((char_u *)(gettext((char *)("1 line moved"))));
748 else
749 smsg((char_u *)_("%" PRId64 " lines moved")gettext((char *)("%" "l" "d" " lines moved")), (int64_t)num_lines);
750 }
751
752 /*
753 * Leave the cursor on the last of the moved lines.
754 */
755 if (dest >= line1)
756 curwin->w_cursor.lnum = dest;
757 else
758 curwin->w_cursor.lnum = dest + (line2 - line1) + 1;
759
760 if (line1 < dest) {
761 dest += num_lines + 1;
762 last_line = curbuf->b_ml.ml_line_count;
763 if (dest > last_line + 1)
764 dest = last_line + 1;
765 changed_lines(line1, 0, dest, 0L);
766 } else
767 changed_lines(dest + 1, 0, line1 + num_lines, 0L);
768
769 return OK1;
770}

834 * Handle the ":!cmd" command. Also for ":r !cmd" and ":w !cmd"
835 * Bangs in the argument are replaced with the previously entered command.
836 * Remember the argument.
837 */
838void do_bang(int addr_count, exarg_T *eap, int forceit, int do_in, int do_out)
839{
840 char_u *arg = eap->arg; /* command */
841 linenr_T line1 = eap->line1; /* start of range */
842 linenr_T line2 = eap->line2; /* end of range */
843 char_u *newcmd = NULL((void*)0); /* the new command */

845 int ins_prevcmd;

851
852 /*
853 * Disallow shell commands for "rvim".
854 * Disallow shell commands from .exrc and .vimrc in current directory for
855 * security reasons.

857 if (check_restricted() || check_secure())

867 * Try to find an embedded bang, like in :!<cmd> ! [args]
868 * (:!! is indicated by the 'forceit' variable)

873 len = (int)STRLEN(trailarg)strlen((char *)(trailarg)) + 1;

892 free(newcmd);

912 }
913 } while (trailarg != NULL((void*)0));

915 free(prevcmd);
916 prevcmd = newcmd;
917
918 if (bangredo) { /* put cmd in redo buffer for ! command */
919 /* If % or # appears in the command, it must have been escaped.
920 * Reescape them, so that redoing them does not substitute them by the
921 * buffername. */
922 char_u *cmd = vim_strsave_escaped(prevcmd, (char_u *)"%#");
923

925 free(cmd);
926 AppendToRedobuff((char_u *)"\n");
927 bangredo = FALSE0;
928 }

930 * Add quotes around the command, for shells that need them.
931 */
932 if (*p_shq != NUL'\000') {
933 newcmd = xmalloc(STRLEN(prevcmd)strlen((char *)(prevcmd)) + 2 * STRLEN(p_shq)strlen((char *)(p_shq)) + 1);
934 STRCPY(newcmd, p_shq)strcpy((char *)(newcmd), (char *)(p_shq));
935 STRCAT(newcmd, prevcmd)strcat((char *)(newcmd), (char *)(prevcmd));
936 STRCAT(newcmd, p_shq)strcat((char *)(newcmd), (char *)(p_shq));
937 free_newcmd = TRUE1;
938 }
939 if (addr_count == 0) { /* :! */
940 /* echo the command */
941 msg_start();
942 msg_putchar(':');
943 msg_putchar('!');
944 msg_outtrans(newcmd);
945 msg_clr_eos();
946 windgoto(msg_row, msg_col);
947
948 do_shell(newcmd, 0);
949 } else { /* :range! */
950 /* Careful: This may recursively call do_bang() again! (because of
951 * autocommands) */
952 do_filter(line1, line2, eap, newcmd, do_in, do_out);
953 apply_autocmds(EVENT_SHELLFILTERPOST, NULL((void*)0), NULL((void*)0), FALSE0, curbuf);
954 }
955 if (free_newcmd)
956 free(newcmd);
957}

959/*
960 * do_filter: filter lines through a command given by the user
961 *
962 * We mostly use temp files and the call_shell() routine here. This would
963 * normally be done using pipes on a UNIX machine, but this is more portable
964 * to non-unix machines. The call_shell() routine needs to be able
965 * to deal with redirection somehow, and should handle things like looking
966 * at the PATH env. variable, and adding reasonable extensions to the
967 * command name given by the user. All reasonable versions of call_shell()
968 * do this.
969 * Alternatively, if on Unix and redirecting input or output, but not both,
970 * and the 'shelltemp' option isn't set, use pipes.
971 * We use input redirection if do_in is TRUE.
972 * We use output redirection if do_out is TRUE.
973 */

975do_filter (
976 linenr_T line1,

978 exarg_T *eap, /* for forced 'ff' and 'fenc' */

991 int shell_flags = 0;

997 cursor_save = curwin->w_cursor;

1000 curwin->w_cursor.col = 0;
1001 changed_line_abv_curs();

1005 * When using temp files:
1006 * 1. * Form temp file names
1007 * 2. * Write the lines to a temp file
1008 * 3. Run the filter command on the temp file
1009 * 4. * Read the output of the command into the buffer
1010 * 5. * Delete the original lines to be filtered
1011 * 6. * Remove the temp files
1012 *

1014 * pipe only need to do 3.
1015 */

1023 curwin->w_cursor.lnum = line2;
1024 } else if (do_in && !do_out && !p_stmp) {
1025 // Use a pipe to write stdin of the command, do not use a temp file.
1026 shell_flags |= kShellOptWrite;
1027 curbuf->b_op_start.lnum = line1;
1028 curbuf->b_op_end.lnum = line2;
1029 } else if (do_in && do_out && !p_stmp) {

1031 // use a temp file.

1037 || (do_out && (otmp = vim_tempname()) == NULL((void*)0))) {
1038 EMSG(_(e_notmp))emsg((char_u *)(gettext((char *)(e_notmp))));
1039 goto filterend;
1040 }

1043 * The writing and reading of temp files will not be shown.
1044 * Vi also doesn't do this and the messages are not very informative.
1045 */
1046 ++no_wait_return; /* don't call wait_return() while busy */

1152 * Adjust '[ and '] (set by buf_write()).
1153 */

1175 _("%" PRId64 " lines filtered")gettext((char *)("%" "l" "d" " lines filtered")), (int64_t)linecount);
1176 if (msg(msg_buf) && !msg_scroll)
1177 /* save message to display it after redraw */
1178 set_keep_msg(msg_buf, 0);
1179 } else
1180 msgmore((long)linecount);
1181 }
1182 } else {
1183error:
1184 /* put cursor back in same position for ":w !cmd" */

1198 if (otmp != NULL((void*)0))

1200 free(itmp);

1202}

1750/*

1752 * first part of the viminfo file which contains everything but the marks that
1753 * are local to a file. Returns TRUE when end-of-file is reached. -- webb

1755static int read_viminfo_up_to_marks(vir_T *virp, int forceit, int writing)
1756{

1758

1760 eof = viminfo_readline(virp);
1761 while (!eof && virp->vir_line[0] != '>') {
1762 switch (virp->vir_line[0]) {
1763 /* Characters reserved for future expansion, ignored now */
1764 case '+': /* "+40 /path/dir file", for running vim without args */
1765 case '|': /* to be defined */
1766 case '^': /* to be defined */
1767 case '<': /* long line - ignored */
1768 /* A comment or empty line. */
1769 case NUL'\000':
1770 case '\r':
1771 case '\n':
1772 case '#':

1778 case '!': /* global variable */
1779 eof = read_viminfo_varlist(virp, writing);
1780 break;
1781 case '%': /* entry for buffer list */
1782 eof = read_viminfo_bufferlist(virp, writing);
1783 break;
1784 case '"':
1785 eof = read_viminfo_register(virp, forceit);
1786 break;
1787 case '/': /* Search string */
1788 case '&': /* Substitute search string */
1789 case '~': /* Last search string, followed by '/' or '&' */

1870 */

1876)

4864 * weighting than the number of characters.

6228 */
6229void set_context_in_sign_cmd(expand_T *xp, char_u *arg)
6230{

6234 int cmd_idx;
6235 char_u *begin_subcmd_args;
6236
6237 /* Default: expand subcommands. */
6238 xp->xp_context = EXPAND_SIGN;
6239 expand_what = EXP_SUBCMD;
6240 xp->xp_pattern = arg;
6241
6242 end_subcmd = skiptowhite(arg);
6243 if (*end_subcmd == NUL'\000')
6244 /* expand subcmd name
6245 * :sign {subcmd}<CTRL-D>*/
6246 return;
6247
6248 cmd_idx = sign_cmd_idx(arg, end_subcmd);
6249
6250 // :sign {subcmd} {subcmd_args}
6251 // |
6252 // begin_subcmd_args
6253 begin_subcmd_args = skipwhite(end_subcmd);
6254 p = skiptowhite(begin_subcmd_args);
6255 if (*p == NUL'\000')
6256 {
6257 /*
6258 * Expand first argument of subcmd when possible.
6259 * For ":jump {id}" and ":unplace {id}", we could
6260 * possibly expand the ids of all signs already placed.
6261 */
6262 xp->xp_pattern = begin_subcmd_args;
6263 switch (cmd_idx)
6264 {
6265 case SIGNCMD_LIST2:
6266 case SIGNCMD_UNDEFINE1:
6267 /* :sign list <CTRL-D>
6268 * :sign undefine <CTRL-D> */

6277 // Expand last argument of subcmd.
6278 //
6279 // :sign define {name} {args}...
6280 // |
6281 // p
6282
6283 // Loop until reaching last argument.
6284 do
6285 {
6286 p = skipwhite(p);
6287 last = p;
6288 p = skiptowhite(p);
6289 } while (*p != NUL'\000');
6290
6291 p = vim_strchr(last, '=');
6292
6293 // :sign define {name} {args}... {last}=
6294 // | |
6295 // last p
6296 if (p == NUL'\000')
6297 {
6298 /* Expand last argument name (before equal sign). */
6299 xp->xp_pattern = last;
6300 switch (cmd_idx)
6301 {
6302 case SIGNCMD_DEFINE0:
6303 expand_what = EXP_DEFINE;
6304 break;
6305 case SIGNCMD_PLACE3:
6306 expand_what = EXP_PLACE;
6307 break;
6308 case SIGNCMD_JUMP5:
6309 case SIGNCMD_UNPLACE4:
6310 expand_what = EXP_UNPLACE;
6311 break;

6314 }
6315 }
6316 else
6317 {
6318 /* Expand last argument value (after equal sign). */
6319 xp->xp_pattern = p + 1;

6321 {
6322 case SIGNCMD_DEFINE0:

6333 expand_what = EXP_SIGN_NAMES;
6334 else
6335 xp->xp_context = EXPAND_NOTHING;
6336 break;
6337 default:
6338 xp->xp_context = EXPAND_NOTHING;
