71/*
72** Transform left associative constructions into right
73** associative ones, for sequence and choice; that is:
74** (t11 + t12) + t2 => t11 + (t12 + t2)
75** (t11 * t12) * t2 => t11 * (t12 * t2)
76** (that is, Op (Op t11 t12) t2 => Op t11 (Op t12 t2))
77*/
78static void correctassociativity (TTree *tree) {
79 TTree *t1 = sib1(tree)((tree) + 1);
80 assert(tree->tag == TChoice || tree->tag == TSeq)((void) (0));
81 while (t1->tag == tree->tag) {
82 int n1size = tree->u.ps - 1; /* t1 == Op t11 t12 */
83 int n11size = t1->u.ps - 1;
84 int n12size = n1size - n11size - 1;
85 memmove(sib1(tree)((tree) + 1), sib1(t1)((t1) + 1), n11size * sizeof(TTree)); /* move t11 */

865/*
866** push on the stack the index and the pattern for the
867** initial rule of grammar at index 'arg' in the stack;
868** also add that index into position table.
869*/
870static void getfirstrule (lua_State *L, int arg, int postab) {
871 lua_rawgeti(L, arg, 1); /* access first element */
872 if (lua_isstring(L, -1)) { /* is it the name of initial rule? */
873 lua_pushvalue(L, -1); /* duplicate it to use as key */
874 lua_gettable(L, arg); /* get associated rule */
875 }
876 else {
877 lua_pushinteger(L, 1); /* key for initial rule */
878 lua_insert(L, -2); /* put it before rule */
879 }
880 if (!testpattern(L, -1)) { /* initial rule not a pattern? */
881 if (lua_isnil(L, -1)(lua_type(L, (-1)) == 0))
882 luaL_error(L, "grammar has no initial rule");
883 else
884 luaL_error(L, "initial rule '%s' is not a pattern", lua_tostring(L, -2)lua_tolstring(L, (-2), ((void*)0)));
885 }
886 lua_pushvalue(L, -2); /* push key */
887 lua_pushinteger(L, 1); /* push rule position (after TGrammar) */
888 lua_settable(L, postab); /* insert pair at position table */
889}

891/*
892** traverse grammar at index 'arg', pushing all its keys and patterns
893** into the stack. Create a new table (before all pairs key-pattern) to
894** collect all keys and their associated positions in the final tree
895** (the "position table").
896** Return the number of rules and (in 'totalsize') the total size
897** for the new tree.
898*/
899static int collectrules (lua_State *L, int arg, int *totalsize) {
900 int n = 1; /* to count number of rules */
901 int postab = lua_gettop(L) + 1; /* index of position table */
902 int size; /* accumulator for total size */
903 lua_newtable(L)lua_createtable(L, 0, 0); /* create position table */
904 getfirstrule(L, arg, postab);
905 size = 2 + getsize(L, postab + 2); /* TGrammar + TRule + rule */
906 lua_pushnil(L); /* prepare to traverse grammar table */
907 while (lua_next(L, arg) != 0) {
908 if (lua_tonumber(L, -2) == 1 ||
909 lua_equal(L, -2, postab + 1)) { /* initial rule? */
910 lua_pop(L, 1)lua_settop(L, -(1)-1); /* remove value (keep key for lua_next) */
911 continue;
912 }
913 if (!testpattern(L, -1)) /* value is not a pattern? */
914 luaL_error(L, "rule '%s' is not a pattern", val2str(L, -2));
915 luaL_checkstack(L, LUA_MINSTACK20, "grammar has too many rules");
916 lua_pushvalue(L, -2); /* push key (to insert into position table) */
917 lua_pushinteger(L, size);
918 lua_settable(L, postab);
919 size += 1 + getsize(L, -1); /* update size */
920 lua_pushvalue(L, -2); /* push key (for next lua_next) */
921 n++;
922 }
923 *totalsize = size + 1; /* TTrue to finish list of rules */
924 return n;
925}

984/*
985** Check whether a rule can be left recursive; raise an error in that
986** case; otherwise return 1 iff pattern is nullable. Assume ktable at
987** the top of the stack.
988*/
989static int verifyrule (lua_State *L, TTree *tree, int *passed, int npassed,
990 int nullable) {
991 tailcall:
992 switch (tree->tag) {
993 case TChar: case TSet: case TAny:
994 case TFalse:
995 return nullable; /* cannot pass from here */
996 case TTrue:
997 case TBehind: /* look-behind cannot have calls */
998 return 1;
999 case TNot: case TAnd: case TRep:
1000 /* return verifyrule(L, sib1(tree), passed, npassed, 1); */
1001 tree = sib1(tree)((tree) + 1); nullable = 1; goto tailcall;
1002 case TCapture: case TRunTime:
1003 /* return verifyrule(L, sib1(tree), passed, npassed); */
1004 tree = sib1(tree)((tree) + 1); goto tailcall;
1005 case TCall:
1006 /* return verifyrule(L, sib2(tree), passed, npassed); */
1007 tree = sib2(tree)((tree) + (tree)->u.ps); goto tailcall;
1008 case TSeq: /* only check 2nd child if first is nullable */
1009 if (!verifyrule(L, sib1(tree)((tree) + 1), passed, npassed, 0))
1010 return nullable;
1011 /* else return verifyrule(L, sib2(tree), passed, npassed); */
1012 tree = sib2(tree)((tree) + (tree)->u.ps); goto tailcall;
1013 case TChoice: /* must check both children */
1014 nullable = verifyrule(L, sib1(tree)((tree) + 1), passed, npassed, nullable);
1015 /* return verifyrule(L, sib2(tree), passed, npassed, nullable); */
1016 tree = sib2(tree)((tree) + (tree)->u.ps); goto tailcall;
1017 case TRule:
1018 if (npassed >= MAXRULES1000)
1019 return verifyerror(L, passed, npassed);
1020 else {
1021 passed[npassed++] = tree->key;
1022 /* return verifyrule(L, sib1(tree), passed, npassed); */
1023 tree = sib1(tree)((tree) + 1); goto tailcall;
1024 }
1025 case TGrammar:
1026 return nullable(tree)checkaux(tree, 0); /* sub-grammar cannot be left recursive */
1027 default: assert(0)((void) (0)); return 0;
1028 }
1029}

1032static void verifygrammar (lua_State *L, TTree *grammar) {
1033 int passed[MAXRULES1000];
1034 TTree *rule;
1035 /* check left-recursive rules */
1036 for (rule = sib1(grammar)((grammar) + 1); rule->tag == TRule; rule = sib2(rule)((rule) + (rule)->u.ps)) {
1037 if (rule->key == 0) continue; /* unused rule */
1038 verifyrule(L, sib1(rule)((rule) + 1), passed, 0, 0);
1039 }
1040 assert(rule->tag == TTrue)((void) (0));
1041 /* check infinite loops inside rules */
1042 for (rule = sib1(grammar)((grammar) + 1); rule->tag == TRule; rule = sib2(rule)((rule) + (rule)->u.ps)) {
1043 if (rule->key == 0) continue; /* unused rule */
1044 if (checkloops(sib1(rule)((rule) + 1))) {
1045 lua_rawgeti(L, -1, rule->key); /* get rule's key */
1046 luaL_error(L, "empty loop in rule '%s'", val2str(L, -1));
1047 }
1048 }
1049 assert(rule->tag == TTrue)((void) (0));
1050}
