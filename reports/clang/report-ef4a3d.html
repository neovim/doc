1909/// Return true if "str" starts with a backslash that should be removed.
1910/// For WIN32 this is only done when the character after the
1911/// backslash is not a normal file name character.
1912/// '$' is a valid file name character, we don't remove the backslash before
1913/// it. This means it is not possible to use an environment variable after a
1914/// backslash. "C:\$VIM\doc" is taken literally, only "$VIM\doc" works.
1915/// Although "\ name" is valid, the backslash in "Program\ files" must be
1916/// removed. Assume a file name doesn't start with a space.
1917/// For multi-byte names, never remove a backslash before a non-ascii
1918/// character, assume that all multi-byte characters are valid file name
1919/// characters.
1920///
1921/// @param str
1922///
1923/// @return true if `str` starts with a backslash that should be removed.
1924bool_Bool rem_backslash(const char_u *str)
1925{
1926#ifdef BACKSLASH_IN_FILENAME
1927 return str[0] == '\\'
1928 && str[1] < 0x80
1929 && (str[1] == ' '
1930 || (str[1] != NUL'\000'
1931 && str[1] != '*'
1932 && str[1] != '?'
1933 && !vim_isfilec(str[1])));
1934
1935#else // ifdef BACKSLASH_IN_FILENAME
1936 return str[0] == '\\' && str[1] != NUL'\000';
1937#endif // ifdef BACKSLASH_IN_FILENAME
1938}
