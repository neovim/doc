461int termkey_start(TermKey *tk)
462{
463 if(tk->is_started)
464 return 1;
465
466 if(tk->fd != -1 && !(tk->flags & TERMKEY_FLAG_NOTERMIOS)) {
467 struct termios termios;
468 if(tcgetattr(tk->fd, &termios) == 0) {
469 tk->restore_termios = termios;
470 tk->restore_termios_valid = 1;
471
472 termios.c_iflag &= ~(IXON0002000|INLCR0000100|ICRNL0000400);
473 termios.c_lflag &= ~(ICANON0000002|ECHO0000010);
474 termios.c_cc[VMIN6] = 1;
475 termios.c_cc[VTIME5] = 0;
476
477 if(tk->flags & TERMKEY_FLAG_CTRLC)
478 /* want no signal keys at all, so just disable ISIG */
479 termios.c_lflag &= ~ISIG0000001;
480 else {
481 /* Disable Ctrl-\==VQUIT and Ctrl-D==VSUSP but leave Ctrl-C as SIGINT */
482 termios.c_cc[VQUIT1] = _POSIX_VDISABLE'\0';
483 termios.c_cc[VSUSP10] = _POSIX_VDISABLE'\0';
484 /* Some OSes have Ctrl-Y==VDSUSP */
485#ifdef VDSUSP
486 termios.c_cc[VDSUSP] = _POSIX_VDISABLE'\0';
487#endif
488 }
489
490#ifdef DEBUG
491 fprintf(stderrstderr, "Setting termios(3) flags\n");
492#endif
493 tcsetattr(tk->fd, TCSANOW0, &termios);
494 }
495 }
496
497 struct TermKeyDriverNode *p;
498 for(p = tk->drivers; p; p = p->next)
499 if(p->driver->start_driver)
500 if(!(*p->driver->start_driver)(tk, p->info))
501 return 0;
502
503#ifdef DEBUG
504 fprintf(stderrstderr, "Drivers started; termkey instance %p is ready\n", tk);
505#endif
506
507 tk->is_started = 1;
508 return 1;

615static inline unsigned int utf8_seqlen(long codepoint)
616{
617 if(codepoint < 0x0000080) return 1;
618 if(codepoint < 0x0000800) return 2;
619 if(codepoint < 0x0010000) return 3;
620 if(codepoint < 0x0200000) return 4;
621 if(codepoint < 0x4000000) return 5;
622 return 6;

625static void fill_utf8(TermKeyKey *key)
626{
627 long codepoint = key->code.codepoint;
628 int nbytes = utf8_seqlen(codepoint);
629
630 key->utf8[nbytes] = 0;
631
632 // This is easier done backwards
633 int b = nbytes;
634 while(b > 1) {
635 b--;
636 key->utf8[b] = 0x80 | (codepoint & 0x3f);
637 codepoint >>= 6;
638 }
639
640 switch(nbytes) {
641 case 1: key->utf8[0] = (codepoint & 0x7f); break;
642 case 2: key->utf8[0] = 0xc0 | (codepoint & 0x1f); break;
643 case 3: key->utf8[0] = 0xe0 | (codepoint & 0x0f); break;
644 case 4: key->utf8[0] = 0xf0 | (codepoint & 0x07); break;
645 case 5: key->utf8[0] = 0xf8 | (codepoint & 0x03); break;
646 case 6: key->utf8[0] = 0xfc | (codepoint & 0x01); break;
647 }
648}

651static TermKeyResult parse_utf8(const unsigned char *bytes, size_t len, long *cp, size_t *nbytep)
652{
653 unsigned int nbytes;
654
655 unsigned char b0 = bytes[0];
656
657 if(b0 < 0x80) {
658 // Single byte ASCII
659 *cp = b0;
660 *nbytep = 1;
661 return TERMKEY_RES_KEY;
662 }
663 else if(b0 < 0xc0) {
664 // Starts with a continuation byte - that's not right
665 *cp = UTF8_INVALID0xFFFD;
666 *nbytep = 1;
667 return TERMKEY_RES_KEY;
668 }
669 else if(b0 < 0xe0) {
670 nbytes = 2;
671 *cp = b0 & 0x1f;
672 }
673 else if(b0 < 0xf0) {
674 nbytes = 3;
675 *cp = b0 & 0x0f;
676 }
677 else if(b0 < 0xf8) {
678 nbytes = 4;
679 *cp = b0 & 0x07;
680 }
681 else if(b0 < 0xfc) {
682 nbytes = 5;
683 *cp = b0 & 0x03;
684 }
685 else if(b0 < 0xfe) {
686 nbytes = 6;
687 *cp = b0 & 0x01;
688 }
689 else {
690 *cp = UTF8_INVALID0xFFFD;
691 *nbytep = 1;
692 return TERMKEY_RES_KEY;
693 }
694
695 for(unsigned int b = 1; b < nbytes; b++) {
696 unsigned char cb;
697
698 if(b >= len)
699 return TERMKEY_RES_AGAIN;
700
701 cb = bytes[b];
702 if(cb < 0x80 || cb >= 0xc0) {
703 *cp = UTF8_INVALID0xFFFD;
704 *nbytep = b;
705 return TERMKEY_RES_KEY;
706 }
707
708 *cp <<= 6;
709 *cp |= cb & 0x3f;
710 }
711
712 // Check for overlong sequences
713 if(nbytes > utf8_seqlen(*cp))
714 *cp = UTF8_INVALID0xFFFD;
715
716 // Check for UTF-16 surrogates or invalid *cps
717 if((*cp >= 0xD800 && *cp <= 0xDFFF) ||
718 *cp == 0xFFFE ||
719 *cp == 0xFFFF)
720 *cp = UTF8_INVALID0xFFFD;
721
722 *nbytep = nbytes;
723 return TERMKEY_RES_KEY;
724}

726static void emit_codepoint(TermKey *tk, long codepoint, TermKeyKey *key)
727{
728 if(codepoint < 0x20) {
729 // C0 range
730 key->code.codepoint = 0;
731 key->modifiers = 0;
732
733 if(!(tk->flags & TERMKEY_FLAG_NOINTERPRET) && tk->c0[codepoint].sym != TERMKEY_SYM_UNKNOWN) {
734 key->code.sym = tk->c0[codepoint].sym;
735 key->modifiers |= tk->c0[codepoint].modifier_set;
736 }
737
738 if(!key->code.sym) {
739 key->type = TERMKEY_TYPE_UNICODE;
740 /* Generically modified Unicode ought not report the SHIFT state, or else
741 * we get into complicationg trying to report Shift-; vs : and so on...
742 * In order to be able to represent Ctrl-Shift-A as CTRL modified
743 * unicode A, we need to call Ctrl-A simply 'a', lowercase
744 */
745 if(codepoint+0x40 >= 'A' && codepoint+0x40 <= 'Z')
746 // it's a letter - use lowecase instead
747 key->code.codepoint = codepoint + 0x60;
748 else
749 key->code.codepoint = codepoint + 0x40;
750 key->modifiers = TERMKEY_KEYMOD_CTRL;
751 }
752 else {
753 key->type = TERMKEY_TYPE_KEYSYM;
754 }
755 }
756 else if(codepoint == 0x7f && !(tk->flags & TERMKEY_FLAG_NOINTERPRET)) {
757 // ASCII DEL
758 key->type = TERMKEY_TYPE_KEYSYM;
759 key->code.sym = TERMKEY_SYM_DEL;
760 key->modifiers = 0;
761 }
762 else if(codepoint >= 0x20 && codepoint < 0x80) {
763 // ASCII lowbyte range
764 key->type = TERMKEY_TYPE_UNICODE;
765 key->code.codepoint = codepoint;
766 key->modifiers = 0;
767 }
768 else if(codepoint >= 0x80 && codepoint < 0xa0) {
769 // UTF-8 never starts with a C1 byte. So we can be sure of these
770 key->type = TERMKEY_TYPE_UNICODE;
771 key->code.codepoint = codepoint - 0x40;
772 key->modifiers = TERMKEY_KEYMOD_CTRL|TERMKEY_KEYMOD_ALT;
773 }
774 else {
775 // UTF-8 codepoint
776 key->type = TERMKEY_TYPE_UNICODE;
777 key->code.codepoint = codepoint;
778 key->modifiers = 0;
779 }
780
781 termkey_canonicalise(tk, key);
782
783 if(key->type == TERMKEY_TYPE_UNICODE)
784 fill_utf8(key);
785}
786
787void termkey_canonicalise(TermKey *tk, TermKeyKey *key)
788{
789 int flags = tk->canonflags;
790
791 if(flags & TERMKEY_CANON_SPACESYMBOL) {
792 if(key->type == TERMKEY_TYPE_UNICODE && key->code.number == 0x20) {
793 key->type = TERMKEY_TYPE_KEYSYM;
794 key->code.sym = TERMKEY_SYM_SPACE;
795 }
796 }
797 else {
798 if(key->type == TERMKEY_TYPE_KEYSYM && key->code.sym == TERMKEY_SYM_SPACE) {
799 key->type = TERMKEY_TYPE_UNICODE;
800 key->code.number = 0x20;
801 fill_utf8(key);
802 }
803 }
804
805 if(flags & TERMKEY_CANON_DELBS) {
806 if(key->type == TERMKEY_TYPE_KEYSYM && key->code.sym == TERMKEY_SYM_DEL) {
807 key->code.sym = TERMKEY_SYM_BACKSPACE;
808 }
809 }
810}

812static TermKeyResult peekkey(TermKey *tk, TermKeyKey *key, int force, size_t *nbytep)
813{
814 int again = 0;
815
816 if(!tk->is_started) {
817 errno(*__errno_location ()) = EINVAL22;
818 return TERMKEY_RES_ERROR;
819 }
820
821#ifdef DEBUG
822 fprintf(stderrstderr, "getkey(force=%d): buffer ", force);
823 print_buffer(tk);
824 fprintf(stderrstderr, "\n");
825#endif
826
827 if(tk->hightide) {
828 tk->buffstart += tk->hightide;
829 tk->buffcount -= tk->hightide;
830 tk->hightide = 0;
831 }
832
833 TermKeyResult ret;
834 struct TermKeyDriverNode *p;
835 for(p = tk->drivers; p; p = p->next) {
836 ret = (p->driver->peekkey)(tk, p->info, key, force, nbytep);
837
838#ifdef DEBUG
839 fprintf(stderrstderr, "Driver %s yields %s\n", p->driver->name, res2str(ret));
840#endif
841
842 switch(ret) {
843 case TERMKEY_RES_KEY:
844#ifdef DEBUG
845 print_key(tk, key); fprintf(stderrstderr, "\n");
846#endif
847 // Slide the data down to stop it running away
848 {
849 size_t halfsize = tk->buffsize / 2;
850
851 if(tk->buffstart > halfsize) {
852 memcpy(tk->buffer, tk->buffer + halfsize, halfsize);
853 tk->buffstart -= halfsize;
854 }
855 }
856
857 /* fallthrough */
858 case TERMKEY_RES_EOF:
859 case TERMKEY_RES_ERROR:
860 return ret;
861
862 case TERMKEY_RES_AGAIN:
863 if(!force)
864 again = 1;
865
866 /* fallthrough */
867 case TERMKEY_RES_NONE:
868 break;
869 }
870 }
871
872 if(again)
873 return TERMKEY_RES_AGAIN;
874
875 ret = peekkey_simple(tk, key, force, nbytep);
876
877#ifdef DEBUG
878 fprintf(stderrstderr, "getkey_simple(force=%d) yields %s\n", force, res2str(ret));
879 if(ret == TERMKEY_RES_KEY) {
880 print_key(tk, key); fprintf(stderrstderr, "\n");
881 }
882#endif
883
884 return ret;
885}

887static TermKeyResult peekkey_simple(TermKey *tk, TermKeyKey *key, int force, size_t *nbytep)
888{
889 if(tk->buffcount == 0)
890 return tk->is_closed ? TERMKEY_RES_EOF : TERMKEY_RES_NONE;
891
892 unsigned char b0 = CHARAT(0)(tk->buffer[tk->buffstart + (0)]);
893
894 if(b0 == 0x1b) {
895 // Escape-prefixed value? Might therefore be Alt+key
896 if(tk->buffcount == 1) {
897 // This might be an <Esc> press, or it may want to be part of a longer
898 // sequence
899 if(!force)
900 return TERMKEY_RES_AGAIN;
901
902 (*tk->method.emit_codepoint)(tk, b0, key);
903 *nbytep = 1;
904 return TERMKEY_RES_KEY;
905 }
906
907 // Try another key there
908 tk->buffstart++;
909 tk->buffcount--;
910
911 // Run the full driver
912 TermKeyResult metakey_result = peekkey(tk, key, force, nbytep);
913
914 tk->buffstart--;
915 tk->buffcount++;
916
917 switch(metakey_result) {
918 case TERMKEY_RES_KEY:
919 key->modifiers |= TERMKEY_KEYMOD_ALT;
920 (*nbytep)++;
921 break;
922
923 case TERMKEY_RES_NONE:
924 case TERMKEY_RES_EOF:
925 case TERMKEY_RES_AGAIN:
926 case TERMKEY_RES_ERROR:
927 break;
928 }
929
930 return metakey_result;
931 }
932 else if(b0 < 0xa0) {
933 // Single byte C0, G0 or C1 - C1 is never UTF-8 initial byte
934 (*tk->method.emit_codepoint)(tk, b0, key);
935 *nbytep = 1;
936 return TERMKEY_RES_KEY;
937 }
938 else if(tk->flags & TERMKEY_FLAG_UTF8) {
939 // Some UTF-8
940 long codepoint;
941 TermKeyResult res = parse_utf8(tk->buffer + tk->buffstart, tk->buffcount, &codepoint, nbytep);
942
943 if(res == TERMKEY_RES_AGAIN && force) {
944 /* There weren't enough bytes for a complete UTF-8 sequence but caller
945 * demands an answer. About the best thing we can do here is eat as many
946 * bytes as we have, and emit a UTF8_INVALID. If the remaining bytes
947 * arrive later, they'll be invalid too.
948 */
949 codepoint = UTF8_INVALID0xFFFD;
950 *nbytep = tk->buffcount;
951 res = TERMKEY_RES_KEY;
952 }
953
954 key->type = TERMKEY_TYPE_UNICODE;
955 key->modifiers = 0;
956 (*tk->method.emit_codepoint)(tk, codepoint, key);
957 return res;
958 }
959 else {
960 // Non UTF-8 case - just report the raw byte
961 key->type = TERMKEY_TYPE_UNICODE;
962 key->code.codepoint = b0;
963 key->modifiers = 0;
964
965 key->utf8[0] = key->code.codepoint;
966 key->utf8[1] = 0;
967
968 *nbytep = 1;
969
970 return TERMKEY_RES_KEY;
971 }
972}

974static TermKeyResult peekkey_mouse(TermKey *tk, TermKeyKey *key, size_t *nbytep)
975{
976 if(tk->buffcount < 3)
977 return TERMKEY_RES_AGAIN;
978
979 key->type = TERMKEY_TYPE_MOUSE;
980 key->code.mouse[0] = CHARAT(0)(tk->buffer[tk->buffstart + (0)]) - 0x20;
981 key->code.mouse[1] = CHARAT(1)(tk->buffer[tk->buffstart + (1)]) - 0x20;
982 key->code.mouse[2] = CHARAT(2)(tk->buffer[tk->buffstart + (2)]) - 0x20;
983 key->code.mouse[3] = 0;
984
985 key->modifiers = (key->code.mouse[0] & 0x1c) >> 2;
986 key->code.mouse[0] &= ~0x1c;
987
988 *nbytep = 3;
989 return TERMKEY_RES_KEY;
990}

992TermKeyResult termkey_getkey(TermKey *tk, TermKeyKey *key)
993{
994 size_t nbytes = 0;
995 TermKeyResult ret = peekkey(tk, key, 0, &nbytes);
996
997 if(ret == TERMKEY_RES_KEY)
998 eat_bytes(tk, nbytes);
999
1000 if(ret == TERMKEY_RES_AGAIN)
1001 /* Call peekkey() again in force mode to obtain whatever it can */
1002 (void)peekkey(tk, key, 1, &nbytes);
1003 /* Don't eat it yet though */
1004
1005 return ret;
1006}

1008TermKeyResult termkey_getkey_force(TermKey *tk, TermKeyKey *key)
1009{
1010 size_t nbytes = 0;
1011 TermKeyResult ret = peekkey(tk, key, 1, &nbytes);
1012
1013 if(ret == TERMKEY_RES_KEY)
1014 eat_bytes(tk, nbytes);
1015
1016 return ret;
1017}

1019TermKeyResult termkey_waitkey(TermKey *tk, TermKeyKey *key)
1020{
1021 if(tk->fd == -1) {
1022 errno(*__errno_location ()) = EBADF9;
1023 return TERMKEY_RES_ERROR;
1024 }
1025
1026 while(1) {
1027 TermKeyResult ret = termkey_getkey(tk, key);
1028
1029 switch(ret) {
1030 case TERMKEY_RES_KEY:
1031 case TERMKEY_RES_EOF:
1032 case TERMKEY_RES_ERROR:
1033 return ret;
1034
1035 case TERMKEY_RES_NONE:
1036 ret = termkey_advisereadable(tk);
1037 if(ret == TERMKEY_RES_ERROR)
1038 return ret;
1039 break;
1040
1041 case TERMKEY_RES_AGAIN:
1042 {
1043 if(tk->is_closed)
1044 // We're closed now. Never going to get more bytes so just go with
1045 // what we have
1046 return termkey_getkey_force(tk, key);
1047
1048 struct pollfd fd;
1049
1050retry:
1051 fd.fd = tk->fd;
1052 fd.events = POLLIN0x001;
1053
1054 int pollret = poll(&fd, 1, tk->waittime);
1055 if(pollret == -1) {
1056 if(errno(*__errno_location ()) == EINTR4 && !(tk->flags & TERMKEY_FLAG_EINTR))
1057 goto retry;
1058
1059 return TERMKEY_RES_ERROR;
1060 }
1061
1062 if(fd.revents & (POLLIN0x001|POLLHUP0x010|POLLERR0x008))
1063 ret = termkey_advisereadable(tk);
1064 else
1065 ret = TERMKEY_RES_NONE;
1066
1067 if(ret == TERMKEY_RES_ERROR)
1068 return ret;
1069 if(ret == TERMKEY_RES_NONE)
1070 return termkey_getkey_force(tk, key);
1071 }
1072 break;
1073 }
1074 }
1075
1076 /* UNREACHABLE */
1077}
1078
1079TermKeyResult termkey_advisereadable(TermKey *tk)
1080{
1081 ssize_t len;
1082
1083 if(tk->fd == -1) {
1084 errno(*__errno_location ()) = EBADF9;
1085 return TERMKEY_RES_ERROR;
1086 }
1087
1088 if(tk->buffstart) {
1089 memmove(tk->buffer, tk->buffer + tk->buffstart, tk->buffcount);
1090 tk->buffstart = 0;
1091 }
1092
1093 /* Not expecting it ever to be greater but doesn't hurt to handle that */
1094 if(tk->buffcount >= tk->buffsize) {
1095 errno(*__errno_location ()) = ENOMEM12;
1096 return TERMKEY_RES_ERROR;
1097 }
1098
1099retry:
1100 len = read(tk->fd, tk->buffer + tk->buffcount, tk->buffsize - tk->buffcount);
1101
1102 if(len == -1) {
1103 if(errno(*__errno_location ()) == EAGAIN11)
1104 return TERMKEY_RES_NONE;
1105 else if(errno(*__errno_location ()) == EINTR4 && !(tk->flags & TERMKEY_FLAG_EINTR))
1106 goto retry;
1107 else
1108 return TERMKEY_RES_ERROR;
1109 }
1110 else if(len < 1) {
1111 tk->is_closed = 1;
1112 return TERMKEY_RES_NONE;
1113 }
1114 else {
1115 tk->buffcount += len;
1116 return TERMKEY_RES_AGAIN;
1117 }
1118}

1120size_t termkey_push_bytes(TermKey *tk, const char *bytes, size_t len)
1121{
1122 if(tk->buffstart) {
1123 memmove(tk->buffer, tk->buffer + tk->buffstart, tk->buffcount);
1124 tk->buffstart = 0;
1125 }
1126
1127 /* Not expecting it ever to be greater but doesn't hurt to handle that */
1128 if(tk->buffcount >= tk->buffsize) {
1129 errno(*__errno_location ()) = ENOMEM12;
1130 return (size_t)-1;
1131 }
1132
1133 if(len > tk->buffsize - tk->buffcount)
1134 len = tk->buffsize - tk->buffcount;
1135
1136 // memcpy(), not strncpy() in case of null bytes in input
1137 memcpy(tk->buffer + tk->buffcount, bytes, len);
1138 tk->buffcount += len;
1139
1140 return len;
1141}

1144{

1146 sym = tk->nkeynames;
1147
1148 if(sym >= tk->nkeynames) {

1178static const char *termkey_lookup_keyname_format(TermKey *tk, const char *str, TermKeySym *sym, TermKeyFormat format)
1179{
1180 /* We store an array, so we can't do better than a linear search. Doesn't
1181 * matter because user won't be calling this too often */
1182
1183 for(*sym = 0; *sym < tk->nkeynames; (*sym)++) {
1184 const char *thiskey = tk->keynames[*sym];
1185 if(!thiskey)
1186 continue;
1187 size_t len = strlen(thiskey);
1188 if(format & TERMKEY_FORMAT_LOWERSPACE) {
1189 const char *thisstr = str;
1190 if(strpncmp_camel(&thisstr, &thiskey, len) == 0)
1191 return thisstr;
1192 }
1193 else {
1194 if(strncmp(str, thiskey, len) == 0)
1195 return (char *)str + len;
1196 }
1197 }
1198
1199 return NULL((void*)0);
1200}

1249}
1250modnames[] = {

1252 { "Shift", "Alt", "Ctrl" }, // LONGMOD
1253 { "S", "M", "C" }, // ALTISMETA
1254 { "Shift", "Meta", "Ctrl" }, // ALTISMETA+LONGMOD

1261size_t termkey_strfkey(TermKey *tk, char *buffer, size_t len, TermKeyKey *key, TermKeyFormat format)
1262{
1263 size_t pos = 0;
1264 size_t l = 0;
1265
1266 struct modnames *mods = &modnames[!!(format & TERMKEY_FORMAT_LONGMOD) +
1267 !!(format & TERMKEY_FORMAT_ALTISMETA) * 2 +
1268 !!(format & TERMKEY_FORMAT_LOWERMOD) * 4];
1269
1270 int wrapbracket = (format & TERMKEY_FORMAT_WRAPBRACKET) &&
1271 (key->type != TERMKEY_TYPE_UNICODE || key->modifiers != 0);
1272
1273 char sep = (format & TERMKEY_FORMAT_SPACEMOD) ? ' ' : '-';
1274
1275 if(format & TERMKEY_FORMAT_CARETCTRL &&
1276 key->type == TERMKEY_TYPE_UNICODE &&
1277 key->modifiers == TERMKEY_KEYMOD_CTRL) {
1278 long codepoint = key->code.codepoint;
1279
1280 // Handle some of the special casesfirst
1281 if(codepoint >= 'a' && codepoint <= 'z') {
1282 l = snprintf(buffer + pos, len - pos, wrapbracket ? "<^%c>" : "^%c", (char)codepoint - 0x20);
1283 if(l <= 0) return pos;
1284 pos += l;
1285 return pos;
1286 }
1287 else if((codepoint >= '@' && codepoint < 'A') ||
1288 (codepoint > 'Z' && codepoint <= '_')) {
1289 l = snprintf(buffer + pos, len - pos, wrapbracket ? "<^%c>" : "^%c", (char)codepoint);
1290 if(l <= 0) return pos;
1291 pos += l;
1292 return pos;
1293 }
1294 }
1295
1296 if(wrapbracket) {
1297 l = snprintf(buffer + pos, len - pos, "<");
1298 if(l <= 0) return pos;
1299 pos += l;
1300 }
1301
1302 if(key->modifiers & TERMKEY_KEYMOD_ALT) {
1303 l = snprintf(buffer + pos, len - pos, "%s%c", mods->alt, sep);
1304 if(l <= 0) return pos;
1305 pos += l;
1306 }
1307
1308 if(key->modifiers & TERMKEY_KEYMOD_CTRL) {
1309 l = snprintf(buffer + pos, len - pos, "%s%c", mods->ctrl, sep);
1310 if(l <= 0) return pos;
1311 pos += l;
1312 }
1313
1314 if(key->modifiers & TERMKEY_KEYMOD_SHIFT) {
1315 l = snprintf(buffer + pos, len - pos, "%s%c", mods->shift, sep);
1316 if(l <= 0) return pos;
1317 pos += l;
1318 }
1319
1320 switch(key->type) {
1321 case TERMKEY_TYPE_UNICODE:
1322 if(!key->utf8[0]) // In case of user-supplied key structures
1323 fill_utf8(key);
1324 l = snprintf(buffer + pos, len - pos, "%s", key->utf8);
1325 break;
1326 case TERMKEY_TYPE_KEYSYM:
1327 {
1328 const char *name = termkey_get_keyname(tk, key->code.sym);
1329 if(format & TERMKEY_FORMAT_LOWERSPACE)
1330 l = snprint_cameltospaces(buffer + pos, len - pos, name);
1331 else
1332 l = snprintf(buffer + pos, len - pos, "%s", name);
1333 }
1334 break;
1335 case TERMKEY_TYPE_FUNCTION:
1336 l = snprintf(buffer + pos, len - pos, "%c%d",
1337 (format & TERMKEY_FORMAT_LOWERSPACE ? 'f' : 'F'), key->code.number);
1338 break;
1339 case TERMKEY_TYPE_MOUSE:
1340 {
1341 TermKeyMouseEvent ev;
1342 int button;
1343 int line, col;
1344 termkey_interpret_mouse(tk, key, &ev, &button, &line, &col);
1345
1346 static const char *evnames[] = { "Unknown", "Press", "Drag", "Release" };
1347
1348 l = snprintf(buffer + pos, len - pos, "Mouse%s(%d)",
1349 evnames[ev], button);
1350
1351 if(format & TERMKEY_FORMAT_MOUSE_POS) {
1352 if(l <= 0) return pos;
1353 pos += l;
1354
1355 l = snprintf(buffer + pos, len - pos, " @ (%u,%u)", col, line);
1356 }
1357 }
1358 break;
1359 case TERMKEY_TYPE_POSITION:
1360 l = snprintf(buffer + pos, len - pos, "Position");
1361 break;
1362 case TERMKEY_TYPE_MODEREPORT:
1363 {
1364 int initial, mode, value;
1365 termkey_interpret_modereport(tk, key, &initial, &mode, &value);
1366 if(initial)

1388const char *termkey_strpkey(TermKey *tk, const char *str, TermKeyKey *key, TermKeyFormat format)
1389{
1390 struct modnames *mods = &modnames[!!(format & TERMKEY_FORMAT_LONGMOD) +
1391 !!(format & TERMKEY_FORMAT_ALTISMETA) * 2 +
1392 !!(format & TERMKEY_FORMAT_LOWERMOD) * 4];
1393
1394 key->modifiers = 0;
1395
1396 if((format & TERMKEY_FORMAT_CARETCTRL) && str[0] == '^' && str[1]) {
1397 str = termkey_strpkey(tk, str+1, key, format & ~TERMKEY_FORMAT_CARETCTRL);
1398
1399 if(!str ||
1400 key->type != TERMKEY_TYPE_UNICODE ||
1401 key->code.codepoint < '@' || key->code.codepoint > '_' ||
1402 key->modifiers != 0)
1403 return NULL((void*)0);
1404
1405 if(key->code.codepoint >= 'A' && key->code.codepoint <= 'Z')
1406 key->code.codepoint += 0x20;
1407 key->modifiers = TERMKEY_KEYMOD_CTRL;
1408 fill_utf8(key);
1409 return (char *)str;
1410 }
1411
1412 const char *sep_at;
1413
1414 while((sep_at = strchr(str, (format & TERMKEY_FORMAT_SPACEMOD) ? ' ' : '-'))) {
1415 size_t n = sep_at - str;
1416
1417 if(n == strlen(mods->alt) && strncmp(mods->alt, str, n) == 0)
1418 key->modifiers |= TERMKEY_KEYMOD_ALT;
1419 else if(n == strlen(mods->ctrl) && strncmp(mods->ctrl, str, n) == 0)
1420 key->modifiers |= TERMKEY_KEYMOD_CTRL;
1421 else if(n == strlen(mods->shift) && strncmp(mods->shift, str, n) == 0)
1422 key->modifiers |= TERMKEY_KEYMOD_SHIFT;
1423
1424 else
1425 break;
1426
1427 str = sep_at + 1;
1428 }
1429
1430 size_t nbytes;
1431 ssize_t snbytes;
1432 const char *endstr;
1433
1434 if((endstr = termkey_lookup_keyname_format(tk, str, &key->code.sym, format))) {
1435 key->type = TERMKEY_TYPE_KEYSYM;
1436 str = endstr;
1437 }
1438 else if(sscanf(str, "F%d%zn", &key->code.number, &snbytes) == 1) {
1439 key->type = TERMKEY_TYPE_FUNCTION;
1440 str += snbytes;
1441 }
1442 // Unicode must be last
1443 else if(parse_utf8((unsigned const char *)str, strlen(str), &key->code.codepoint, &nbytes) == TERMKEY_RES_KEY) {
1444 key->type = TERMKEY_TYPE_UNICODE;
1445 fill_utf8(key);
1446 str += nbytes;
1447 }
1448 // TODO: Consider mouse events?
1449 else
1450 return NULL((void*)0);
1451
1452 termkey_canonicalise(tk, key);
1453
1454 return (char *)str;
1455}

1457int termkey_keycmp(TermKey *tk, const TermKeyKey *key1p, const TermKeyKey *key2p)
1458{
1459 /* Copy the key structs since we'll be modifying them */
1460 TermKeyKey key1 = *key1p, key2 = *key2p;
1461
1462 termkey_canonicalise(tk, &key1);
1463 termkey_canonicalise(tk, &key2);
1464
1465 if(key1.type != key2.type)
1466 return key1.type - key2.type;
1467
1468 switch(key1.type) {
1469 case TERMKEY_TYPE_UNICODE:
1470 if(key1.code.codepoint != key2.code.codepoint)
1471 return key1.code.codepoint - key2.code.codepoint;
1472 break;
1473 case TERMKEY_TYPE_KEYSYM:
1474 if(key1.code.sym != key2.code.sym)
1475 return key1.code.sym - key2.code.sym;
1476 break;
1477 case TERMKEY_TYPE_FUNCTION:
1478 case TERMKEY_TYPE_UNKNOWN_CSI:
1479 if(key1.code.number != key2.code.number)
1480 return key1.code.number - key2.code.number;
1481 break;
1482 case TERMKEY_TYPE_MOUSE:
1483 {
1484 int cmp = strncmp(key1.code.mouse, key2.code.mouse, 4);
1485 if(cmp != 0)
1486 return cmp;
1487 }
1488 break;
1489 case TERMKEY_TYPE_POSITION:
1490 {
1491 int line1, col1, line2, col2;
1492 termkey_interpret_position(tk, &key1, &line1, &col1);
1493 termkey_interpret_position(tk, &key2, &line2, &col2);
1494 if(line1 != line2)
1495 return line1 - line2;
1496 return col1 - col2;
1497 }
1498 break;
1499 case TERMKEY_TYPE_MODEREPORT:
1500 {
1501 int initial1, initial2, mode1, mode2, value1, value2;
1502 termkey_interpret_modereport(tk, &key1, &initial1, &mode1, &value1);
1503 termkey_interpret_modereport(tk, &key2, &initial2, &mode2, &value2);
1504 if(initial1 != initial2)
1505 return initial1 - initial2;
1506 if(mode1 != mode2)
1507 return mode1 - mode2;
1508 return value1 - value2;
1509 }
1510 }
1511
1512 return key1.modifiers - key2.modifiers;
1513}
