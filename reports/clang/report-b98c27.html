651 prt_get_attr(id, &settings.number, settings.modec);
652 }
653
654 /*
655 * Estimate the total lines to be printed
656 */
657 for (lnum = eap->line1; lnum <= eap->line2; lnum++)
658 bytes_to_print += (long_u)STRLEN(skipwhite(ml_get(lnum)))strlen((char *)(skipwhite(ml_get(lnum))));
659 if (bytes_to_print == 0) {
660 MSG(_("No text to be printed"))msg((char_u *)(gettext((char *)("No text to be printed"))));
661 goto print_fail_no_begin;
662 }
663
664 /* Set colors and font to normal. */
665 curr_bg = (long_u)0xffffffffL;
666 curr_fg = (long_u)0xffffffffL;
667 curr_italic = MAYBE2;
668 curr_bold = MAYBE2;
669 curr_underline = MAYBE2;
670
671 prt_set_fg(PRCOLOR_BLACK(long_u)0);
672 prt_set_bg(PRCOLOR_WHITE(long_u)0xFFFFFFL);
673 prt_set_font(FALSE0, FALSE0, FALSE0);
674 current_syn_id = -1;
675
676 jobsplit = (printer_opts[OPT_PRINT_JOBSPLIT12].present
677 && TOLOWER_ASC(printer_opts[OPT_PRINT_JOBSPLIT].string[0])(((printer_opts[12].string[0]) < 'A' || (printer_opts[12].
string[0]) > 'Z') ? (printer_opts[12].string[0]) : (printer_opts
[12].string[0]) + ('a' - 'A')) == 'y');
678
679 if (!mch_print_begin(&settings))
680 goto print_fail_no_begin;
681
682 /*
683 * Loop over collated copies: 1 2 3, 1 2 3, ...
684 */
685 page_count = 0;
686 for (collated_copies = 0;
687 collated_copies < settings.n_collated_copies;
688 collated_copies++) {
689 prt_pos_T prtpos; /* current print position */
690 prt_pos_T page_prtpos; /* print position at page start */
691 int side;
692
693 memset(&page_prtpos, 0, sizeof(prt_pos_T));
694 page_prtpos.file_line = eap->line1;
695 prtpos = page_prtpos;
696
697 if (jobsplit && collated_copies > 0) {
698 /* Splitting jobs: Stop a previous job and start a new one. */
699 mch_print_end(&settings);
700 if (!mch_print_begin(&settings))
701 goto print_fail_no_begin;
702 }
703
704 /*
705 * Loop over all pages in the print job: 1 2 3 ...
706 */
707 for (page_count = 0; prtpos.file_line <= eap->line2; ++page_count) {
708 /*
709 * Loop over uncollated copies: 1 1 1, 2 2 2, 3 3 3, ...
710 * For duplex: 12 12 12 34 34 34, ...
711 */
712 for (uncollated_copies = 0;
713 uncollated_copies < settings.n_uncollated_copies;
714 uncollated_copies++) {
715 /* Set the print position to the start of this page. */
716 prtpos = page_prtpos;
717
718 /*
719 * Do front and rear side of a page.
720 */
721 for (side = 0; side <= settings.duplex; ++side) {
722 /*
723 * Print one page.
724 */
725
726 /* Check for interrupt character every page. */
727 ui_breakcheck();
728 if (got_int || settings.user_abort)
729 goto print_fail;
730
731 sprintf((char *)IObuff, _("Printing page %d (%d%%)")gettext((char *)("Printing page %d (%d%%)")),
732 page_count + 1 + side,
733 prtpos.bytes_printed > 1000000
734 ? (int)(prtpos.bytes_printed /
735 (bytes_to_print / 100))
736 : (int)((prtpos.bytes_printed * 100)
737 / bytes_to_print));
738 if (!mch_print_begin_page(IObuff))
739 goto print_fail;
740
741 if (settings.n_collated_copies > 1)
742 sprintf((char *)IObuff + STRLEN(IObuff)strlen((char *)(IObuff)),
743 _(" Copy %d of %d")gettext((char *)(" Copy %d of %d")),
744 collated_copies + 1,
745 settings.n_collated_copies);
746 prt_message(IObuff);
747
748 /*
749 * Output header if required
750 */
751 if (prt_header_height() > 0)
752 prt_header(&settings, page_count + 1 + side,
753 prtpos.file_line);
754
755 for (page_line = 0; page_line < settings.lines_per_page;
756 ++page_line) {
757 prtpos.column = hardcopy_line(&settings,
758 page_line, &prtpos);
759 if (prtpos.column == 0) {
760 /* finished a file line */
761 prtpos.bytes_printed +=
762 STRLEN(skipwhite(ml_get(prtpos.file_line)))strlen((char *)(skipwhite(ml_get(prtpos.file_line))));
763 if (++prtpos.file_line > eap->line2)
764 break; /* reached the end */
765 } else if (prtpos.ff) {
766 /* Line had a formfeed in it - start new page but
767 * stay on the current line */
768 break;
769 }
770 }
771
772 if (!mch_print_end_page())
773 goto print_fail;
774 if (prtpos.file_line > eap->line2)
775 break; /* reached the end */
776 }
777
778 /*
779 * Extra blank page for duplexing with odd number of pages and
780 * more copies to come.
781 */
782 if (prtpos.file_line > eap->line2 && settings.duplex
783 && side == 0
784 && uncollated_copies + 1 < settings.n_uncollated_copies) {
785 if (!mch_print_blank_page())
786 goto print_fail;
787 }
788 }
789 if (settings.duplex && prtpos.file_line <= eap->line2)
790 ++page_count;
791
792 /* Remember the position where the next page starts. */
793 page_prtpos = prtpos;
794 }
795
796 vim_snprintf((char *)IObuff, IOSIZE(1024+1), _("Printed: %s")gettext((char *)("Printed: %s")),
797 settings.jobname);
798 prt_message(IObuff);
799 }
800
801print_fail:
802 if (got_int || settings.user_abort) {
803 sprintf((char *)IObuff, "%s", _("Printing aborted")gettext((char *)("Printing aborted")));
804 prt_message(IObuff);
805 }
806 mch_print_end(&settings);
807
808print_fail_no_begin:
809 mch_print_cleanup();
810}

812/*
813 * Print one page line.
814 * Return the next column to print, or zero if the line is finished.
815 */
816static colnr_T hardcopy_line(prt_settings_T *psettings, int page_line, prt_pos_T *ppos)
817{
818 colnr_T col;
819 char_u *line;
820 int need_break = FALSE0;
821 int outputlen;
822 int tab_spaces;
823 long_u print_pos;
824 prt_text_attr_T attr;
825 int id;
826
827 if (ppos->column == 0 || ppos->ff) {
828 print_pos = 0;
829 tab_spaces = 0;
830 if (!ppos->ff && prt_use_number())
831 prt_line_number(psettings, page_line, ppos->file_line);
832 ppos->ff = FALSE0;
833 } else {
834 /* left over from wrap halfway a tab */
835 print_pos = ppos->print_pos;
836 tab_spaces = ppos->lead_spaces;
837 }
838
839 mch_print_start_line(0, page_line);
840 line = ml_get(ppos->file_line);
841
842 /*
843 * Loop over the columns until the end of the file line or right margin.
844 */
845 for (col = ppos->column; line[col] != NUL'\000' && !need_break; col += outputlen) {
846 outputlen = 1;
847 if (has_mbyte && (outputlen = (*mb_ptr2len)(line + col)) < 1)
848 outputlen = 1;
849 /*
850 * syntax highlighting stuff.
851 */
852 if (psettings->do_syntax) {
853 id = syn_get_id(curwin, ppos->file_line, col, 1, NULL((void*)0), FALSE0);
854 if (id > 0)
855 id = syn_get_final_id(id);
856 else
857 id = 0;
858 /* Get the line again, a multi-line regexp may invalidate it. */
859 line = ml_get(ppos->file_line);
860
861 if (id != current_syn_id) {
862 current_syn_id = id;
863 prt_get_attr(id, &attr, psettings->modec);
864 prt_set_font(attr.bold, attr.italic, attr.underline);
865 prt_set_fg(attr.fg_color);
866 prt_set_bg(attr.bg_color);
867 }
868 }
869
870 /*
871 * Appropriately expand any tabs to spaces.
872 */
873 if (line[col] == TAB'\011' || tab_spaces != 0) {
874 if (tab_spaces == 0)
875 tab_spaces = (int)(curbuf->b_p_ts - (print_pos % curbuf->b_p_ts));
876
877 while (tab_spaces > 0) {
878 need_break = mch_print_text_out((char_u *)" ", 1);
879 print_pos++;
880 tab_spaces--;
881 if (need_break)
882 break;
883 }
884 /* Keep the TAB if we didn't finish it. */
885 if (need_break && tab_spaces > 0)
886 break;
887 } else if (line[col] == FF'\014'
888 && printer_opts[OPT_PRINT_FORMFEED13].present
889 && TOLOWER_ASC(printer_opts[OPT_PRINT_FORMFEED].string[0])(((printer_opts[13].string[0]) < 'A' || (printer_opts[13].
string[0]) > 'Z') ? (printer_opts[13].string[0]) : (printer_opts
[13].string[0]) + ('a' - 'A'))
890 == 'y') {
891 ppos->ff = TRUE1;
892 need_break = 1;
893 } else {
894 need_break = mch_print_text_out(line + col, outputlen);
895 if (has_mbyte)
896 print_pos += (*mb_ptr2cells)(line + col);
897 else
898 print_pos++;
899 }
900 }
901
902 ppos->lead_spaces = tab_spaces;
903 ppos->print_pos = (int)print_pos;
904
905 /*
906 * Start next line of file if we clip lines, or have reached end of the
907 * line, unless we are doing a formfeed.
908 */
909 if (!ppos->ff
910 && (line[col] == NUL'\000'
911 || (printer_opts[OPT_PRINT_WRAP7].present
912 && TOLOWER_ASC(printer_opts[OPT_PRINT_WRAP].string[0])(((printer_opts[7].string[0]) < 'A' || (printer_opts[7].string
[0]) > 'Z') ? (printer_opts[7].string[0]) : (printer_opts[
7].string[0]) + ('a' - 'A'))
913 == 'n')))
914 return 0;
915 return col;
916}

1396/*
1397 * Convert a real value into an integer and fractional part as integers, with
1398 * the fractional part being in the range [0,10^precision). The fractional part
1399 * is also rounded based on the precision + 1'th fractional digit.
1400 */
1401static void prt_real_bits(double real, int precision, int *pinteger, int *pfraction)
1402{
1403 int i;
1404 int integer;
1405 float fraction;
1406
1407 integer = (int)real;
1408 fraction = (float)(real - integer);
1409 if (real < (double)integer)
1410 fraction = -fraction;
1411 for (i = 0; i < precision; i++)
1412 fraction *= 10.0;
1413
1414 *pinteger = integer;
1415 *pfraction = (int)(fraction + 0.5);
1416}

1420 * We use prt_real_bits() as %f in sprintf uses the locale setting to decide
1421 * what decimal point character to use, but PS always requires a '.'.
1422 */
1423static void prt_write_real(double val, int prec)
1424{

1438 }

1440 sprintf((char *)prt_line_buffer, ".%0*d", prec, fraction);
1441 prt_write_file(prt_line_buffer);
