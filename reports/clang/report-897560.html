924/*

949 ++i;

951 /*
952 * Repeat to match several nested comment strings.
953 */

955 /*
956 * scan through the 'comments' option for a match
957 */

960 /* Get one option part into part_buf[]. Advance "list" to next
961 * one. Put "string" at start of string. */

968 continue;

970

972 * is not a middle or end. */
973 if (middle_match_len != 0

975 && vim_strchr(part_buf, COM_END'e') == NULL((void*)0))
976 break;
977
978 /* When we already found a nested comment, only accept further
979 * nested comments. */
980 if (got_com && vim_strchr(part_buf, COM_NEST'n') == NULL((void*)0))
981 continue;
982
983 /* When 'O' flag present and using "O" command skip this one. */
984 if (backward && vim_strchr(part_buf, COM_NOBACK'O') != NULL((void*)0))
985 continue;

988 * When string starts with white space, must have some white space
989 * (but the amount does not need to match, there might be a mix of
990 * TABs and spaces). */

993 continue; /* missing white space */
994 while (ascii_iswhite(string[0]))
995 ++string;
996 }

998 ;

1000 continue; /* string doesn't match */
1001

1003 * end-of-line after the string in the line. */
1004 if (vim_strchr(part_buf, COM_BLANK'b') != NULL((void*)0)

1006 continue;
1007

1009 * comment. The middle comment can be a substring of the end
1010 * comment in which case it's better to return the length of the
1011 * end comment and its flags. Thus we keep searching with middle
1012 * and end matches and use an end match if it matches better. */
1013 if (vim_strchr(part_buf, COM_MIDDLE'm') != NULL((void*)0)) {
1014 if (middle_match_len == 0) {
1015 middle_match_len = j;
1016 saved_flags = prev_list;
1017 }
1018 continue;
1019 }
1020 if (middle_match_len != 0 && j > middle_match_len)
1021 /* Use this match instead of the middle match, since it's a
1022 * longer thus better match. */
1023 middle_match_len = 0;
1024
1025 if (middle_match_len == 0)
1026 i += j;
1027 found_one = TRUE1;
1028 break;
1029 }
1030
1031 if (middle_match_len != 0) {
1032 /* Use the previously found middle match after failing to find a
1033 * match with an end. */
1034 if (!got_com && flags != NULL((void*)0))
1035 *flags = saved_flags;
1036 i += middle_match_len;
1037 found_one = TRUE1;
1038 }
1039
1040 /* No match found, stop scanning. */
1041 if (!found_one)
1042 break;
1043
1044 result = i;
1045
1046 /* Include any trailing white space. */
1047 while (ascii_iswhite(line[i]))
1048 ++i;
1049
1050 if (include_space)
1051 result = i;
1052
1053 /* If this comment doesn't nest, stop here. */
1054 got_com = TRUE1;
1055 if (vim_strchr(part_buf, COM_NEST'n') == NULL((void*)0))
1056 break;
1057 }
1058 return result;
1059}

1481
1482 // Fill with spaces when necessary.
1483 for (size_t i = charlen; i < newlen; i++) {
1484 p[i] = ' ';
1485 }

1488 ml_replace(lnum, newp, FALSE0);

1491 changed_bytes(lnum, (colnr_T)col);

1496 */

1498 && msg_silent == 0
1499 && !ins_compl_active()

1503 else

1509 curwin->w_cursor.col += (int)charlen;

1512 * TODO: should try to update w_row here, to avoid recomputing it later.
1513 */

2718/// Get the stdout of an external command.
2719/// If "ret_len" is NULL replace NUL characters with NL. When "ret_len" is not
2720/// NULL store the length there.
2721///
2722/// @param cmd command to execute
2723/// @param infile optional input file name
2724/// @param flags can be kShellOptSilent or 0
2725/// @param ret_len length of the stdout
2726///
2727/// @return an allocated string, or NULL for error.
2728char_u *get_cmd_output(char_u *cmd, char_u *infile, ShellOpts flags,
2729 size_t *ret_len)
2730{
2731 char_u *buffer = NULL((void*)0);
2732
2733 if (check_restricted() || check_secure())
2734 return NULL((void*)0);
2735
2736 // get a name for the temp file
2737 char_u *tempname = vim_tempname();
2738 if (tempname == NULL((void*)0)) {
2739 EMSG(_(e_notmp))emsg((char_u *)(gettext((char *)(e_notmp))));
2740 return NULL((void*)0);
2741 }
2742
2743 // Add the redirection stuff
2744 char_u *command = make_filter_cmd(cmd, infile, tempname);
2745
2746 /*
2747 * Call the shell to execute the command (errors are ignored).
2748 * Don't check timestamps here.
2749 */
2750 ++no_check_timestamps;
2751 call_shell(command, kShellOptDoOut | kShellOptExpand | flags, NULL((void*)0));
2752 --no_check_timestamps;
2753
2754 xfree(command);
2755
2756 // read the names from the file into memory
2757 FILE *fd = mch_fopen((char *)tempname, READBIN)fopen(((char *)tempname), ("rb"));
2758
2759 if (fd == NULL((void*)0)) {
2760 EMSG2(_(e_notopen), tempname)emsgf((const char *) (gettext((char *)(e_notopen))), (tempname
));
2761 goto done;
2762 }
2763
2764 fseek(fd, 0L, SEEK_END2);
2765 size_t len = (size_t)ftell(fd); // get size of temp file
2766 fseek(fd, 0L, SEEK_SET0);
2767
2768 buffer = xmalloc(len + 1);
2769 size_t i = fread((char *)buffer, 1, len, fd);
2770 fclose(fd);
2771 os_remove((char *)tempname);
2772 if (i != len) {
2773 EMSG2(_(e_notread), tempname)emsgf((const char *) (gettext((char *)(e_notread))), (tempname
));
2774 xfree(buffer);
2775 buffer = NULL((void*)0);
2776 } else if (ret_len == NULL((void*)0)) {
2777 /* Change NUL into SOH, otherwise the string is truncated. */
2778 for (i = 0; i < len; ++i)
2779 if (buffer[i] == NUL'\000')
2780 buffer[i] = 1;
2781
2782 buffer[len] = NUL'\000'; /* make sure the buffer is terminated */
2783 } else {

2785 }
