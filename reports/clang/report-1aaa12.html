539static void invalidate(UI *ui, int top, int bot, int left, int right)
540{
541 TUIData *data = ui->data;
542 Rect *intersects = NULL((void*)0);
543 // Increase dimensions before comparing to ensure adjacent regions are
544 // treated as intersecting
545 --top;
546 ++bot;
547 --left;
548 ++right;
549
550 for (size_t i = 0; i < kv_size(data->invalid_regions)((data->invalid_regions).size); i++) {
551 Rect *r = &kv_A(data->invalid_regions, i)((data->invalid_regions).items[(i)]);
552 if (!(top > r->bot || bot < r->top
553 || left > r->right || right < r->left)) {
554 intersects = r;
555 break;
556 }
557 }
558
559 ++top;
560 --bot;
561 ++left;
562 --right;
563
564 if (intersects) {
565 // If top/bot/left/right intersects with a invalid rect, we replace it
566 // by the union
567 intersects->top = MIN(top, intersects->top)(((top)<(intersects->top))?(top):(intersects->top));
568 intersects->bot = MAX(bot, intersects->bot)(((bot)>(intersects->bot))?(bot):(intersects->bot));
569 intersects->left = MIN(left, intersects->left)(((left)<(intersects->left))?(left):(intersects->left
));
570 intersects->right = MAX(right, intersects->right)(((right)>(intersects->right))?(right):(intersects->
right));
571 } else {
572 // Else just add a new entry;
573 kv_push(Rect, data->invalid_regions, ((Rect){top, bot, left, right}))do { if ((data->invalid_regions).size == (data->invalid_regions
).capacity) { (data->invalid_regions).capacity = (data->
invalid_regions).capacity? (data->invalid_regions).capacity
<<1 : 8; (data->invalid_regions).items = (Rect*)xrealloc
((data->invalid_regions).items, sizeof(Rect) * (data->invalid_regions
).capacity); } (data->invalid_regions).items[(data->invalid_regions
).size++] = (((Rect){top, bot, left, right})); } while (0);
574 }
575}

577static void update_size(UI *ui)
578{
579 TUIData *data = ui->data;
580 int width = 0, height = 0;
581 // 1 - try from a system call(ioctl/TIOCGWINSZ on unix)
582 if (!uv_tty_get_winsize(&data->output_handle, &width, &height)) {
583 goto end;
584 }
585
586 // 2 - use $LINES/$COLUMNS if available
587 const char *val;
588 int advance;
589 if ((val = os_getenv("LINES"))
590 && sscanf(val, "%d%n", &height, &advance) != EOF(-1) && advance
591 && (val = os_getenv("COLUMNS"))
592 && sscanf(val, "%d%n", &width, &advance) != EOF(-1) && advance) {
593 goto end;
594 }
595
596 // 3- read from terminfo if available
597 height = unibi_get_num(data->ut, unibi_lines);
598 width = unibi_get_num(data->ut, unibi_columns);
599
600end:
601 if (width <= 0 || height <= 0) {
602 // use a default of 80x24
603 width = 80;
604 height = 24;
605 }
606
607 ui->width = width;
608 ui->height = height;
609}

619static void unibi_out(UI *ui, int unibi_index, char *nvim_override)
620{
621 TUIData *data = ui->data;
622
623 const char *str = NULL((void*)0);
624
625 if (nvim_override) {
626 str = get_term_option(ui, nvim_override);
627 } else if (unibi_index >= 0) {
628 if (unibi_index < unibi_string_begin_) {
629 str = unibi_get_ext_str(data->ut, (unsigned)unibi_index);
630 } else {
631 str = unibi_get_str(data->ut, (unsigned)unibi_index);
632 }
633 }
634
635 if (str) {
636 data->bufpos += unibi_run(str, data->params, data->buf + data->bufpos,
637 sizeof(data->buf) - data->bufpos);
638 }
639}

656static void fix_terminfo(TUIData *data)
657{
658 unibi_term *ut = data->ut;
659
660 const char *term = os_getenv("TERM");
661 if (!term) {
662 goto end;
663 }
664
665#define STARTS_WITH(str, prefix)(!memcmp(str, prefix, sizeof(prefix) - 1)) (!memcmp(str, prefix, sizeof(prefix) - 1))
666
667 if (STARTS_WITH(term, "rxvt")(!memcmp(term, "rxvt", sizeof("rxvt") - 1))) {
668 unibi_set_if_empty(ut, unibi_exit_attribute_mode, "\x1b[m\x1b(B");
669 unibi_set_if_empty(ut, unibi_flash_screen, "\x1b[?5h$<20/>\x1b[?5l");
670 unibi_set_if_empty(ut, unibi_enter_italics_mode, "\x1b[3m");
671 } else if (STARTS_WITH(term, "screen")(!memcmp(term, "screen", sizeof("screen") - 1))) {
672 unibi_set_if_empty(ut, unibi_to_status_line, "\x1b_");
673 unibi_set_if_empty(ut, unibi_from_status_line, "\x1b\\");
674 }
675
676 if (STARTS_WITH(term, "xterm")(!memcmp(term, "xterm", sizeof("xterm") - 1)) || STARTS_WITH(term, "rxvt")(!memcmp(term, "rxvt", sizeof("rxvt") - 1))) {
677 unibi_set_if_empty(ut, unibi_cursor_normal, "\x1b[?12l\x1b[?25h");
678 unibi_set_if_empty(ut, unibi_cursor_invisible, "\x1b[?25l");
679 unibi_set_if_empty(ut, unibi_flash_screen, "\x1b[?5h$<100/>\x1b[?5l");
680 unibi_set_if_empty(ut, unibi_exit_attribute_mode, "\x1b(B\x1b[m");
681 unibi_set_if_empty(ut, unibi_change_scroll_region, "\x1b[%i%p1%d;%p2%dr");
682 unibi_set_if_empty(ut, unibi_clear_screen, "\x1b[H\x1b[2J");
683 unibi_set_if_empty(ut, unibi_to_status_line, "\x1b]2");
684 unibi_set_if_empty(ut, unibi_from_status_line, "\x07");
685 }
686
687end:
688 // Fill some empty slots with common terminal strings
689 data->unibi_ext.enable_mouse = unibi_add_ext_str(ut, NULL((void*)0),
690 "\x1b[?1002h\x1b[?1006h");
691 data->unibi_ext.disable_mouse = unibi_add_ext_str(ut, NULL((void*)0),
692 "\x1b[?1002l\x1b[?1006l");
693
694 unibi_set_if_empty(ut, unibi_cursor_address, "\x1b[%i%p1%d;%p2%dH");
695 unibi_set_if_empty(ut, unibi_exit_attribute_mode, "\x1b[0;10m");
696 unibi_set_if_empty(ut, unibi_set_a_foreground,
697 "\x1b[%?%p1%{8}%<%t3%p1%d%e%p1%{16}%<%t9%p1%{8}%-%d%e38;5;%p1%d%;m");
698 unibi_set_if_empty(ut, unibi_set_a_background,
699 "\x1b[%?%p1%{8}%<%t4%p1%d%e%p1%{16}%<%t10%p1%{8}%-%d%e48;5;%p1%d%;m");
700 unibi_set_if_empty(ut, unibi_enter_bold_mode, "\x1b[1m");
701 unibi_set_if_empty(ut, unibi_enter_underline_mode, "\x1b[4m");
702 unibi_set_if_empty(ut, unibi_enter_reverse_mode, "\x1b[7m");
703 unibi_set_if_empty(ut, unibi_bell, "\x07");
704 unibi_set_if_empty(data->ut, unibi_enter_ca_mode, "\x1b[?1049h");
705 unibi_set_if_empty(data->ut, unibi_exit_ca_mode, "\x1b[?1049l");
706 unibi_set_if_empty(ut, unibi_delete_line, "\x1b[M");
707 unibi_set_if_empty(ut, unibi_parm_delete_line, "\x1b[%p1%dM");
708 unibi_set_if_empty(ut, unibi_insert_line, "\x1b[L");
709 unibi_set_if_empty(ut, unibi_parm_insert_line, "\x1b[%p1%dL");
710 unibi_set_if_empty(ut, unibi_clear_screen, "\x1b[H\x1b[J");
711 unibi_set_if_empty(ut, unibi_clr_eol, "\x1b[K");
712 unibi_set_if_empty(ut, unibi_clr_eos, "\x1b[J");
713}

716{

720 buf.base = data->buf;
721 buf.len = data->bufpos;

728{
729 TUIData *data = ui->data;

731 char *rv = pmap_get(cstr_t)map_cstr_t_ptr_t_get(data->option_cache, option);

734 Object val = vim_get_option(cstr_as_string(option), &err);
735 if (val.type == kObjectTypeString) {
736 rv = val.data.string.data;
