1619static void process_hunk(diff_T **dpp, diff_T **dprevp, int idx_orig, int idx_new, diffhunk_T *hunk,
1620 bool_Bool *notsetp)
1621{
1622 diff_T *dp = *dpp;
1623 diff_T *dprev = *dprevp;
1624
1625 // Go over blocks before the change, for which orig and new are equal.
1626 // Copy blocks from orig to new.
1627 while (dp != NULL((void*)0)
1628 && hunk->lnum_orig > dp->df_lnum[idx_orig] + dp->df_count[idx_orig]) {
1629 if (*notsetp) {
1630 diff_copy_entry(dprev, dp, idx_orig, idx_new);
1631 }
1632 dprev = dp;
1633 dp = dp->df_next;
1634 *notsetp = true1;
1635 }
1636
1637 if ((dp != NULL((void*)0))
1638 && (hunk->lnum_orig <= dp->df_lnum[idx_orig] + dp->df_count[idx_orig])
1639 && (hunk->lnum_orig + hunk->count_orig >= dp->df_lnum[idx_orig])) {
1640 // New block overlaps with existing block(s).
1641 // First find last block that overlaps.
1642 diff_T *dpl;
1643 for (dpl = dp; dpl->df_next != NULL((void*)0); dpl = dpl->df_next) {
1644 if (hunk->lnum_orig + hunk->count_orig < dpl->df_next->df_lnum[idx_orig]) {
1645 break;
1646 }
1647 }
1648
1649 // If the newly found block starts before the old one, set the
1650 // start back a number of lines.
1651 linenr_T off = dp->df_lnum[idx_orig] - hunk->lnum_orig;
1652
1653 if (off > 0) {
1654 for (int i = idx_orig; i < idx_new; i++) {
1655 if (curtab->tp_diffbuf[i] != NULL((void*)0)) {
1656 dp->df_lnum[i] -= off;
1657 }
1658 }
1659 dp->df_lnum[idx_new] = hunk->lnum_new;
1660 dp->df_count[idx_new] = (linenr_T)hunk->count_new;
1661 } else if (*notsetp) {
1662 // new block inside existing one, adjust new block
1663 dp->df_lnum[idx_new] = hunk->lnum_new + off;
1664 dp->df_count[idx_new] = (linenr_T)hunk->count_new - off;
1665 } else {
1666 // second overlap of new block with existing block
1667 dp->df_count[idx_new] += (linenr_T)hunk->count_new - (linenr_T)hunk->count_orig
1668 + dpl->df_lnum[idx_orig] +
1669 dpl->df_count[idx_orig]
1670 - (dp->df_lnum[idx_orig] +
1671 dp->df_count[idx_orig]);
1672 }
1673
1674 // Adjust the size of the block to include all the lines to the
1675 // end of the existing block or the new diff, whatever ends last.
1676 off = (hunk->lnum_orig + (linenr_T)hunk->count_orig)
1677 - (dpl->df_lnum[idx_orig] + dpl->df_count[idx_orig]);
1678
1679 if (off < 0) {
1680 // new change ends in existing block, adjust the end if not
1681 // done already
1682 if (*notsetp) {
1683 dp->df_count[idx_new] += -off;
1684 }
1685 off = 0;
1686 }
1687
1688 for (int i = idx_orig; i < idx_new; i++) {
1689 if (curtab->tp_diffbuf[i] != NULL((void*)0)) {
1690 dp->df_count[i] = dpl->df_lnum[i] + dpl->df_count[i]
1691 - dp->df_lnum[i] + off;
1692 }
1693 }
1694
1695 // Delete the diff blocks that have been merged into one.
1696 diff_T *dn = dp->df_next;
1697 dp->df_next = dpl->df_next;
1698
1699 while (dn != dp->df_next) {
1700 dpl = dn->df_next;
1701 xfree(dn);
1702 dn = dpl;
1703 }
1704 } else {
1705 // Allocate a new diffblock.
1706 dp = diff_alloc_new(curtab, dprev, dp);
1707
1708 dp->df_lnum[idx_orig] = hunk->lnum_orig;
1709 dp->df_count[idx_orig] = (linenr_T)hunk->count_orig;
1710 dp->df_lnum[idx_new] = hunk->lnum_new;
1711 dp->df_count[idx_new] = (linenr_T)hunk->count_new;
1712
1713 // Set values for other buffers, these must be equal to the
1714 // original buffer, otherwise there would have been a change
1715 // already.
1716 for (int i = idx_orig + 1; i < idx_new; i++) {
1717 if (curtab->tp_diffbuf[i] != NULL((void*)0)) {
1718 diff_copy_entry(dprev, dp, idx_orig, i);
1719 }
1720 }
1721 }
