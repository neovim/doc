965void unibi_format(
966 unibi_var_t var_dyn[26],
967 unibi_var_t var_static[26],
968 const char *fmt,
969 unibi_var_t param[9],
970 void (*out)(void *, const char *, size_t),
971 void *ctx1,
972 void (*pad)(void *, size_t, int, int),
973 void *ctx2
974) {
975 const unibi_var_t zero = {0};
976 unibi_var_t stack[123] = {{0}};
977 size_t sp = 0;
978
979#define UC(F, C) (F((unsigned char)(C)))
980
981#define POP() (sp ? stack[--sp] : zero)
982#define PUSH(X) do { if (sp < COUNTOF(stack)(sizeof (stack) / sizeof *(stack))) { stack[sp++] = (X); } } while (0)
983#define PUSHi(N) do { unibi_var_t tmp_; tmp_.i = (N); PUSH(tmp_); } while (0)
984
985 while (*fmt) {
986 {
987 size_t r = strcspn(fmt, "%$");
988 if (r) {
989 out(ctx1, fmt, r);
990 fmt += r;
991 if (!*fmt) {
992 break;
993 }
994 }
995 }
996
997 if (*fmt == '$') {
998 ++fmt;
999 if (*fmt == '<' && UC(isdigit, fmt[1])) {
1000 int scale = 0, force = 0;
1001 const char *v = fmt + 1;
1002 size_t n = cstrtol(v, &v);
1003 n *= 10;
1004 if (*v == '.') {
1005 ++v;
1006 }
1007 if (UC(isdigit, *v)) {
1008 n += *v++ - '0';
1009 }
1010 if (*v == '/') {
1011 ++v;
1012 force = 1;
1013 if (*v == '*') {
1014 ++v;
1015 scale = 1;
1016 }
1017 } else if (*v == '*') {
1018 ++v;
1019 scale = 1;
1020 if (*v == '/') {
1021 ++v;
1022 force = 1;
1023 }
1024 }
1025 if (*v == '>') {
1026 fmt = v + 1;
1027 if (pad) {
1028 pad(ctx2, n, scale, force);
1029 }
1030 } else {
1031 out(ctx1, fmt - 1, 1);
1032 }
1033 } else {
1034 out(ctx1, fmt - 1, 1);
1035 }
1036 continue;
1037 }
1038
1039 assert(*fmt == '%')((*fmt == '%') ? (void) (0) : __assert_fail ("*fmt == '%'", "unibilium.c"
, 1039, __PRETTY_FUNCTION__));
1040 ++fmt;
1041
1042 if (UC(isdigit, *fmt) || (*fmt && strchr(":# .doxX", *fmt))) {
1043 enum {
1044 FlagAlt = 1,
1045 FlagSpc = 2,
1046 FlagSgn = 4,
1047 FlagLft = 8,
1048 FlagZro = 16
1049 };
1050 int flags = 0, width = -1, prec = -1;
1051 const char *v = fmt;
1052 if (*v == ':') {
1053 ++v;
1054 }
1055 while (1) {
1056 switch (*v++) {
1057 case '#': flags |= FlagAlt; continue;
1058 case ' ': flags |= FlagSpc; continue;
1059 case '0': flags |= FlagZro; continue;
1060 case '+': flags |= FlagSgn; continue;
1061 case '-': flags |= FlagLft; continue;
1062 }
1063 --v;
1064 break;
1065 }
1066 if (UC(isdigit, *v)) {
1067 width = cstrtol(v, &v);
1068 }
1069 if (*v == '.' && UC(isdigit, v[1])) {
1070 ++v;
1071 prec = cstrtol(v, &v);
1072 }
1073 if (*v && strchr("doxXs", *v)) {
1074 char gen[sizeof "%# +-0*.*d"], *g = gen;
1075 *g++ = '%';
1076 if (flags & FlagAlt) { *g++ = '#'; }
1077 if (flags & FlagSpc) { *g++ = ' '; }
1078 if (flags & FlagSgn) { *g++ = '+'; }
1079 if (flags & FlagLft) { *g++ = '-'; }
1080 if (flags & FlagZro) { *g++ = '0'; }
1081 if (width != -1) { *g++ = '*'; }
1082 if (prec != -1) { *g++ = '.'; *g++ = '*'; }
1083 *g++ = *v;
1084 *g = '\0';
1085 dput(*v, gen, width, prec, POP(), out, ctx1);
1086 fmt = v;
1087 } else {
1088 out(ctx1, fmt - 1, 2);
1089 }
1090 ++fmt;
1091 continue;
1092 }
1093
1094 switch (*fmt++) {
1095 default:
1096 out(ctx1, fmt - 2, 2);
1097 break;
1098
1099 case '\0':
1100 --fmt;
1101 out(ctx1, "%", 1);
1102 break;
1103
1104 case '%':
1105 out(ctx1, "%", 1);
1106 break;
1107
1108 case 'c': {
1109 unsigned char c;
1110 c = POP().i;
1111 out(ctx1, (char *)&c, 1);
1112 break;
1113 }
1114
1115 case 's': {
1116 char *s;
1117 s = POP().p;
1118 out(ctx1, s, strlen(s));
1119 break;
1120 }
1121
1122 case 'p':
1123 if (*fmt >= '1' && *fmt <= '9') {
1124 size_t n = *fmt++ - '1';
1125 PUSH(param[n]);
1126 } else {
1127 out(ctx1, fmt - 2, 2);
1128 }
1129 break;
1130
1131 case 'P':
1132 if (*fmt >= 'a' && *fmt <= 'z') {
1133 ++fmt;
1134 var_dyn[*fmt - 'a'] = POP();
1135 } else if (*fmt >= 'A' && *fmt <= 'Z') {
1136 ++fmt;
1137 var_static[*fmt - 'A'] = POP();
1138 } else {
1139 out(ctx1, fmt - 2, 2);
1140 }
1141 break;
1142
1143 case 'g':
1144 if (*fmt >= 'a' && *fmt <= 'z') {
1145 ++fmt;
1146 PUSH(var_dyn[*fmt - 'a']);
1147 } else if (*fmt >= 'A' && *fmt <= 'Z') {
1148 ++fmt;
1149 PUSH(var_static[*fmt - 'A']);
1150 } else {
1151 out(ctx1, fmt - 2, 2);
1152 }
1153 break;
1154
1155 case '\'':
1156 if (*fmt && fmt[1] == '\'') {
1157 PUSHi((unsigned char)*fmt);
1158 fmt += 2;
1159 } else {
1160 out(ctx1, fmt - 2, 2);
1161 }
1162 break;
1163
1164 case '{': {
1165 size_t r = strspn(fmt, "0123456789");
1166 if (r && fmt[r] == '}') {
1167 PUSHi(atoi(fmt));
1168 fmt += r + 1;
1169 } else {
1170 out(ctx1, fmt - 2, 2);
1171 }
1172 break;
1173 }
1174
1175 case 'l':
1176 PUSHi(strlen(POP().p));
1177 break;
1178
1179 case 'i':
1180 ++param[0].i;
1181 ++param[1].i;
1182 break;
1183
1184 case '?':
1185 break;
1186
1187 case 't': {
1188 int c = POP().i;
1189 if (!c) {
1190 size_t nesting = 0;
1191 for (; *fmt; ++fmt) {
1192 if (*fmt == '%') {
1193 ++fmt;
1194 if (*fmt == '?') {
1195 ++nesting;
1196 } else if (*fmt == ';' || *fmt == 'e') {
1197 if (!nesting) {
1198 ++fmt;
1199 break;
1200 }
1201 --nesting;
1202 } else if (!*fmt) {
1203 break;
1204 }
1205 }
1206 }
1207 }
1208 break;
1209 }
1210
1211 case 'e': {
1212 size_t nesting = 0;
1213 for (; *fmt; ++fmt) {
1214 if (*fmt == '%') {
1215 ++fmt;
1216 if (*fmt == '?') {
1217 ++nesting;
1218 } else if (*fmt == ';') {
1219 if (!nesting) {
1220 ++fmt;
1221 break;
1222 }
1223 --nesting;
1224 } else if (!*fmt) {
1225 break;
1226 }
1227 }
1228 }
1229 break;
1230 }
1231
1232 case ';':
1233 break;
1234
1235#define ARITH2(C, O) \
1236 case (C): { \
1237 unibi_var_t x, y; \
1238 y = POP(); \
1239 x = POP(); \
1240 PUSHi(x.i O y.i); \
1241 } break
1242
1243 ARITH2('+', +);
1244 ARITH2('-', -);
1245 ARITH2('*', *);
1246 ARITH2('/', /);
1247 ARITH2('m', %);
1248 ARITH2('&', &);
1249 ARITH2('|', |);
1250 ARITH2('^', ^);
1251 ARITH2('=', ==);
1252 ARITH2('<', <);
1253 ARITH2('>', >);
1254 ARITH2('A', &&);
1255 ARITH2('O', ||);
1256
1257#undef ARITH2
1258
1259#define ARITH1(C, O) \
1260 case (C): \
1261 PUSHi(O POP().i); \
1262 break
1263
1264 ARITH1('!', !);
1265 ARITH1('~', ~);
1266
1267#undef ARITH1
1268 }
1269 }
1270
1271#undef PUSHi
1272#undef PUSH
1273#undef POP
1274
1275#undef UC
1276}
