13/*
14 * The buffer list is a double linked list of all buffers.
15 * Each buffer can be in one of these states:
16 * never loaded: BF_NEVERLOADED is set, only the file name is valid
17 * not loaded: b_ml.ml_mfp == NULL, no memfile allocated
18 * hidden: b_nwindows == 0, loaded but not displayed in a window
19 * normal: loaded and displayed in a window
20 *
21 * Instead of storing file names all over the place, each file name is
22 * stored in the buffer list. It can be referenced by a number.

24 * The current implementation remembers all file names ever used.

1681 * Return fnum of the found buffer.
1682 * Return < 0 for error.
1683 */

1685buflist_findpat (
1686 char_u *pattern,
1687 char_u *pattern_end, /* pointer to first char after pattern */
1688 int unlisted, /* find unlisted buffers */
1689 int diffmode, /* find diff-mode buffers only */
1690 int curtab_only /* find buffers in current tab only */
1691)
1692{

1694 regprog_T *prog;
1695 int match = -1;

1698 char_u *patend;

1702
1703 if (pattern_end == pattern + 1 && (*pattern == '%' || *pattern == '#')) {
1704 if (*pattern == '%')
1705 match = curbuf->b_fnum;
1706 else
1707 match = curwin->w_alt_fnum;

1709 match = -1;
1710 }
1711 /*
1712 * Try four ways of matching a listed buffer:
1713 * attempt == 0: without '^' or '$' (at any position)
1714 * attempt == 1: with '^' at start (only at position 0)
1715 * attempt == 2: with '$' at end (only match at end)
1716 * attempt == 3: with '^' at start and '$' at end (only full match)
1717 * Repeat this for finding an unlisted buffer if there was no matching
1718 * listed buffer.
1719 */
1720 else {

1722 if (pat == NULL((void*)0))

1728 * is TRUE, try finding an unlisted buffer. */
1729 find_listed = TRUE1;
1730 for (;; ) {
1731 for (attempt = 0; attempt <= 3; ++attempt) {
1732 /* may add '^' and '$' */
1733 if (toggledollar)
1734 *patend = (attempt < 2) ? NUL'\000' : '$'; /* add/remove '$' */
1735 p = pat;
1736 if (*p == '^' && !(attempt & 1)) /* add/remove '^' */
1737 ++p;
1738 prog = vim_regcomp(p, p_magic ? RE_MAGIC1 : 0);
1739 if (prog == NULL((void*)0)) {
1740 free(pat);
1741 return -1;
1742 }
1743
1744 for (buf = firstbuf; buf != NULL((void*)0); buf = buf->b_next)
1745 if (buf->b_p_bl == find_listed
1746 && (!diffmode || diff_mode_buf(buf))
1747 && buflist_match(prog, buf) != NULL((void*)0)) {

1750 * the current tab. */

1754 if (wp->w_buffer == buf)
1755 break;
1756 if (wp == NULL((void*)0))
1757 continue;
1758 }
1759 if (match >= 0) { /* already found a match */
1760 match = -2;
1761 break;
1762 }
1763 match = buf->b_fnum; /* remember first match */
1764 }
1765
1766 vim_regfree(prog);
1767 if (match >= 0) /* found one match */
1768 break;
1769 }
1770
1771 /* Only search for unlisted buffers if there was no match with
1772 * a listed buffer. */
1773 if (!unlisted || !find_listed || match != -1)
1774 break;
1775 find_listed = FALSE0;
1776 }
1777
1778 free(pat);
1779 }
1780
1781 if (match == -2)
1782 EMSG2(_("E93: More than one match for %s"), pattern)emsg2((char_u *)(gettext((char *)("E93: More than one match for %s"
))), (char_u *)(pattern));
1783 else if (match < 0)
1784 EMSG2(_("E94: No matching buffer for %s"), pattern)emsg2((char_u *)(gettext((char *)("E94: No matching buffer for %s"
))), (char_u *)(pattern));
1785 return match;
1786}

1836 if (!buf->b_p_bl) /* skip unlisted buffers */
1837 continue;
1838 p = buflist_match(prog, buf);
1839 if (p != NULL((void*)0)) {
1840 if (round == 1)
1841 ++count;
1842 else {
1843 if (options & WILD_HOME_REPLACE2)
1844 p = home_replace_save(buf, p);
1845 else
1846 p = vim_strsave(p);
1847 (*file)[count++] = p;
1848 }
1849 }
1850 }
1851 if (count == 0) /* no match found, break here */
1852 break;
1853 if (round == 1) {
1854 *file = xmalloc(count * sizeof(**file));
1855 }
1856 }
1857 vim_regfree(prog);
1858 if (count) /* match(es) found, break here */
1859 break;
1860 }
1861
1862 if (patc != pat)
1863 free(patc);
1864
1865 *num_file = count;
1866 return count == 0 ? FAIL0 : OK1;
