1138 * literally ("literally" TRUE) or interpret is as typed characters.

1147 * stuff K_SPECIAL to get the effect of a special key when "literally"

1154 ++arg;

1156 stuffReadbuffLen(start, (long)(arg - start));

1160 if (has_mbyte)
1161 c = mb_cptr2char_adv(&arg);

1163 c = *arg++;

1165 stuffcharReadbuff(Ctrl_V22);
1166 stuffcharReadbuff(c);
1167 }

4837///
4838/// When the register is not empty, the string is appended.
4839///

4873 * Count the number of lines within the string
4874 */

4885 --newlines; /* uncount newline when appending first line */

4890 * If the register was not empty, move the existing lines to the new array.

4902 for (start = 0; start < len + extraline; start += i + 1) {
4903 for (i = start; i < len; ++i) /* find the end of the line */

4925 if (*s == NUL'\000')

4927 ++s;
4928 }

4933 if (type == MBLOCK2)

4945/*
4946 * Count the number of bytes, characters and "words" in a line.
4947 *
4948 * "Words" are counted by looking for boundaries between non-space and
4949 * space characters. (it seems to produce results that match 'wc'.)
4950 *
4951 * Return value is byte count; word count for the line is added to "*wc".
4952 * Char count is added to "*cc".
4953 *
4954 * The function will only examine the first "limit" characters in the
4955 * line, stopping if it encounters an end-of-line (NUL byte). In that
4956 * case, eol_size will be added to the character count to account for
4957 * the size of the EOL character.
4958 */
4959static long line_count_info(char_u *line, long *wc, long *cc, long limit, int eol_size)
4960{

4962 long words = 0;
4963 long chars = 0;
4964 int is_word = 0;
4965
4966 for (i = 0; i < limit && line[i] != NUL'\000'; ) {
4967 if (is_word) {
4968 if (vim_isspace(line[i])) {
4969 words++;
4970 is_word = 0;
4971 }
4972 } else if (!vim_isspace(line[i]))
4973 is_word = 1;
4974 ++chars;
4975 i += (*mb_ptr2len)(line + i);
4976 }
4977
4978 if (is_word)
4979 words++;
4980 *wc += words;
4981
4982 /* Add eol_size if the end of line was reached before hitting limit. */
4983 if (i < limit && line[i] == NUL'\000') {
4984 i += eol_size;
4985 chars += eol_size;
4986 }
4987 *cc += chars;
4988 return i;
4989}
