1352/*
1353 * Do Arabic shaping on character "c". Returns the shaped character.
1354 * out: "ccp" points to the first byte of the character to be shaped.
1355 * in/out: "c1p" points to the first composing char for "c".
1356 * in: "prev_c" is the previous character (not shaped)
1357 * in: "prev_c1" is the first composing char for the previous char
1358 * (not shaped)
1359 * in: "next_c" is the next character (not shaped).
1360 */
1361int arabic_shape(int c, int *ccp, int *c1p, int prev_c, int prev_c1,
1362 int next_c)
1363{
1364 int curr_c;
1365 int shape_c;
1366 int curr_laa;
1367 int prev_laa;
1368
1369 /* Deal only with Arabic character, pass back all others */
1370 if (!A_is_ok(c)) {
1371 return c;
1372 }
1373
1374 /* half-shape current and previous character */
1375 shape_c = half_shape(prev_c);
1376
1377 /* Save away current character */
1378 curr_c = c;
Value stored to 'curr_c' is never read
1379
1380 curr_laa = A_firstc_laa(c, *c1p);
1381 prev_laa = A_firstc_laa(prev_c, prev_c1);
1382
1383 if (curr_laa) {
1384 if (A_is_valid(prev_c) && !A_is_f(shape_c) && !A_is_s(shape_c) &&
1385 !prev_laa) {
1386 curr_c = chg_c_laa2f(curr_laa);
1387 } else {
1388 curr_c = chg_c_laa2i(curr_laa);
1389 }
1390
1391 /* Remove the composing character */
1392 *c1p = 0;
1393 } else if (!A_is_valid(prev_c) && A_is_valid(next_c)) {
1394 curr_c = chg_c_a2i(c);
1395 } else if (!shape_c || A_is_f(shape_c) || A_is_s(shape_c) || prev_laa) {
1396 curr_c = A_is_valid(next_c) ? chg_c_a2i(c) : chg_c_a2s(c);
1397 } else if (A_is_valid(next_c)) {
1398 curr_c = A_is_iso(c) ? chg_c_a2m(c) : chg_c_i2m(c);
1399 } else if (A_is_valid(prev_c)) {
1400 curr_c = chg_c_a2f(c);
1401 } else {
1402 curr_c = chg_c_a2s(c);
1403 }
1404
1405 /* Sanity check -- curr_c should, in the future, never be 0.
1406 * We should, in the future, insert a fatal error here. */
1407 if (curr_c == NUL'\000') {
1408 curr_c = c;
1409 }
1410
1411 if ((curr_c != c) && (ccp != NULL((void*)0))) {
1412 char_u buf[MB_MAXBYTES21 + 1];
1413
1414 /* Update the first byte of the character. */
1415 (*mb_char2bytes)(curr_c, buf);
1416 *ccp = buf[0];
1417 }
1418
1419 /* Return the shaped character */
1420 return curr_c;
1421}
