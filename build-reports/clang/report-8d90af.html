1070/*
1071 * Exchange current and next window
1072 */
1073static void win_exchange(long Prenum)
1074{
1075 frame_T *frp;
1076 frame_T *frp2;
1077 win_T *wp;
1078 win_T *wp2;
1079 int temp;
1080
1081 if (lastwin == firstwin) { /* just one window */
1082 beep_flush();
1083 return;
1084 }
1085
1086
1087 /*
1088 * find window to exchange with
1089 */
1090 if (Prenum) {
1091 frp = curwin->w_frame->fr_parent->fr_child;
1092 while (frp != NULL((void*)0) && --Prenum > 0)
1093 frp = frp->fr_next;
1094 } else if (curwin->w_frame->fr_next != NULL((void*)0)) /* Swap with next */
1095 frp = curwin->w_frame->fr_next;
1096 else /* Swap last window in row/col with previous */
1097 frp = curwin->w_frame->fr_prev;
1098
1099 /* We can only exchange a window with another window, not with a frame
1100 * containing windows. */
1101 if (frp == NULL((void*)0) || frp->fr_win == NULL((void*)0) || frp->fr_win == curwin)
1102 return;
1103 wp = frp->fr_win;
1104
1105 /*
1106 * 1. remove curwin from the list. Remember after which window it was in wp2
1107 * 2. insert curwin before wp in the list
1108 * if wp != wp2
1109 * 3. remove wp from the list
1110 * 4. insert wp after wp2
1111 * 5. exchange the status line height and vsep width.
1112 */
1113 wp2 = curwin->w_prev;
1114 frp2 = curwin->w_frame->fr_prev;
1115 if (wp->w_prev != curwin) {
1116 win_remove(curwin, NULL((void*)0));
1117 frame_remove(curwin->w_frame);
1118 win_append(wp->w_prev, curwin);
1119 frame_insert(frp, curwin->w_frame);
1120 }
1121 if (wp != wp2) {
1122 win_remove(wp, NULL((void*)0));
1123 frame_remove(wp->w_frame);
1124 win_append(wp2, wp);
1125 if (frp2 == NULL((void*)0))
1126 frame_insert(wp->w_frame->fr_parent->fr_child, wp->w_frame);
1127 else
1128 frame_append(frp2, wp->w_frame);
1129 }
1130 temp = curwin->w_status_height;
1131 curwin->w_status_height = wp->w_status_height;
1132 wp->w_status_height = temp;
1133 temp = curwin->w_vsep_width;
1134 curwin->w_vsep_width = wp->w_vsep_width;
1135 wp->w_vsep_width = temp;
1136
1137 /* If the windows are not in the same frame, exchange the sizes to avoid
1138 * messing up the window layout. Otherwise fix the frame sizes. */
1139 if (curwin->w_frame->fr_parent != wp->w_frame->fr_parent) {
1140 temp = curwin->w_height;
1141 curwin->w_height = wp->w_height;
1142 wp->w_height = temp;
1143 temp = curwin->w_width;
1144 curwin->w_width = wp->w_width;
1145 wp->w_width = temp;
1146 } else {
1147 frame_fix_height(curwin);
1148 frame_fix_height(wp);
1149 frame_fix_width(curwin);
1150 frame_fix_width(wp);
1151 }
1152
1153 (void)win_comp_pos(); /* recompute window positions */
1154
1155 win_enter(wp, true1);
1156 redraw_later(CLEAR50);
1157}

1159/*
1160 * rotate windows: if upwards TRUE the second window becomes the first one
1161 * if upwards FALSE the first window becomes the second one
1162 */
1163static void win_rotate(int upwards, int count)
1164{
1165 win_T *wp1;
1166 win_T *wp2;
1167 frame_T *frp;
1168 int n;
1169
1170 if (firstwin == lastwin) { /* nothing to do */
1171 beep_flush();
1172 return;
1173 }
1174
1175
1176 /* Check if all frames in this row/col have one window. */
1177 for (frp = curwin->w_frame->fr_parent->fr_child; frp != NULL((void*)0);
1178 frp = frp->fr_next)
1179 if (frp->fr_win == NULL((void*)0)) {
1180 EMSG(_("E443: Cannot rotate when another window is split"))emsg((char_u *)(gettext((char *)("E443: Cannot rotate when another window is split"
))));
1181 return;
1182 }
1183
1184 while (count--) {
1185 if (upwards) { /* first window becomes last window */
1186 /* remove first window/frame from the list */
1187 frp = curwin->w_frame->fr_parent->fr_child;
1188 wp1 = frp->fr_win;
1189 win_remove(wp1, NULL((void*)0));
1190 frame_remove(frp);
1191
1192 /* find last frame and append removed window/frame after it */
1193 for (; frp->fr_next != NULL((void*)0); frp = frp->fr_next)
1194 ;
1195 win_append(frp->fr_win, wp1);
1196 frame_append(frp, wp1->w_frame);
1197
1198 wp2 = frp->fr_win; /* previously last window */
1199 } else { /* last window becomes first window */
1200 /* find last window/frame in the list and remove it */
1201 for (frp = curwin->w_frame; frp->fr_next != NULL((void*)0);
1202 frp = frp->fr_next)
1203 ;
1204 wp1 = frp->fr_win;
1205 wp2 = wp1->w_prev; /* will become last window */
1206 win_remove(wp1, NULL((void*)0));
1207 frame_remove(frp);
1208
1209 /* append the removed window/frame before the first in the list */
1210 win_append(frp->fr_parent->fr_child->fr_win->w_prev, wp1);
1211 frame_insert(frp->fr_parent->fr_child, frp);
1212 }
1213
1214 /* exchange status height and vsep width of old and new last window */
1215 n = wp2->w_status_height;
1216 wp2->w_status_height = wp1->w_status_height;
1217 wp1->w_status_height = n;
1218 frame_fix_height(wp1);
1219 frame_fix_height(wp2);
1220 n = wp2->w_vsep_width;
1221 wp2->w_vsep_width = wp1->w_vsep_width;
1222 wp1->w_vsep_width = n;
1223 frame_fix_width(wp1);
1224 frame_fix_width(wp2);
1225
1226 /* recompute w_winrow and w_wincol for all windows */
1227 (void)win_comp_pos();
1228 }
1229
1230 redraw_later(CLEAR50);
1231}
